21  /**
22   * Utilities to convert JCA types to VData types.
23   *
24   * @author carcassi
25   */
26  class DataUtils {
27  
28      private static final Logger log = Logger.getLogger(DataUtils.class.getName());
29  
30      /**
31       * Constant to convert epics seconds to UNIX seconds. It counts the number
32       * of seconds for 20 years, 5 of which leap years. It does _not_ count the
33       * number of leap seconds (which should have been 15).
34       */
35      static long TS_EPOCH_SEC_PAST_1970=631152000; //7305*86400;
36      
37      /**
38       * Converts a JCA timestamp to an epics.util timestamp.
39       * 
40       * @param epicsTimeStamp the epics timestamp
41       * @return a new epics.util timestamp
42       */
43      static org.epics.util.time.Timestamp timestampOf(gov.aps.jca.dbr.TimeStamp epicsTimeStamp) {
44          if (epicsTimeStamp == null)
45              return null;
46          
47          return org.epics.util.time.Timestamp.of(epicsTimeStamp.secPastEpoch() + TS_EPOCH_SEC_PAST_1970, 0)
48                  .plus(TimeDuration.ofNanos(epicsTimeStamp.nsec()));
49      }
50  
51      /**
52       * Converts an alarm severity from JCA to VData.
53       *
54       * @param severity the JCA severity
55       * @return the VData severity
56       */
57      static AlarmSeverity fromEpics(Severity severity) {
58          if (Severity.NO_ALARM.isEqualTo(severity)) {
59              return AlarmSeverity.NONE;
60          } else if (Severity.MINOR_ALARM.isEqualTo(severity)) {
61              return AlarmSeverity.MINOR;
62          } else if (Severity.MAJOR_ALARM.isEqualTo(severity)) {
63              return AlarmSeverity.MAJOR;
64          } else if (Severity.INVALID_ALARM.isEqualTo(severity)) {
65              return AlarmSeverity.INVALID;
66          } else {
67              return AlarmSeverity.UNDEFINED;
68          }
69      }
70  
71      // Creates the list of EPICS 3 status by
72      // iterating over all Status defined in JCA
73      private static final List<String> epicsPossibleStatus;
74      static {
75          List<String> mutableList = new ArrayList<String>();
76          Status aStatus = Status.forValue(0);
77          while (aStatus != null) {
78              mutableList.add(aStatus.getName());
79              aStatus = Status.forValue(aStatus.getValue() + 1);
80          }
81          epicsPossibleStatus = Collections.unmodifiableList(mutableList);
82      }
83  
84      /**
85       * Returns all the possible EPICS 3 status.
86       *
87       * @return all possible status
88       */
89      static List<String> epicsPossibleStatus() {
90          return epicsPossibleStatus;
91      }
92  
93      /**
94       * Determines whether the timestamp represents good data or not. It
95       * checks whether the seconds are either a UNIX 0 or a Epics 0.
96       *
97       * @param timeStamp a timestamp
98       * @return false if timeStamp is null or represents a UNIX 0 or an Epics 0
99       */
100     static boolean isTimeValid(TimeStamp timeStamp) {
101         if (timeStamp == null)
102             return false;
103         
104         long sec = timeStamp.secPastEpoch() + TS_EPOCH_SEC_PAST_1970;
105         if (sec == 0 || sec == TS_EPOCH_SEC_PAST_1970) {
106             return false;
107         }
108         
109         long nanosec = timeStamp.nsec();
110         if (nanosec < 0 || nanosec > 999_999_999) {
111             return false;
112         }
113         
114         return true;
115     }

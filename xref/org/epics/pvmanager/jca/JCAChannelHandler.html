36  /**
37   * A ChannelHandler for the JCADataSource.
38   * <p>
39   * NOTE: this class is extensible as per Bastian request so that DESY can hook
40   * a different type factory. This is a temporary measure until the problem
41   * is solved in better, more general way, so that data sources
42   * can work only with data source specific types, while allowing
43   * conversions to normalized type through operators. The contract of this
44   * class is, therefore, expected to change.
45   * <p>
46   * Related changes are marked so that they are not accidentally removed in the
47   * meantime, and can be intentionally removed when a better solution is implemented.
48   *
49   * @author carcassi
50   */

75      public JCAChannelHandler(String channelName, JCADataSource jcaDataSource) {
76          super(channelName);
77          setProcessMessageOnReconnect(false);
78          this.jcaDataSource = jcaDataSource;
79          
80          boolean longStringName = longStringPattern.matcher(channelName).matches();
81          
82          // Parse parameters
83          // Done here so that they can be immutable
84          Matcher matcher = hasOptions.matcher(getChannelName());
85          if (matcher.matches()) {
86              jcaChannelName = matcher.group(1);
87              String clientOptions = matcher.group(2);
88              // TODO: Hack, this should have a real JSON parser
89              switch (clientOptions) {
90                  case "{\"putCallback\":true}":
91                      putCallback = true;
92                      longString = longStringName;
93                      break;
94                  case "{\"putCallback\":false}":
95                      putCallback = false;
96                      longString = longStringName;
97                      break;
98                  case "{\"longString\":true}":
99                      putCallback = false;
100                     longString = true;
101                     break;
102                 case "{\"longString\":false}":
103                     putCallback = false;
104                     longString = false;
105                     break;
106                 default:
107                     throw new IllegalArgumentException("Option not recognized for " + getChannelName());
108             }
109         } else {
110             longString = longStringName;
111             putCallback = false;
112             jcaChannelName = channelName;
113         }
114     }

158     @Override
159     public synchronized void connect() {
160         needsMonitor = true;
161         needsAccessChangeListener.set(true);
162         
163         try {
164             // Give the listener right away so that no event gets lost
165 	    // If it's a large array, connect using lower priority
166 	    if (largeArray) {
167                 channel = jcaDataSource.getContext().createChannel(getJcaChannelName(), connectionListener, Channel.PRIORITY_MIN);
168 	    } else {
169                 channel = jcaDataSource.getContext().createChannel(getJcaChannelName(), connectionListener, (short) (Channel.PRIORITY_MIN + 1));
170 	    }
171         } catch (CAException ex) {
172             throw new RuntimeException("JCA Connection failed", ex);
173         }
174     }

176     private void putWithCallback(Object newValue, final ChannelWriteCallback callback) throws CAException {
177         PutListener listener = new PutListener() {
178 
179             @Override
180             public void putCompleted(PutEvent ev) {
181                 if (log.isLoggable(Level.FINEST)) {
182                     log.log(Level.FINEST, "JCA putCompleted for channel {0} event {1}", new Object[] {getChannelName(), ev});
183                 }
184                 
185                 if (ev.getStatus().isSuccessful()) {
186                     callback.channelWritten(null);
187                 } else {
188                     callback.channelWritten(new Exception(ev.toString()));
189                 }
190             }
191         };
192         // If it's a ListNumber, extract the array
193         if (newValue instanceof ListNumber) {
194             ListNumber data = (ListNumber) newValue;
195             Object wrappedArray = CollectionNumbers.wrappedArray(data);
196             if (wrappedArray == null) {
197                 newValue = CollectionNumbers.doubleArrayCopyOf(data);
198             } else {
199                 newValue = wrappedArray;
200             }
201         }
202         if (newValue instanceof String) {
203             if (isLongString()) {
204                 channel.put(toBytes(newValue.toString()), listener);
205             } else {
206                 if (channel.getFieldType().isBYTE() && channel.getElementCount() > 1) {
207                     log.warning("You are writing the String " + newValue + " to BYTE channel " + getChannelName() + ": use {\"longString\":true} for support");
208                     channel.put(toBytes(newValue.toString()), listener);
209                 } else {
210                     channel.put(newValue.toString(), listener);
211                 }
212             }
213         } else if (newValue instanceof byte[]) {
214             channel.put((byte[]) newValue, listener);
215         } else if (newValue instanceof short[]) {
216             channel.put((short[]) newValue, listener);
217         } else if (newValue instanceof int[]) {
218             channel.put((int[]) newValue, listener);
219         } else if (newValue instanceof float[]) {
220             channel.put((float[]) newValue, listener);
221         } else if (newValue instanceof double[]) {
222             channel.put((double[]) newValue, listener);
223         } else if (newValue instanceof Byte || newValue instanceof Short
224                 || newValue instanceof Integer || newValue instanceof Long) {
225             channel.put(((Number) newValue).longValue(), listener);
226         } else if (newValue instanceof Float || newValue instanceof Double) {
227             channel.put(((Number) newValue).doubleValue(), listener);
228         } else {
229             throw new RuntimeException("Unsupported type for CA: " + newValue.getClass());
230         }
231         jcaDataSource.getContext().flushIO();
232     }

234     private void put(Object newValue, final ChannelWriteCallback callback) throws CAException {
235         // If it's a ListNumber, extract the array
236         if (newValue instanceof ListNumber) {
237             ListNumber data = (ListNumber) newValue;
238             Object wrappedArray = CollectionNumbers.wrappedArray(data);
239             if (wrappedArray == null) {
240                 newValue = CollectionNumbers.doubleArrayCopyOf(data);
241             } else {
242                 newValue = wrappedArray;
243             }
244         }
245         if (newValue instanceof Double[]) {
246             log.warning("You are writing a Double[] to channel " + getChannelName() + ": use org.epics.util.array.ListDouble instead");
247             final Double dbl[] = (Double[]) newValue;
248             final double val[] = new double[dbl.length];
249             for (int i = 0; i < val.length; ++i) {
250                 val[i] = dbl[i].doubleValue();
251             }
252             newValue = val;
253         }
254         if (newValue instanceof Integer[]) {
255             log.warning("You are writing a Integer[] to channel " + getChannelName() + ": use org.epics.util.array.ListInt instead");
256             final Integer ival[] = (Integer[]) newValue;
257             final int val[] = new int[ival.length];
258             for (int i = 0; i < val.length; ++i) {
259                 val[i] = ival[i].intValue();
260             }
261             newValue = val;
262         }
263         
264         if (newValue instanceof String) {
265             if (isLongString()) {
266                 channel.put(toBytes(newValue.toString()));
267             } else {
268                 if (channel.getFieldType().isBYTE() && channel.getElementCount() > 1) {
269                     log.warning("You are writing the String " + newValue + " to BYTE channel " + getChannelName() + ": use {\"longString\":true} for support");
270                     channel.put(toBytes(newValue.toString()));
271                 } else {
272                     channel.put(newValue.toString());
273                 }
274             }
275         } else if (newValue instanceof byte[]) {
276             channel.put((byte[]) newValue);
277         } else if (newValue instanceof short[]) {
278             channel.put((short[]) newValue);
279         } else if (newValue instanceof int[]) {
280             channel.put((int[]) newValue);
281         } else if (newValue instanceof float[]) {
282             channel.put((float[]) newValue);
283         } else if (newValue instanceof double[]) {
284             channel.put((double[]) newValue);
285         } else if (newValue instanceof Byte || newValue instanceof Short
286                 || newValue instanceof Integer || newValue instanceof Long) {
287             channel.put(((Number) newValue).longValue());
288         } else if (newValue instanceof Float || newValue instanceof Double) {
289             channel.put(((Number) newValue).doubleValue());
290         } else {
291             callback.channelWritten(new Exception(new RuntimeException("Unsupported type for CA: " + newValue.getClass())));
292             return;
293         }
294         jcaDataSource.getContext().flushIO();
295         callback.channelWritten(null);
296     }

298     private void setup(Channel channel) throws CAException {
299         DBRType metaType = metadataFor(channel);
300         
301         // If metadata is needed, get it
302         if (metaType != null) {
303             // Need to use callback for the listener instead of doing a synchronous get
304             // (which seemed to perform better) because JCA (JNI implementation)
305             // would return an empty list of labels for the Enum metadata
306             channel.get(metaType, 1, new GetListener() {
307 
308                 @Override
309                 public void getCompleted(GetEvent ev) {
310                     synchronized(JCAChannelHandler.this) {
311                         if (log.isLoggable(Level.FINEST)) {
312                             log.log(Level.FINEST, "JCA metadata getCompleted for channel {0} event {1}", new Object[] {getChannelName(), ev});
313                         }
314                         
315                         // In case the metadata arrives after the monitor
316                         MonitorEvent event = null;
317                         if (getLastMessagePayload() != null) {
318                             event = getLastMessagePayload().getEvent();
319                         }
320                         processMessage(new JCAMessagePayload(ev.getDBR(), event));
321                     }
322                 }
323             });
324         }
325 
326         if (needsMonitor) {
327             // At each (re)connect, we need to create a new monitor:
328             // since the type could be changed, we would have a type mismatch
329             // between the current type and the old type when the monitor was
330             // created
331             
332             // XXX: Ideally, we would destroy the monitor on reconnect,
333             // but currently this does not work with CAJ (you get an
334             // IllegalStateException because the transport is not there
335             // anymore). So, for now, we destroy the monitor during the 
336             // the connection callback.
337             
338             // XXX: Ideally, we should just close (clear) the monitor, but
339             // this would cause one last event to reach the monitorListener.
340             // So, we remove the monitorListener right before the clear.
341             
342             // TODO: we could remember the previous type, and reconnect
343             // only if the type actually changed
344             if (valueMonitor != null) {
345                 valueMonitor.removeMonitorListener(monitorListener);
346                 valueMonitor.clear();
347                 valueMonitor = null;
348             }
349             
350             valueMonitor = channel.addMonitor(valueTypeFor(channel), countFor(channel), jcaDataSource.getMonitorMask(), monitorListener);
351             needsMonitor = false;
352         }
353 
354         // Remove current metadata monitor
355         if (metadataMonitor != null) {
356             metadataMonitor.removeMonitorListener(metadataListener);
357             metadataMonitor.clear();
358             metadataMonitor = null;
359         }
360         
361         // Setup metadata monitor if required
362         if (jcaDataSource.isDbePropertySupported() && metaType != null) {
363             metadataMonitor = channel.addMonitor(metaType, 1, Monitor.PROPERTY, metadataListener);
364         }
365 
366         // Flush the entire context (it's the best we can do)
367         channel.getContext().flushIO();
368     }

370     private final ConnectionListener connectionListener = new ConnectionListener() {
371 
372             @Override
373             public void connectionChanged(ConnectionEvent ev) {
374                 synchronized(JCAChannelHandler.this) {
375                     try {
376                         if (log.isLoggable(Level.FINEST)) {
377                             log.log(Level.FINEST, "JCA connectionChanged for channel {0} event {1}", new Object[] {getChannelName(), ev});
378                         }
379 
380                         // Take the channel from the event so that there is no
381                         // synchronization problem
382                         Channel channel = (Channel) ev.getSource();
383 
384                         // Check whether the channel is large and was opened
385                         // as large. Reconnect if does not match
386                         if (ev.isConnected() && channel.getElementCount() >= LARGE_ARRAY && !largeArray) {
387                             disconnect();
388                             largeArray = true;
389                             connect();
390                             return;
391                         }
392 
393                         processConnection(new JCAConnectionPayload(JCAChannelHandler.this, channel, getConnectionPayload()));
394                         if (ev.isConnected()) {
395                             // If connected, no write access and exception was not sent, notify writers
396                             if (!channel.getWriteAccess() && !sentReadOnlyException) {
397                                 reportExceptionToAllWriters(createReadOnlyException());
398                                 sentReadOnlyException = true;
399                             }
400                             
401                             // Setup monitors on connection
402                             setup(channel);
403                         } else {
404                             resetMessage();
405                             // Next connection, resend the read only exception if that's the case
406                             sentReadOnlyException = false;
407                             needsMonitor = true;
408                         }
409                         
410                     } catch (Exception ex) {
411                         reportExceptionToAllReadersAndWriters(ex);
412                     }
413                 }
414                 
415                 // XXX: because of the JNI implementation this section cannot
416                 // be part of the previous atomic section. The problem is that
417                 // adding the listener causes the listener to be called
418                 // right away on a different thread, and the addAccessRightsListener
419                 // seems to return only after the event is processed. This
420                 // means that you cannot serialize the addListener call
421                 // and the listener callback.
422                 // Since we have to make a choice between having either the add
423                 // or the callback properly synchronized, we choose the callback
424                 boolean addListener = needsAccessChangeListener.getAndSet(false);
425                 if (addListener) {
426                     try {
427                         Channel channel = (Channel) ev.getSource();
428                         channel.addAccessRightsListener(new AccessRightsListener() {
429 
430                             @Override
431                             public void accessRightsChanged(AccessRightsEvent ev) {
432                                 if (log.isLoggable(Level.FINEST)) {
433                                     log.log(Level.FINEST, "JCA accessRightsChanged for channel {0} event {1}", new Object[] {getChannelName(), ev});
434                                 }
435                                 
436                                 // Some JNI implementation lock if calling getState
437                                 // from within this callback. We context switch in that case
438                                 final Channel channel = (Channel) ev.getSource();
439                                 Runnable task = new Runnable() {
440 
441                                     @Override
442                                     public void run() {
443                                         synchronized(JCAChannelHandler.this) {
444                                             processConnection(new JCAConnectionPayload(JCAChannelHandler.this, channel, getConnectionPayload()));
445                                             if (!sentReadOnlyException && !channel.getWriteAccess()) {
446                                                 reportExceptionToAllWriters(createReadOnlyException());
447                                                 sentReadOnlyException = true;
448                                             }
449                                         }
450                                     }
451                                 };
452                                 if (jcaDataSource.useContextSwitchForAccessRightCallback()) {
453                                     jcaDataSource.getContextSwitch().submit(task);
454                                 } else {
455                                     task.run();
456                                 }
457                             }
458                         });
459                     } catch (Exception ex) {
460                         reportExceptionToAllReadersAndWriters(ex);
461                     }
462                 }
463             }
464         };;

466     private String toStringDBR(DBR value) {
467         StringBuilder builder = new StringBuilder();
468         if (value == null) {
469             return "null";
470         }
471         if (value.getValue() instanceof double[]) {
472             builder.append(Arrays.toString((double[]) value.getValue()));
473         } else if (value.getValue() instanceof short[]) {
474             builder.append(Arrays.toString((short[]) value.getValue()));
475         } else if (value.getValue() instanceof String[]) {
476             builder.append(Arrays.toString((String[]) value.getValue()));
477         } else {
478             builder.append(value.getValue().toString());
479         }
480         return builder.toString();
481     }

520     @Override
521     public synchronized void disconnect() {
522         try {
523             // Close the channel
524             // Need to guard because the channel may be closed if the
525             // context was already destroyed
526             if (channel.getConnectionState() != Channel.ConnectionState.CLOSED) {
527                 channel.removeConnectionListener(connectionListener);
528                 channel.destroy();
529             }
530         } catch (CAException ex) {
531             throw new RuntimeException("JCA Disconnect fail", ex);
532         } finally {
533             channel = null;
534             sentReadOnlyException = false;
535             processConnection(null);
536         }
537     }

539     @Override
540     public void write(Object newValue, final ChannelWriteCallback callback) {
541         try {
542             if (isPutCallback())
543                 putWithCallback(newValue, callback);
544             else
545                 put(newValue, callback);
546         } catch (CAException ex) {
547             callback.channelWritten(ex);
548         }
549     }

574     @Override
575     public synchronized Map<String, Object> getProperties() {
576         Map<String, Object> properties = new HashMap<String, Object>();
577         if (channel != null) {
578             properties.put("CA Channel name", channel.getName());
579             properties.put("CA Connection state", channel.getConnectionState().getName());
580             if (channel.getConnectionState() == Channel.ConnectionState.CONNECTED) {
581                 properties.put("CA Hostname", channel.getHostName());
582                 properties.put("CA Channel type", channel.getFieldType().getName());
583                 properties.put("CA Element count", channel.getElementCount());
584                 properties.put("CA Read access", channel.getReadAccess());
585                 properties.put("CA Write access", channel.getWriteAccess());
586             }
587             properties.put("isLongString", isLongString());
588             properties.put("isPutCallback", isPutCallback());
589             properties.put("Connected", isConnected());
590             properties.put("Write Connected", isWriteConnected());
591             properties.put("Connection payload", getConnectionPayload());
592             properties.put("Last message payload", getLastMessagePayload());
593         }
594         return properties;
595     }

597     protected DBRType metadataFor(Channel channel) {
598         DBRType type = channel.getFieldType();
599         
600         if (type.isBYTE() || type.isSHORT() || type.isINT() || type.isFLOAT() || type.isDOUBLE())
601             return DBR_CTRL_Double.TYPE;
602         
603         if (type.isENUM())
604             return DBR_LABELS_Enum.TYPE;
605         
606         return null;
607     }
608 
609     protected int countFor(Channel channel) {
610         if (channel.getElementCount() == 1)
611             return 1;
612         
613         if (jcaDataSource.isVarArraySupported())
614             return 0;
615         else
616             return channel.getElementCount();
617     }

619     static Pattern rtypeStringPattern = Pattern.compile(".+\\.RTYP.*");
620 
621     protected DBRType valueTypeFor(Channel channel) {
622         DBRType type = channel.getFieldType();
623         
624         if (type.isBYTE()) {
625             return DBR_TIME_Byte.TYPE;
626         } else if (type.isSHORT()) {
627             return DBR_TIME_Short.TYPE;
628         } else if (type.isINT()) {
629             return DBR_TIME_Int.TYPE;
630         } else if (type.isFLOAT()) {
631             return DBR_TIME_Float.TYPE;
632         } else if (type.isDOUBLE()) {
633             return DBR_TIME_Double.TYPE;
634         } else if (type.isENUM()) {
635             return DBR_TIME_Enum.TYPE;
636         } else if (type.isSTRING()) {
637             if (jcaDataSource.isRtypValueOnly() &&
638                     rtypeStringPattern.matcher(channel.getName()).matches()) {
639                 return DBR_String.TYPE;
640             }
641             return DBR_TIME_String.TYPE;
642         }
643         
644         throw new IllegalArgumentException("Unsupported type " + type);
645     }

647     /**
648      * Converts a String into byte array.
649      * 
650      * @param text the string to be converted
651      * @return byte array, always including '\0' termination
652      */
653     static byte[] toBytes(final String text) {
654         // TODO: it's unclear what encoding is used and how
655         
656         // Write string as byte array WITH '\0' TERMINATION!
657         final byte[] bytes = new byte[text.length() + 1];
658         System.arraycopy(text.getBytes(), 0, bytes, 0, text.length());
659         bytes[text.length()] = '\0';
660         return bytes;
661     }

663     /**
664      * Converts a byte array into a String. It
665      * 
666      * @param data the array to be converted
667      * @return the string
668      */
669     static String toString(byte[] data) {
670         int index = 0;
671         while (index < data.length && data[index] != '\0') {
672             index++;
673         }
674         
675         return new String(data, 0, index);
676     }

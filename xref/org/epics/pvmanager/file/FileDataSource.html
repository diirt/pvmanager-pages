24  /**
25   * Data source for locally written data. Each instance of this
26   * data source will have its own separate channels and values.
27   *
28   * @author carcassi
29   */
30  public final class FileDataSource extends DataSource {
31  
32      static {
33          // Install type support for the types it generates.
34          DataTypeSupport.install();
35      }
36  
37      /**
38       * Creates a new data source.
39       */
40      public FileDataSource() {
41          super(true);
42      }
43      
44      private final FileWatcherService fileWatchService =
45              new FileWatcherFileSystemService(Executors.newSingleThreadScheduledExecutor(org.epics.pvmanager.util.Executors.namedPool("diirt - file watch")),
46                      TimeDuration.ofSeconds(1.0));
47  
48      FileWatcherService getFileWatchService() {
49          return fileWatchService;
50      }
51      
52      @Override
53      protected ChannelHandler createChannel(String channelName) {
54          if (channelName.endsWith(".png") || channelName.endsWith(".bmp")) {
55              return new ImageChannelHandler(this, channelName, new File(URI.create("file://" + channelName)));
56          }
57          if (channelName.endsWith(".list")) {
58              return new ListChannelHandler(this, channelName, new File(URI.create("file://" + channelName)));
59          }
60          return new FileChannelHandler(this, channelName, new File(URI.create("file://" + channelName)));
61      }
62      
63  }

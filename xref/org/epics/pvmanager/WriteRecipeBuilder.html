5   package org.epics.pvmanager;
6   
7   import java.util.Collection;
8   import java.util.HashMap;
9   import java.util.HashSet;
10  import java.util.Map;
11  import java.util.Set;
12  
13  /**
14   * A builder for {@link WriteRecipe }.
15   *
16   * @author carcassi
17   */
18  public class WriteRecipeBuilder {
19  
20      private final Map<String, Collection<WriteCache<?>>> caches;
21  
22      /**
23       * A new builder
24       */
25      public WriteRecipeBuilder() {
26          caches = new HashMap<>();
27      }
28      
29      /**
30       * Adds a channel and its write cache to the recipe.
31       *
32       * @param channelName the name of the channel
33       * @param writeCache the cache that will store the value to write
34       * @return this builder
35       */
36      public WriteRecipeBuilder addChannel(String channelName, WriteCache<?> writeCache) {
37          Collection<WriteCache<?>> cachesForChannel = caches.get(channelName);
38          if (cachesForChannel == null) {
39              cachesForChannel = new HashSet<>();
40              caches.put(channelName, cachesForChannel);
41          }
42          cachesForChannel.add(writeCache);
43          return this;
44      }
45  
46      /**
47       * Builds the recipe.
48       * <p>
49       * To finish building the recipe, one needs to specify where to send errors
50       * and where to send the connection status changes.
51       * 
52       * @param exceptionWriteFunction where exception should be routed
53       * @param connectionCollector where connection status should be routed
54       * @return a new WriteRecipe
55       */
56      public WriteRecipe build(WriteFunction<Exception> exceptionWriteFunction, ConnectionCollector connectionCollector) {
57          Set<ChannelWriteRecipe> recipes = new HashSet<>();
58          for (Map.Entry<String, Collection<WriteCache<?>>> entry : caches.entrySet()) {
59              String channelName = entry.getKey();
60              Collection<WriteCache<?>> valueCaches = entry.getValue();
61              for (WriteCache<?> valueCache : valueCaches) {
62                  recipes.add(new ChannelWriteRecipe(channelName, 
63                          new ChannelHandlerWriteSubscription(valueCache, exceptionWriteFunction, connectionCollector.addChannel(channelName))));
64              }
65          }
66          return new WriteRecipe(recipes);
67      }
68      
69  }

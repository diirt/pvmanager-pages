5   package org.epics.pvmanager.exec;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   import java.util.concurrent.ExecutorService;
10  import org.epics.pvmanager.service.ServiceMethodDescription;
11  import org.epics.vtype.VType;
12  
13  /**
14   * The description for an executor service method (i.e. a shell command).
15   *
16   * @author carcassi
17   */
18  public class ExecServiceMethodDescription {
19      
20      final ServiceMethodDescription serviceMethodDescription;
21      boolean resultAdded = false;
22      ExecutorService executorService;
23      String shell;
24      String shellArg;
25      String command;
26      final List<String> orderedParameterNames = new ArrayList<>();
27  
28      /**
29       * A new service method with the given name and description.
30       * 
31       * @param name the method name
32       * @param description the method description
33       */
34      public ExecServiceMethodDescription(String name, String description) {
35          serviceMethodDescription = new ServiceMethodDescription(name, description);
36      }
37      
38      /**
39       * Adds an argument for the query.
40       * <p>
41       * Arguments need to be specified in the same order as they appear in the query.
42       * 
43       * @param name argument name
44       * @param description argument description
45       * @param type the expected type of the argument
46       * @return this
47       */
48      public ExecServiceMethodDescription addArgument(String name, String description, Class<?> type) {
49          serviceMethodDescription.addArgument(name, description, type);
50          orderedParameterNames.add(name);
51          return this;
52      }
53      
54      /**
55       * Adds a result for the script.
56       * 
57       * @param name the result name
58       * @param description the result description
59       * @return this
60       */
61      public ExecServiceMethodDescription queryResult(String name, String description) {
62          if (resultAdded) {
63              throw new IllegalArgumentException("The query can only have one result");
64          }
65          serviceMethodDescription.addResult(name, description, VType.class);
66          return this;
67      }
68      
69      ExecServiceMethodDescription executorService(ExecutorService executorService) {
70          if (this.executorService != null) {
71              throw new IllegalArgumentException("ExecutorService was already set");
72          }
73          this.executorService = executorService;
74          return this;
75      }
76  
77      public ExecServiceMethodDescription command(String command) {
78          this.command = command;
79          return this;
80      }
81      
82  }

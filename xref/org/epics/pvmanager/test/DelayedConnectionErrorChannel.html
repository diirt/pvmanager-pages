5   package org.epics.pvmanager.test;
6   
7   import org.epics.pvmanager.ChannelWriteCallback;
8   import org.epics.pvmanager.MultiplexedChannelHandler;
9   import org.epics.pvmanager.PVManager;
10  import java.util.concurrent.TimeUnit;
11  import org.epics.pvmanager.*;
12  
13  /**
14   * Implementation for channels of a {@link TestDataSource}.
15   *
16   * @author carcassi
17   */
18  class DelayedConnectionErrorChannel extends MultiplexedChannelHandler<Object, Object> {
19  
20      DelayedConnectionErrorChannel(String channelName) {
21          super(channelName);
22      }
23  
24      @Override
25      public void connect() {
26          PVManager.getReadScannerExecutorService().schedule(new Runnable() {
27  
28              @Override
29              public void run() {
30                  reportExceptionToAllReadersAndWriters(new RuntimeException("Connection error"));
31              }
32          }, 1, TimeUnit.SECONDS);
33      }
34  
35      @Override
36      public void disconnect() {
37          processConnection(null);
38      }
39  
40      @Override
41      public void write(Object newValue, ChannelWriteCallback callback) {
42          // Do nothing
43      }
44  
45      @Override
46      public boolean isConnected(Object obj) {
47          return false;
48      }
49      
50  }

5   package org.epics.pvmanager.test;
6   
7   import java.util.List;
8   import java.util.concurrent.CopyOnWriteArrayList;
9   import org.epics.pvmanager.ChannelHandler;
10  import org.epics.pvmanager.ChannelReadRecipe;
11  import org.epics.pvmanager.ChannelWriteRecipe;
12  import org.epics.pvmanager.DataSource;
13  import org.epics.pvmanager.ExceptionHandler;
14  import org.epics.pvmanager.ReadRecipe;
15  import org.epics.pvmanager.WriteRecipe;
16  
17  /**
18   *
19   * @author carcassi
20   */
21  public class MockDataSource extends DataSource {
22      
23      private final List<ChannelReadRecipe> connectedReadRecipes = new CopyOnWriteArrayList<>();
24      private final List<ChannelWriteRecipe> connectedWriteRecipes = new CopyOnWriteArrayList<>();
25      private volatile ReadRecipe readRecipe;
26      private volatile WriteRecipe writeRecipe;
27      private volatile WriteRecipe writeRecipeForWrite;
28  
29      public MockDataSource() {
30          super(true);
31      }
32  
33      @Override
34      protected ChannelHandler createChannel(String channelName) {
35          throw new UnsupportedOperationException("Not supported yet.");
36      }
37  
38      @Override
39      public void connectRead(ReadRecipe recipe) {
40          this.readRecipe = recipe;
41          connectedReadRecipes.addAll(recipe.getChannelReadRecipes());
42      }
43  
44      @Override
45      public void disconnectRead(ReadRecipe recipe) {
46          this.readRecipe = recipe;
47          connectedReadRecipes.removeAll(recipe.getChannelReadRecipes());
48      }
49  
50      @Override
51      public void connectWrite(WriteRecipe writeRecipe) {
52          this.writeRecipe = writeRecipe;
53          connectedWriteRecipes.addAll(writeRecipe.getChannelWriteRecipes());
54      }
55  
56      @Override
57      public void disconnectWrite(WriteRecipe writeRecipe) {
58          this.writeRecipe = writeRecipe;
59          connectedWriteRecipes.removeAll(writeRecipe.getChannelWriteRecipes());
60      }
61  
62      @Override
63      public void write(WriteRecipe writeRecipe, Runnable callback, ExceptionHandler exceptionHandler) {
64          this.writeRecipeForWrite = writeRecipe;
65      }
66  
67      public ReadRecipe getReadRecipe() {
68          return readRecipe;
69      }
70  
71      public WriteRecipe getWriteRecipe() {
72          return writeRecipe;
73      }
74  
75      public WriteRecipe getWriteRecipeForWrite() {
76          return writeRecipeForWrite;
77      }
78  
79      public List<ChannelReadRecipe> getConnectedReadRecipes() {
80          return connectedReadRecipes;
81      }
82  
83      public List<ChannelWriteRecipe> getConnectedWriteRecipes() {
84          return connectedWriteRecipes;
85      }
86      
87  }

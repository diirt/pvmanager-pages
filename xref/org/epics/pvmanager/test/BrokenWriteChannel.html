5   package org.epics.pvmanager.test;
6   
7   import org.epics.pvmanager.ChannelWriteCallback;
8   import org.epics.pvmanager.MultiplexedChannelHandler;
9   import org.epics.pvmanager.*;
10  
11  /**
12   * Implementation for channels of a {@link TestDataSource}.
13   *
14   * @author carcassi
15   */
16  class BrokenWriteChannel extends MultiplexedChannelHandler<Object, Object> {
17  
18      BrokenWriteChannel(String channelName) {
19          super(channelName);
20      }
21  
22      @Override
23      public void connect() {
24          processConnection(new Object());
25      }
26  
27      @Override
28      public void disconnect() {
29          processConnection(null);
30      }
31  
32      @Override
33      public void write(Object newValue, ChannelWriteCallback callback) {
34          try {
35              callback.channelWritten(new RuntimeException("BrokenWriteChannel"));
36          } catch (Exception ex) {
37              callback.channelWritten(ex);
38          }
39      }
40      
41  }

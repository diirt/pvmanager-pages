5   package org.epics.pvmanager.expression;
6   
7   import java.util.List;
8   import org.epics.pvmanager.CacheCollector;
9   import org.epics.pvmanager.WriteFunction;
10  
11  /**
12   * A cache expression.
13   * <p>
14   * The expression can be used to collect values from pvmanager datasources or from
15   * external sources.
16   *
17   * @param <T> the type of values in the cache
18   * @author carcassi
19   */
20  public class Cache<T> extends DesiredRateExpressionImpl<List<T>> {
21      
22      private static <T> CacheCollector<T> createCache(int maxElements) {
23          return new CacheCollector<>(maxElements);
24      }
25  
26      /**
27       * Creates a new cache expression.
28       *
29       * @param maxSize the maximum number of elements in the cache
30       */
31      public Cache(int maxSize) {
32          super(new DesiredRateExpressionListImpl<Object>(), Cache.<T>createCache(maxSize), "queue");
33      }
34  
35      /**
36       * Creates a new cache expression.
37       *
38       * @param sourceExpression the source rate expression that will fill the cache
39       * @param maxSize the maximum number of elements in the cache
40       */
41      public Cache(SourceRateExpression<T> sourceExpression, int maxSize) {
42          super(sourceExpression, Cache.<T>createCache(maxSize), "queue");
43      }
44      
45      /**
46       * The write function to be used to fill the cache.
47       *
48       * @return a write function
49       */
50      public WriteFunction<T> getWriteFunction() {
51          return getCollector();
52      }
53      
54      @SuppressWarnings("unchecked")
55      private CacheCollector<T> getCollector() {
56          return (CacheCollector<T>) getFunction();
57      }
58      
59      /**
60       * Changes the maximum size of the cache.
61       *
62       * @param maxSize the number of values kept in the cache
63       * @return this expression
64       */
65      public Cache<T> maxSize(int maxSize) {
66          getCollector().setMaxSize(maxSize);
67          return this;
68      }
69      
70      /**
71       * Adds a new value to the cache.
72       *
73       * @param newValue the value to be added
74       */
75      public void add(T newValue) {
76          getWriteFunction().writeValue(newValue);
77      }
78      
79  }

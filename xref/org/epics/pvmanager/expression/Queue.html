5   package org.epics.pvmanager.expression;
6   
7   import java.util.List;
8   import org.epics.pvmanager.QueueCollector;
9   import org.epics.pvmanager.WriteFunction;
10  
11  /**
12   * A queue expression.
13   * <p>
14   * The expression can be used to collect values from pvmanager datasources or from
15   * external sources.
16   *
17   * @param <T> the type of values in the queue
18   * @author carcassi
19   */
20  public class Queue<T> extends DesiredRateExpressionImpl<List<T>> {
21      
22      private static <T> QueueCollector<T> createQueue(int maxSize) {
23          return new QueueCollector<>(maxSize);
24      }
25  
26      /**
27       * Creates a new queue expression.
28       *
29       * @param maxSize the maximum number of elements in the cache
30       */
31      public Queue(int maxSize) {
32          super(new DesiredRateExpressionListImpl<Object>(), Queue.<T>createQueue(maxSize), "queue");
33      }
34  
35      /**
36       * Creates a new queue expression.
37       *
38       * @param sourceExpression the source rate expression that will fill the queue
39       * @param maxSize the maximum number of elements in the cache
40       */
41      public Queue(SourceRateExpression<T> sourceExpression, int maxSize) {
42          super(sourceExpression, Queue.<T>createQueue(maxSize), "queue");
43      }
44      
45      /**
46       * The write function to be used to fill the queue.
47       *
48       * @return a write function
49       */
50      public WriteFunction<T> getWriteFunction() {
51          return getCollector();
52      }
53      
54      @SuppressWarnings("unchecked")
55      private QueueCollector<T> getCollector() {
56          return (QueueCollector<T>) getFunction();
57      }
58      
59      /**
60       * Changes the maximum size of the queue.
61       *
62       * @param maxSize the number of values kept in the queue
63       * @return this expression
64       */
65      public Queue<T> maxSize(int maxSize) {
66          getCollector().setMaxSize(maxSize);
67          return this;
68      }
69      
70      /**
71       *
72       * @param newValue
73       */
74      public void add(T newValue) {
75          getWriteFunction().writeValue(newValue);
76      }
77      
78  }

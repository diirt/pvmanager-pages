5   package org.epics.pvmanager.expression;
6   
7   import java.util.Collection;
8   
9   /**
10   * Represents a list of channel, which can be both read or written.
11   *
12   * @param <R> type of the read payload
13   * @param <W> type of the write payload
14   * @author carcassi
15   */
16  public class ChannelExpressionList<R, W> extends SourceRateReadWriteExpressionListImpl<R, W> {
17  
18      /**
19       * An expression for a list of channels with the given names, which are expected to
20       * provide a read payload of {@code readClass} and accept a write payload
21       * of {@code writeClass}.
22       * 
23       * @param readClass type of the read payload
24       * @param writeClass type of the write payload
25       * @param channelNames the names of the channels
26       */
27      public ChannelExpressionList(Collection<String> channelNames, Class<R> readClass, Class<W> writeClass) {
28          for (String channelName : channelNames) {
29              if (channelName == null) {
30                  and(new ChannelExpression<R, W>(readClass, writeClass));
31              } else {
32                  and(new ChannelExpression<R, W>(channelName, readClass, writeClass));
33              }
34          }
35      }
36      
37      /**
38       * For writes only, marks that these channels should be written only after the
39       * given channels.
40       * 
41       * @param channelNames preceding channel names
42       * @return this
43       */
44      public ChannelExpressionList<R, W> after(String... channelNames) {
45          for (SourceRateReadWriteExpression<R, W> expression : getSourceRateReadWriteExpressions()) {
46              ChannelExpression<R, W> channel = (ChannelExpression<R, W>) expression;
47              channel.after(channelNames);
48          }
49          return this;
50      }
51      
52      
53  }

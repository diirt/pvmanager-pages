5   package org.epics.pvmanager.expression;
6   
7   import org.epics.pvmanager.ReadFunction;
8   
9   /**
10   * An expression to read at the rate of the source.
11   * <p>
12   * Don't implement objects with this interface, use {@link SourceRateExpressionImpl}.
13   *
14   * @param <R> type of the read payload
15   * @author carcassi
16   */
17  public interface SourceRateExpression<R> extends SourceRateExpressionList<R> {
18      
19      /**
20       * Changes the name for this expression
21       * 
22       * @param name new name
23       * @return this
24       */
25      public SourceRateExpression<R> as(String name);
26  
27      /**
28       * Name of the expression.
29       *
30       * @return the expression name
31       */
32      public String getName();
33  
34      /**
35       * The function that calculates this expression.
36       *
37       * @return the expression function
38       */
39      public ReadFunction<R> getFunction();
40      
41      /**
42       * The implementation of this expression.
43       * 
44       * @return the implementation
45       */
46      public SourceRateExpressionImpl<R> getSourceRateExpressionImpl();
47  
48  }

5   package org.epics.pvmanager.expression;
6   
7   import java.util.ArrayList;
8   import java.util.Collections;
9   import java.util.List;
10  
11  /**
12   * Implementation class for {@link SourceRateReadWriteExpressionList}.
13   *
14   * @param <R> type of the read payload
15   * @param <W> type of the write payload
16   * @author carcassi
17   */
18  public class SourceRateReadWriteExpressionListImpl<R, W> implements SourceRateReadWriteExpressionList<R, W> {
19      
20      private List<SourceRateReadWriteExpression<R, W>> sourceRateReadWriteExpressions = new ArrayList<SourceRateReadWriteExpression<R, W>>();
21      
22      final void addThis() {
23          sourceRateReadWriteExpressions.add((SourceRateReadWriteExpression<R, W>) this);
24      }
25  
26      @Override
27      public final SourceRateReadWriteExpressionList<R, W> and(SourceRateReadWriteExpressionList<? extends R, ? extends W> expressions) {
28          @SuppressWarnings("unchecked")
29          SourceRateReadWriteExpressionList<R, W> newExpression = (SourceRateReadWriteExpressionList<R, W>) (SourceRateReadWriteExpressionList) expressions;
30          sourceRateReadWriteExpressions.addAll(newExpression.getSourceRateReadWriteExpressions());
31          return this;
32      }
33  
34      @Override
35      public final List<SourceRateReadWriteExpression<R, W>> getSourceRateReadWriteExpressions() {
36          return sourceRateReadWriteExpressions;
37      }
38  
39      @Override
40      public final SourceRateExpressionList<R> and(SourceRateExpressionList<? extends R> expressions) {
41          @SuppressWarnings("unchecked")
42          SourceRateExpressionList<R> newExpression = (SourceRateExpressionList<R>) (SourceRateExpressionList) expressions;
43          return new SourceRateExpressionListImpl<R>(sourceRateReadWriteExpressions).and(newExpression);
44      }
45  
46      @Override
47      public final List<SourceRateExpression<R>> getSourceRateExpressions() {
48          return Collections.<SourceRateExpression<R>>unmodifiableList(sourceRateReadWriteExpressions);
49      }
50  
51      @Override
52      public final WriteExpressionList<W> and(WriteExpressionList<? extends W> expressions) {
53          @SuppressWarnings("unchecked")
54          WriteExpressionList<W> newExpression = (WriteExpressionList<W>) (WriteExpressionList) expressions;
55          return new WriteExpressionListImpl<W>(sourceRateReadWriteExpressions).and(newExpression);
56      }
57  
58      @Override
59      public final List<WriteExpression<W>> getWriteExpressions() {
60          return Collections.<WriteExpression<W>>unmodifiableList(sourceRateReadWriteExpressions);
61      }
62      
63  }

5   package org.epics.pvmanager.expression;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   import org.epics.pvmanager.ReadFunction;
10  
11  /**
12   * Utility class for expressions.
13   *
14   * @author carcassi
15   */
16  public class Expressions {
17  
18      /**
19       * Extract the list of functions from an expression list.
20       * 
21       * @param list the expressions
22       * @return the functions of the expression
23       */
24      public static List<ReadFunction<?>> functionsOf(DesiredRateExpressionList<?> list) {
25          List<ReadFunction<?>> result = new ArrayList<ReadFunction<?>>();
26          for (DesiredRateExpression<?> desiredRateExpression : list.getDesiredRateExpressions()) {
27              result.add(desiredRateExpression.getFunction());
28          }
29          return result;
30      }
31  }

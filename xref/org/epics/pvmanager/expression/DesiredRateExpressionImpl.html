5   package org.epics.pvmanager.expression;
6   
7   import org.epics.pvmanager.Collector;
8   import org.epics.pvmanager.PVReaderDirector;
9   import org.epics.pvmanager.ReadFunction;
10  import org.epics.pvmanager.ReadRecipeBuilder;
11  
12  /**
13   * Implementation class for {@link DesiredRateExpression}.
14   *
15   * @param <R> type of the read payload
16   * @author carcassi
17   */
18  public class DesiredRateExpressionImpl<R> extends DesiredRateExpressionListImpl<R> implements DesiredRateExpression<R> {
19  
20      private final ReadFunction<R> function;
21      private String name;
22      private final SourceRateExpression<?> sourceRateChild;
23      private final DesiredRateExpressionList<?> desiredRateChildren;
24      
25      {
26          // Make sure that the list includes this expression
27          addThis();
28      }
29  
30      @Override
31      public final DesiredRateExpressionImpl<R> as(String name) {
32          this.name = name;
33          return this;
34      }
35  
36      /**
37       * Creates a new expression at the desired rate. Use this constructor when making
38       * an DesiredRateExpression out of a collector and a SourceRateExpression.
39       *
40       * @param expression the original source rate expression
41       * @param collector the collector for the original source
42       * @param defaultName the display name of the expression
43       */
44      public DesiredRateExpressionImpl(SourceRateExpression<?> expression, Collector<?, R> collector, String defaultName) {
45          this.sourceRateChild = expression;
46          this.desiredRateChildren = null;
47          this.function = collector;
48          this.name = defaultName;
49      }
50  
51      /**
52       * Creates a new aggregated expression. Use this constructor when making
53       * a {@code DesiredRateExpression} that is a function of a number of
54       * {@code DesiredRateExpression}s.
55       *
56       * @param childExpressions expressions for the arguments of the function
57       * @param function the function that calculates the value of the new expression
58       * @param defaultName the display name of the expression
59       */
60      public DesiredRateExpressionImpl(DesiredRateExpressionList<?> childExpressions, ReadFunction<R> function, String defaultName) {
61          this.sourceRateChild = null;
62          this.desiredRateChildren = childExpressions;
63          this.function = function;
64          this.name = defaultName;
65      }
66  
67      /**
68       * The default name for a PV of this expression.
69       *
70       * @return the default name
71       */
72      @Override
73      public final String getName() {
74          return name;
75      }
76  
77      @Override
78      @SuppressWarnings("unchecked")
79      public void fillReadRecipe(PVReaderDirector director, ReadRecipeBuilder builder) {
80          if (function instanceof Collector) {
81              director.registerCollector((Collector) function);
82          }
83          if (sourceRateChild != null) {
84              sourceRateChild.getSourceRateExpressionImpl().fillDataRecipe(director, (Collector) function, builder);
85          } else if (desiredRateChildren != null) {
86              for (DesiredRateExpression<?> desiredRateExpression : desiredRateChildren.getDesiredRateExpressions()) {
87                  desiredRateExpression.fillReadRecipe(director, builder);
88              }
89          }
90      }
91  
92      /**
93       * The function that calculates new values for this expression.
94       *
95       * @return a function
96       */
97      @Override
98      public final ReadFunction<R> getFunction() {
99          return function;
100     }
101 
102     /**
103      * The implementation for this expression.
104      * 
105      * @return returns the implementation for this desired rate
106      */
107     @Override
108     public final DesiredRateExpressionImpl<R> getDesiredRateExpressionImpl() {
109         return this;
110     }
111     
112 }

5   package org.epics.pvmanager.expression;
6   
7   import java.util.List;
8   
9   /**
10   * A list of expressions to read at the rate of the source.
11   * <p>
12   * Don't implement objects with this interface, use {@link SourceRateExpressionListImpl}.
13   *
14   * @param <R> type of the read payload
15   * @author carcassi
16   */
17  public interface SourceRateExpressionList<R> {
18      
19      /**
20       * Adds the given expressions to this list.
21       * 
22       * @param expressions a list of expressions
23       * @return this
24       */
25      public SourceRateExpressionList<R> and(SourceRateExpressionList<? extends R> expressions);
26  
27      /**
28       * The expressions of this list.
29       * 
30       * @return a list of expressions
31       */
32      public List<SourceRateExpression<R>> getSourceRateExpressions();
33      
34  }

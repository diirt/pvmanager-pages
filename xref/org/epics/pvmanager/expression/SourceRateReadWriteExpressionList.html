5   package org.epics.pvmanager.expression;
6   
7   import java.util.List;
8   
9   /**
10   * A list of expressions to write and to read at the rate of the source.
11   * <p>
12   * Don't implement objects with this interface, use {@link SourceRateReadWriteExpressionListImpl}.
13   *
14   * @param <R> type of the read payload
15   * @param <W> type of the write payload
16   * 
17   * A list of desired rate expression, to have functions that work on multiple
18   * expressions at the same time.
19   *
20   * @author carcassi
21   */
22  public interface SourceRateReadWriteExpressionList<R, W> extends SourceRateExpressionList<R>, WriteExpressionList<W> {
23      
24      /**
25       * Adds the given expressions to this list.
26       * 
27       * @param expressions a list of expressions
28       * @return this
29       */
30      public SourceRateReadWriteExpressionList<R, W> and(SourceRateReadWriteExpressionList<? extends R, ? extends W> expressions);
31  
32      /**
33       * The expressions of this list.
34       * 
35       * @return a list of expressions
36       */
37      public List<SourceRateReadWriteExpression<R, W>> getSourceRateReadWriteExpressions();
38  }

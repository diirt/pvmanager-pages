6   package org.epics.pvmanager;
7   
8   import java.util.concurrent.ScheduledExecutorService;
9   import java.util.concurrent.ScheduledFuture;
10  import java.util.concurrent.TimeUnit;
11  import org.epics.util.time.TimeDuration;
12  import org.epics.util.time.Timestamp;
13  
14  /**
15   *
16   * @author carcassi
17   */
18  class PassiveScanner implements Scanner {
19      /** Executor used to scan the connection/exception queues */
20      private final ScheduledExecutorService scannerExecutor;
21      private volatile ScheduledFuture<?> scanTaskHandle;
22      private final PVReaderDirector readerDirector;
23      private final TimeDuration maxDuration;
24      private final Object lock = new Object();
25      private Timestamp lastSubmission;
26      // Keeps track of whether 
27      private boolean processingEvent;
28      private boolean eventQueued;
29  
30      PassiveScanner(ScheduledExecutorService scannerExecutor, PVReaderDirector readerDirector, TimeDuration maxDuration,
31              TimeDuration timeout, final String timeoutMessage) {
32          this.scannerExecutor = scannerExecutor;
33          this.readerDirector = readerDirector;
34          this.maxDuration = maxDuration;
35          if (timeout != null) {
36              scannerExecutor.schedule(new Runnable() {
37                  @Override
38                  public void run() {
39                      PassiveScanner.this.readerDirector.processTimeout(timeoutMessage);
40                  }
41              }, timeout.toNanosLong(), TimeUnit.NANOSECONDS);
42          }
43      }
44      
45      private final Runnable notificationTask = new Runnable() {
46  
47          @Override
48          public void run() {
49              if (readerDirector.isActive()) {
50                  // If paused, simply skip without stopping the scan
51                  if (!readerDirector.isPaused()) {
52                      synchronized (lock) {
53                          processingEvent = true;
54                      }
55                      readerDirector.notifyPv();
56                  }
57              } else {
58                  stop();
59              }
60          }
61      };
62      
63      void scheduleNext() {
64          Timestamp now = Timestamp.now();
65          if (lastSubmission == null) {
66              scannerExecutor.submit(notificationTask);
67          } else {
68              TimeDuration delay = maxDuration.minus(now.durationBetween(lastSubmission));
69              if (delay.isPositive()) {
70                  scannerExecutor.schedule(notificationTask, delay.toNanosLong(), TimeUnit.NANOSECONDS);
71              }
72          }
73      }
74      
75      private void newEvent() {
76          boolean submit;
77          synchronized(lock) {
78              if (processingEvent) {
79                  // If there is already an event processing, we should
80                  // submit after the previous event is done
81                  submit = false;
82                  eventQueued = true;
83              } else {
84                  // If there isn't an event processing,
85                  // we need to submit anyway. We can clear the event
86                  // queued at this point: we are going to submit anyway
87                  submit = true;
88                  eventQueued = false;
89              }
90          }
91          if (submit) {
92              scheduleNext();
93          }
94      }
95      
96      @Override
97      public void start() {
98          newEvent();
99      }
100     
101     @Override
102     public void stop() {
103         readerDirector.close();
104         readerDirector.disconnect();
105         synchronized(lock) {
106             eventQueued = false;
107         }
108     }
109 
110     @Override
111     public void pause() {
112         // Do nothing
113     }
114 
115     @Override
116     public void resume() {
117         // Do nothing
118     }
119 
120     @Override
121     public void collectorChange() {
122         newEvent();
123     }
124 
125     @Override
126     public void notifiedPv() {
127         boolean submit;
128         synchronized(lock) {
129             processingEvent = false;
130             if (eventQueued) {
131                 // If there is an event in queue, we should submit
132                 // next execution
133                 submit = true;
134                 eventQueued = false;
135             } else {
136                 // Do nothing
137                 submit = false;
138             }
139         }
140         if (submit) {
141             scheduleNext();
142         }
143     }
144     
145 }

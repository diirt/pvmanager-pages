5   package org.epics.pvmanager;
6   
7   import org.epics.pvmanager.test.WriteRecipeUtil;
8   import org.epics.pvmanager.expression.WriteExpression;
9   
10  /**
11   *
12   * @author carcassi
13   */
14  public class WriteExpressionTester {
15  
16      private WriteExpression<?> expression;
17      private PVWriterDirector<?> pvWriterDirector = new PVWriterDirector<Object>(null, null, null, null, null, null, null, null, null);
18  
19      public WriteExpressionTester(WriteExpression<?> expression) {
20          this.expression = expression;
21          pvWriterDirector.connectExpression(expression);
22      }
23  
24      public Object readValue(String name) {
25          return WriteRecipeUtil.valueFor(pvWriterDirector.getCurrentWriteRecipe(), name);
26      }
27      
28      public ChannelWriteRecipe recipeFor(String channelName) {
29          return WriteRecipeUtil.recipeFor(pvWriterDirector.getCurrentWriteRecipe(), channelName);
30      }
31      
32      public WriteRecipe getWriteRecipe() {
33          return pvWriterDirector.getCurrentWriteRecipe();
34      }
35      
36      @SuppressWarnings("unchecked")
37      public WriteFunction<Object> getWriteFunction() {
38          return (WriteFunction<Object>) expression.getWriteFunction();
39      }
40      
41      public WriteExpression<?> getExpression() {
42          return expression;
43      }
44      
45      public void setValue(Object value) {
46          getWriteFunction().writeValue(value);
47      }
48  }

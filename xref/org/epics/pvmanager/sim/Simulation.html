5   package org.epics.pvmanager.sim;
6   
7   import java.util.List;
8   import java.util.logging.Logger;
9   import org.epics.util.time.TimeDuration;
10  import org.epics.util.time.TimeInterval;
11  import org.epics.util.time.Timestamp;
12  
13  /**
14   * Base class for all simulated signals. It provides the common mechanism for
15   * registering the update on a timer and a few other utilities.
16   *
17   * @author carcassi
18   */
19  abstract class Simulation<T> {
20  
21      private static final Logger log = Logger.getLogger(Simulation.class.getName());
22  
23      private final long intervalBetweenExecution;
24      private final Class<T> classToken;
25      volatile Timestamp lastTime;
26  
27      /**
28       * Creates a new simulation.
29       *
30       * @param secondsBeetwenSamples seconds between each samples
31       */
32      Simulation(TimeDuration scanRate, Class<T> classToken) {
33          // XXX: this is dead code at this point. If needed, use it when setting up
34          // the scan job. Ignore otherwise.
35          if (scanRate.compareTo(TimeDuration.ofMillis(1)) < 0) {
36              throw new IllegalArgumentException("Scans must be at least every ms (was " + scanRate + ")");
37          }
38          this.intervalBetweenExecution = Math.max(scanRate.toNanosLong() / 1000000, 1);
39          this.classToken = classToken;
40      }
41  
42      /**
43       * Computes all the new values in the given time slice.
44       *
45       * @param interval the interval where the data should be generated
46       * @return the new values
47       */
48      abstract List<T> createValues(TimeInterval interval);
49  
50      /**
51       * Changes the time at which the data will be generated.
52       *
53       * @param lastTime new timestamp
54       */
55      void setLastTime(Timestamp lastTime) {
56          this.lastTime = lastTime;
57      }
58  
59  }

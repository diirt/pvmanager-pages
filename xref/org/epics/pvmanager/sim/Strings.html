5   package org.epics.pvmanager.sim;
6   
7   import org.epics.vtype.VString;
8   import static org.epics.vtype.ValueFactory.*;
9   
10  /**
11   * Function to simulate a signal that generates Strings.
12   *
13   * @author carcassi
14   */
15  public class Strings extends SimFunction<VString> {
16  
17      private StringBuffer buffer = new StringBuffer();
18  
19      /**
20       * Creates a signal uniformly distributed between -5.0 and 5.0, updating
21       * every 100ms (10Hz).
22       */
23      public Strings() {
24          this(0.1);
25      }
26  
27      /**
28       * Creates a signal uniformly distributed between min and max, updating
29       * every interval seconds.
30       *
31       * @param interval interval between samples in seconds
32       */
33      public Strings(Double interval) {
34          super(interval, VString.class);
35          if (interval <= 0.0) {
36              throw new IllegalArgumentException("Interval must be greater than zero (was " + interval + ")");
37          }
38      }
39  
40      @Override
41      VString nextValue() {
42          return newVString(nextString(), alarmNone(), timeNow());
43      }
44  
45      String nextString() {
46          if (buffer.length() > 10) {
47              buffer.setLength(0);
48          }
49          buffer.append("A");
50          return buffer.toString();
51      }
52  }

5   package org.epics.pvmanager.sim;
6   
7   import java.util.Random;
8   import org.epics.vtype.VDouble;
9   import static org.epics.vtype.ValueFactory.*;
10  import org.epics.util.time.Timestamp;
11  
12  /**
13   * Function to simulate a signal that has a uniform distribution. The warning
14   * limits are set at 80% of the range and the alarm at 90% the range.
15   * All values are going to have no alarm status, with the timestamp set at the
16   * moment the sample was generated.
17   *
18   * @author carcassi
19   */
20  public class Noise extends SimFunction<VDouble> {
21  
22      private Random rand = new Random();
23      private double min;
24      private double max;
25      private double range;
26      private VDouble lastValue;
27  
28      /**
29       * Creates a signal uniformly distributed between -5.0 and 5.0, updating
30       * every 100ms (10Hz).
31       */
32      public Noise() {
33          this(-5.0, 5.0, 1.0);
34      }
35      
36      /**
37       * Do not use: only provided to provide some sort of error message
38       * for people migrating from utility.pv.
39       * 
40       * @param min minimum value
41       * @param max maximum value
42       * @param step ignored
43       * @param interval interval between samples in seconds
44       */
45      public Noise(Double min, Double max, Double step, Double interval) {
46          super(interval, VDouble.class);
47          throw new IllegalArgumentException("Please rename to noise(" + min + ", " + max + ", " + interval + ")");
48      }
49  
50      /**
51       * Creates a signal uniformly distributed between min and max, updating
52       * every interval seconds.
53       *
54       * @param min minimum value
55       * @param max maximum value
56       * @param interval interval between samples in seconds
57       */
58      public Noise(Double min, Double max, Double interval) {
59          super(interval, VDouble.class);
60          if (interval <= 0.0) {
61              throw new IllegalArgumentException("Interval must be greater than zero (was " + interval + ")");
62          }
63          this.min = min;
64          this.max = max;
65          range = this.max - this.min;
66          lastValue = newVDouble(min, alarmNone(), timeNow(),
67                  newDisplay(min, min + range * 0.1, min + range * 0.2, "x", Constants.DOUBLE_FORMAT,
68                  min + range * 0.8, min + range * 0.9, max, min, max));
69      }
70  
71      @Override
72      VDouble nextValue() {
73          return newValue(min + rand.nextDouble() * range, lastValue);
74      }
75  }

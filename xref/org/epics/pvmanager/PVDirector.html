20  /**
21   * Orchestrates the different elements of pvmanager to make a reader functional.
22   * <p>
23   * This class is responsible for the correct read operation, including:
24   * <ul>
25   * <li>Setting up the collector for notifications</li>
26   * <li>Setting up the collector for connection notification</li>
27   * <li>Building connection recipes and forwarding them to the datasource<li>
28   * <li>Managing the scanning task and notification for new values, connection status
29   * or errors</li>
30   * <li>Disconnecting the expressions from the datasources if the reader is closed
31   * or if it's garbage collected</li>
32   * </ul>
33   *
34   * @param <T> value type for the reader managed by this director
35   * @author carcassi
36   */
37  public class PVDirector<T> {
38      
39      private static final Logger log = Logger.getLogger(PVDirector.class.getName());
40  
41      // Required for connection and exception notification
42  
43      /** Executor used to notify of new values/connection/exception */
44      private final Executor notificationExecutor;
45      /** Executor used to scan the connection/exception queues */
46      private final ScheduledExecutorService scannerExecutor;
47      /** PVReader to update during the notification */
48      private final WeakReference<PVReaderImpl<T>> pvReaderRef;
49      /** Function for the new value */
50      private final ReadFunction<T> readFunction;
51      /** Creation for stack trace */
52      private final Exception creationStackTrace = new Exception("PV was never closed (stack trace for creation)");
53      
54      // Required to connect/disconnect expressions
55      private final DataSource dataSource;
56      private final Object lock = new Object();
57      private final Map<DesiredRateExpression<?>, ReadRecipe> readRecipies =
58              new HashMap<>();
59      private SourceDesiredRateDecoupler scanStrategy;
60  
61      // Required for multiple operations
62      /** Connection collector required to connect/disconnect expressions and for connection notification */
63      private final ConnectionCollector readConnCollector =
64              new ConnectionCollector();
65      /** Exception queue to be used to connect/disconnect expression and for exception notification */
66      private final QueueCollector<Exception> readExceptionCollector;
67      
68      void setScanner(final SourceDesiredRateDecoupler scanStrategy) {
69          synchronized(lock) {
70              this.scanStrategy = scanStrategy;
71          }
72          readExceptionCollector.setChangeNotification(new Runnable() {
73  
74              @Override
75              public void run() {
76                  scanStrategy.newReadExceptionEvent();
77              }
78          });
79          readConnCollector.setChangeNotification(new Runnable() {
80  
81              @Override
82              public void run() {
83                  scanStrategy.newReadConnectionEvent();
84              }
85          });
86      }
87      
88      public void registerCollector(Collector<?, ?> collector) {
89          collector.setChangeNotification(new Runnable() {
90  
91              @Override
92              public void run() {
93                  if (scanStrategy != null) {
94                      scanStrategy.newValueEvent();
95                  }
96              }
97          });
98      }

100     ReadRecipe getCurrentReadRecipe() {
101         ReadRecipeBuilder builder = new ReadRecipeBuilder();
102         for (Map.Entry<DesiredRateExpression<?>, ReadRecipe> entry : readRecipies.entrySet()) {
103             ReadRecipe readRecipe = entry.getValue();
104             for (ChannelReadRecipe channelReadRecipe : readRecipe.getChannelReadRecipes()) {
105                 builder.addChannel(channelReadRecipe.getChannelName(), channelReadRecipe.getReadSubscription().getValueCache());
106             }
107         }
108         return builder.build(readExceptionCollector, readConnCollector);
109     }
110     
111     /**
112      * Connects the given expression.
113      * <p>
114      * This can be used for dynamic expression to add and connect child expressions.
115      * The added expression will be automatically closed when the associated
116      * reader is closed, if it's not disconnected first.
117      * 
118      * @param expression the expression to connect
119      */
120     public void connectReadExpression(DesiredRateExpression<?> expression) {
121         ReadRecipeBuilder builder = new ReadRecipeBuilder();
122         expression.fillReadRecipe(this, builder);
123         ReadRecipe recipe = builder.build(readExceptionCollector, readConnCollector);
124         synchronized(lock) {
125             readRecipies.put(expression, recipe);
126         }
127         if (!recipe.getChannelReadRecipes().isEmpty()) {
128             try {
129                 dataSource.connectRead(recipe);
130             } catch(Exception ex) {
131                 recipe.getChannelReadRecipes().iterator().next().getReadSubscription().getExceptionWriteFunction().writeValue(ex);
132             }
133         }
134     }
135     
136     /**
137      * Simulate a static connection in which the channel has one exception
138      * and the connection will never change.
139      * <p>
140      * This is a temporary method an will be subject to change in the future.
141      * The aim is to allow to connect expressions that are not channels
142      * but can influence exception and connection state. For example,
143      * to report problems encountered during expression creation as runtime
144      * problems through the normal exception/connection methods.
145      * <p>
146      * In the future, this should be generalized to allow fully fledged expressions
147      * that connect/disconnect and can report errors.
148      * 
149      * @param ex the exception to queue
150      * @param connection the connection flag
151      * @param channelName the channel name
152      */
153     public void connectStaticRead(Exception ex, boolean connection, String channelName) {
154         readExceptionCollector.writeValue(ex);
155         readConnCollector.addChannel(channelName).writeValue(connection);
156     }
157     
158     /**
159      * Disconnects the given expression.
160      * <p>
161      * This can be used for dynamic expression, to remove and disconnects child
162      * expressions.
163      *
164      * @param expression the expression to disconnect
165      */
166     public void disconnectReadExpression(DesiredRateExpression<?> expression) {
167         ReadRecipe recipe;
168         synchronized(lock) {
169             recipe = readRecipies.remove(expression);
170         }
171         if (recipe == null) {
172             log.log(Level.SEVERE, "Director was asked to disconnect expression '" + expression + "' which was not found.");
173         }
174         
175         if (!recipe.getChannelReadRecipes().isEmpty()) {
176             try {
177                 for (ChannelReadRecipe channelRecipe : recipe.getChannelReadRecipes()) {
178                     readConnCollector.removeChannel(channelRecipe.getChannelName());
179                 }
180                 dataSource.disconnectRead(recipe);
181             } catch(Exception ex) {
182                 recipe.getChannelReadRecipes().iterator().next().getReadSubscription().getExceptionWriteFunction().writeValue(ex);
183             }
184         }
185     }
186     
187     private volatile boolean closed = false;
188     
189     void close() {
190         closed = true;
191         disconnect();
192     }
193 
194     /**
195      * Close and disconnects all the child expressions.
196      */
197     private void disconnect() {
198         synchronized(lock) {
199             while (!readRecipies.isEmpty()) {
200                 DesiredRateExpression<?> expression = readRecipies.keySet().iterator().next();
201                 disconnectReadExpression(expression);
202             }
203         }
204     }

206     /**
207      * Creates a new notifier. The new notifier will notifier the given pv
208      * with new values calculated by the function, and will use onThread to
209      * perform the notifications.
210      * <p>
211      * After construction, one MUST set the pvRecipe, so that the
212      * dataSource is appropriately closed.
213      *
214      * @param pv the pv on which to notify
215      * @param function the function used to calculate new values
216      * @param notificationExecutor the thread switching mechanism
217      */
218     PVDirector(PVReaderImpl<T> pv, ReadFunction<T> function, ScheduledExecutorService scannerExecutor,
219             Executor notificationExecutor, DataSource dataSource, ExceptionHandler exceptionHandler) {
220         this.pvReaderRef = new WeakReference<>(pv);
221         this.readFunction = function;
222         this.notificationExecutor = notificationExecutor;
223         this.scannerExecutor = scannerExecutor;
224         this.dataSource = dataSource;
225         if (exceptionHandler == null) {
226             readExceptionCollector = new QueueCollector<>(1);
227         } else {
228             readExceptionCollector = new LastExceptionCollector(1, exceptionHandler);
229         }
230     }
231 
232     /**
233      * Determines whether the notifier is active or not.
234      * <p>
235      * The notifier becomes inactive if the PVReader is closed or is garbage collected.
236      * The first time this function determines that the notifier is inactive,
237      * it will ask the data source to close all channels relative to the
238      * pv.
239      *
240      * @return true if new notification should be performed
241      */
242     private boolean isActive() {
243         // Making sure to get the reference once for thread safety
244         final PVReader<T> pv = pvReaderRef.get();
245         if (pv != null && !pv.isClosed()) {
246             return true;
247         } else if (pv == null && closed != true) {
248             log.log(Level.WARNING, "PVReader wasn't properly closed and it was garbage collected. Closing the associated connections...", creationStackTrace);
249             return false;
250         } else {
251             return false;
252         }
253     }

257     }

263     private volatile boolean notificationInFlight = false;
264     
265     /**
266      * Notifies the PVReader of a new value.
267      */
268     private void notifyPv() {
269         // Don't even calculate if notification is in flight.
270         // This makes pvManager automatically throttle back if the consumer
271         // is slower than the producer.
272         if (notificationInFlight)
273             return;
274         
275         // Calculate new value
276         T newValue = null;
277         Exception calculationException = null;
278         boolean calculationSucceeded = false;
279         try {
280             // Tries to calculate the value
281             newValue = readFunction.readValue();
282             if (newValue != null) {
283                 NotificationSupport.findNotificationSupportFor(newValue);
284             }
285             calculationSucceeded = true;
286         } catch (RuntimeException ex) {
287             // Calculation failed
288             calculationException = ex;
289         } catch (Throwable ex) {
290             log.log(Level.SEVERE, "Unrecoverable error during scanning", ex);
291         }
292         
293         // Calculate new connection
294         final boolean connected = readConnCollector.readValue();
295         List<Exception> exceptions = readExceptionCollector.readValue();
296         final Exception lastException;
297         if (calculationException != null) {
298             lastException = calculationException;
299         } else if (exceptions.isEmpty()) {
300             lastException = null;
301         } else {
302             lastException = exceptions.get(exceptions.size() - 1);
303         }
304         
305         // TODO: if payload is immutable, the difference test should be done here
306         // and not in the runnable (to save SWT time)
307         
308         // Prepare values to ship to the other thread.
309         // The data will be shipped as part of the task,
310         // which is properly synchronized by the executor
311         final T finalValue = newValue;
312         final boolean finalCalculationSucceeded = calculationSucceeded;
313         notificationInFlight = true;
314         notificationExecutor.execute(new Runnable() {
315 
316             @Override
317             public void run() {
318                 try {
319                     PVReaderImpl<T> pv = pvReaderRef.get();
320                     // Proceed with notification only if PVReader was not garbage
321                     // collected
322                     if (pv != null) {
323                         
324                         // Atomicity guaranteed by:
325                         //  - all the modification on the PVReader
326                         //    are done here, on the same thread where the listeners will be called.
327                         //    This means the callbacks are guaranteed to run after all
328                         //    changes are done
329                         //  - notificationInFlight guarantees that no other notification
330                         //    will run while one notification is running. This means
331                         //    the next event is serialized after the end of this one.
332                         pv.setConnected(connected);
333                         if (lastException != null) {
334                             pv.setLastException(lastException);
335                         }
336                         
337                         // XXX Are we sure that we should skip notifications if values are null?
338                         if (finalCalculationSucceeded && finalValue != null) {
339                             Notification<T> notification =
340                                     NotificationSupport.notification(pv.getValue(), finalValue);
341                             // Remember to notify anyway if an exception need to be notified
342                             if (notification.isNotificationNeeded()) {
343                                 pv.setValue(notification.getNewValue());
344                             } else if (pv.isLastExceptionToNotify() || pv.isReadConnectionToNotify()) {
345                                 pv.firePvValueChanged();
346                             }
347                         } else {
348                             // Remember to notify anyway if an exception need to be notified
349                             if (pv.isLastExceptionToNotify() || pv.isReadConnectionToNotify()) {
350                                 pv.firePvValueChanged();
351                             }
352                         }
353                     }
354                 } finally {
355                     notificationInFlight = false;
356                     scanStrategy.readyForNextEvent();
357                 }
358             }
359         });
360     }

362     /**
363      * Posts a readTimeout exception in the exception queue.
364      * 
365      * @param timeoutMessage the message for the readTimeout
366      */
367     private void processReadTimeout(String timeoutMessage) {
368         PVReaderImpl<T> pv = pvReaderRef.get();
369         if (pv != null && !pv.isSentFirsEvent()) {
370             readExceptionCollector.writeValue(new TimeoutException(timeoutMessage));
371         }
372     }
373     
374     void readTimeout(TimeDuration timeout, final String timeoutMessage) {
375         scannerExecutor.schedule(new Runnable() {
376             @Override
377             public void run() {
378                 processReadTimeout(timeoutMessage);
379             }
380         }, timeout.toNanosLong(), TimeUnit.NANOSECONDS);
381     }

383     private final DesiredRateEventListener desiredRateEventListener = new DesiredRateEventListener() {
384 
385         @Override
386         public void desiredRateEvent(DesiredRateEvent event) {
387             if (isActive()) {
388                 notifyPv();
389             } else {
390                 close();
391             }
392         }
393 
394     };
395 
396     DesiredRateEventListener getDesiredRateEventListener() {
397         return desiredRateEventListener;

5   package org.epics.pvmanager.formula;
6   
7   import org.epics.vtype.ValueFactory;
8   import java.util.Arrays;
9   import java.util.List;
10  import java.util.Objects;
11  import org.epics.util.array.ListDouble;
12  import org.epics.util.array.ListInt;
13  import org.epics.vtype.VString;
14  import org.epics.vtype.VTable;
15  import org.epics.vtype.VType;
16  
17  /**
18   * Extracts a columns from a VTable.
19   *
20   * @author carcassi
21   */
22  class ColumnOfVTableFunction implements FormulaFunction {
23  
24      @Override
25      public boolean isPure() {
26          return true;
27      }
28  
29      @Override
30      public boolean isVarArgs() {
31          return false;
32      }
33  
34      @Override
35      public String getName() {
36          return "columnOf";
37      }
38  
39      @Override
40      public String getDescription() {
41          return "Extracts a column from the given table";
42      }
43  
44      @Override
45      public List<Class<?>> getArgumentTypes() {
46          return Arrays.<Class<?>>asList(VTable.class, VString.class);
47      }
48  
49      @Override
50      public List<String> getArgumentNames() {
51          return Arrays.asList("table", "columName");
52      }
53  
54      @Override
55      public Class<?> getReturnType() {
56          return VType.class;
57      }
58  
59      @Override
60      public Object calculate(final List<Object> args) {
61          VTable table = (VTable) args.get(0);
62          VString columnName = (VString) args.get(1);
63          
64          if (columnName == null || table == null) {
65              return null;
66          }
67          
68          int index = -1;
69          for (int i = 0; i < table.getColumnCount(); i++) {
70              if (Objects.equals(columnName.getValue(), table.getColumnName(i))) {
71                  index = i;
72              }
73          }
74          if (index == -1) {
75              throw new RuntimeException("Table does not contain column '" + columnName.getValue() + "'");
76          }
77          
78          Class<?> type = table.getColumnType(index);
79          
80          if (String.class.isAssignableFrom(type)) {
81              @SuppressWarnings("unchecked")
82              List<String> data = (List<String>) table.getColumnData(index);
83              return ValueFactory.newVStringArray(data, ValueFactory.alarmNone(), ValueFactory.timeNow());
84          }
85          
86          if (Double.TYPE.isAssignableFrom(type)) {
87              ListDouble data = (ListDouble) table.getColumnData(index);
88              return ValueFactory.newVDoubleArray(data, ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
89          }
90          
91          if (Integer.TYPE.isAssignableFrom(type)) {
92              ListInt data = (ListInt) table.getColumnData(index);
93              return ValueFactory.newVIntArray(data, ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
94          }
95          
96          throw new RuntimeException("Unsupported type " + type.getSimpleName());
97      }
98      
99  }

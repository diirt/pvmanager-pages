5   package org.epics.pvmanager.formula;
6   
7   /**
8    * A set of functions to work with Strings.
9    * 
10   * @author shroffk
11   * 
12   */
13  public class StringFunctionSet extends FormulaFunctionSet {
14  
15      /**
16       * Creates a new set.
17       */
18      public StringFunctionSet() {
19  	super(new FormulaFunctionSetDescription("string",
20  		"Function to aggregate and manipulate strings")
21  		.addFormulaFunction(new ConcatStringArrayFunction())
22  		.addFormulaFunction(new ConcatStringsFunction())
23  		.addFormulaFunction(new PvFormulaFunction())
24  		.addFormulaFunction(new PvsFormulaFunction())
25  		.addFormulaFunction(new HighestSeverityFunction())
26  		.addFormulaFunction(new ToStringFunction())
27                  );
28      }
29  
30  }

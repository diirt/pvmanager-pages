5   package org.epics.pvmanager.formula;
6   
7   import java.util.Arrays;
8   import java.util.List;
9   import org.epics.vtype.VNumber;
10  import org.epics.vtype.table.ListNumberProvider;
11  import org.epics.vtype.table.VTableFactory;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  class RangeFormulaFunction implements FormulaFunction {
18  
19      @Override
20      public boolean isPure() {
21          return true;
22      }
23  
24      @Override
25      public boolean isVarArgs() {
26          return false;
27      }
28  
29      @Override
30      public String getName() {
31          return "range";
32      }
33  
34      @Override
35      public String getDescription() {
36          return "A generator for values between a range";
37      }
38  
39      @Override
40      public List<Class<?>> getArgumentTypes() {
41          return Arrays.<Class<?>>asList(VNumber.class, VNumber.class);
42      }
43  
44      @Override
45      public List<String> getArgumentNames() {
46          return Arrays.asList("minValue", "maxValue");
47      }
48  
49      @Override
50      public Class<?> getReturnType() {
51          return ListNumberProvider.class;
52      }
53  
54      @Override
55      public Object calculate(final List<Object> args) {
56          VNumber minValue = (VNumber) args.get(0);
57          VNumber maxValue = (VNumber) args.get(1);
58          
59          if (minValue == null || maxValue == null) {
60              return null;
61          }
62          
63          return VTableFactory.range(minValue.getValue().doubleValue(), maxValue.getValue().doubleValue());
64      }
65      
66  }

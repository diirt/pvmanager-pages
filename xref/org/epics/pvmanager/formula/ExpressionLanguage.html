26   * @author carcassi

28  public class ExpressionLanguage {
29      private ExpressionLanguage() {
30          // No instances
31      }

33      static FormulaParser createParser(String text) {
34          CharStream stream = new ANTLRStringStream(text);
35          FormulaLexer lexer = new FormulaLexer(stream);
36          TokenStream tokenStream = new CommonTokenStream(lexer);
37          return new FormulaParser(tokenStream);
38      }
39      
40      /**
41       * If the formula represents a single channels it returns the name,
42       * null otherwise.
43       * 
44       * @param formula the formula to parse
45       * @return the channel it represents or null
46       */
47      public static String channelFromFormula(String formula) {
48          if (!formula.startsWith("=")) {
49              if (formula.trim().matches(StringUtil.SINGLEQUOTED_STRING_REGEX)) {
50                  return StringUtil.unquote(formula);
51              }
52              return formula;
53          } else {
54              formula = formula.substring(1);
55          }
56          try {
57              FormulaParser parser = createParser(formula);
58              DesiredRateExpression<?> exp = parser.singlePv();
59              if (parser.failed()) {
60                  return null;
61              }
62              if (exp instanceof LastOfChannelExpression) {
63                  LastOfChannelExpression channelExp = (LastOfChannelExpression) exp;
64                  return channelExp.getName();
65              }
66              return null;
67          } catch (RecognitionException ex) {
68              return null;
69          } catch (Exception ex) {
70              return null;
71          }
72      }
73  
74      /**
75       * Returns the expression that will return the live value of the
76       * given formula.
77       * 
78       * @param formula the formula to parse
79       * @return an expression for the formula
80       */
81      public static DesiredRateReadWriteExpression<?, Object> formula(String formula) {
82          DesiredRateExpression<?> exp = parseFormula(formula);
83              
84          if (exp instanceof LastOfChannelExpression) {
85              return new DesiredRateReadWriteExpressionImpl<>(exp, org.epics.pvmanager.vtype.ExpressionLanguage.vType(exp.getName()));
86          } else if (exp instanceof ErrorDesiredRateExpression) {
87              return new DesiredRateReadWriteExpressionImpl<>(exp, readOnlyWriteExpression("Parsing error")); 
88          } else {
89              return new DesiredRateReadWriteExpressionImpl<>(exp, readOnlyWriteExpression("Read-only formula"));
90          }
91      }
92      
93      private static DesiredRateExpression<?> parseFormula(String formula) {
94          if (!formula.startsWith("=")) {
95              return cachedPv(channelFromFormula(formula));
96          } else {
97              formula = formula.substring(1);
98          }
99          
100         RuntimeException parsingError;
101         try {
102             DesiredRateExpression<?> exp = createParser(formula).formula();
103             if (exp == null) {
104                 throw new NullPointerException("Parsing failed");
105             }
106             return exp;
107         } catch (RecognitionException ex) {
108             parsingError = new IllegalArgumentException("Error parsing formula: " + ex.getMessage(), ex);
109         } catch (Exception ex) {
110             parsingError = new IllegalArgumentException("Malformed formula '" + formula + "'", ex);
111         }
112         return errorDesiredRateExpression(parsingError); 
113     }

115     /**
116      * An expression that returns the value of the formula and return null
117      * for empty or null formula.
118      * <p>
119      * Some expressions allow for null expression arguments to handle
120      * optional elements. In those cases, using this method makes
121      * undeclared arguments fall through.
122      * 
123      * @param formula the formula, can be null
124      * @return an expression of the given type; null if formula is null or empty
125      */
126     public static DesiredRateExpression<?> formulaArg(String formula) {
127         if (formula == null || formula.trim().isEmpty()) {
128             return null;
129         }
130         
131         return parseFormula(formula);
132     }
133     
134     /**
135      * An expression that returns the value of the formula making sure
136      * it's of the given type.
137      * 
138      * @param <T> the type to read
139      * @param formula the formula
140      * @param readType the type to read
141      * @return an expression of the given type
142      */
143     public static <T> DesiredRateExpression<T> formula(String formula, Class<T> readType) {
144         DesiredRateExpression<?> exp = parseFormula(formula);
145         return checkReturnType(readType, "Value", exp);
146     }

148     static DesiredRateExpression<?> cachedPv(String channelName) {
149         return new LastOfChannelExpression<>(channelName, Object.class);
150     }
151     
152     static <T> DesiredRateExpression<T> cast(Class<T> clazz, DesiredRateExpression<?> arg1) {
153         if (arg1 instanceof LastOfChannelExpression) {
154             return ((LastOfChannelExpression<?>)arg1).cast(clazz);
155         }
156         @SuppressWarnings("unchecked")
157         DesiredRateExpression<T> op1 = (DesiredRateExpression<T>) arg1;
158         return op1;
159     }
160     
161     static <T> DesiredRateExpressionList<T> cast(Class<T> clazz, DesiredRateExpressionList<?> args) {
162         for (DesiredRateExpression<? extends Object> desiredRateExpression : args.getDesiredRateExpressions()) {
163             cast(clazz, desiredRateExpression);
164         }
165         @SuppressWarnings("unchecked")
166         DesiredRateExpressionList<T> op1 = (DesiredRateExpressionList<T>) args;
167         return op1;
168     }

170     static String opName(String op, DesiredRateExpression<?> arg1, DesiredRateExpression<?> arg2) {
171         return "(" + arg1.getName() + op + arg2.getName() + ")";
172     }

174     static String opName(String op, DesiredRateExpression<?> arg) {
175         return op + arg.getName();
176     }
177     
178     static String funName(String fun, DesiredRateExpression<?> arg) {
179         return fun + "(" + arg.getName()+ ")";
180     }

182     static DesiredRateExpression<?> powCast(DesiredRateExpression<?> arg1, DesiredRateExpression<?> arg2) {
183         return function("^", new DesiredRateExpressionListImpl<Object>().and(arg1).and(arg2));
184     }

186     static DesiredRateExpression<?> threeArgOp(String opName, DesiredRateExpression<?> arg1, DesiredRateExpression<?> arg2, DesiredRateExpression<?> arg3) {
187         return function(opName, new DesiredRateExpressionListImpl<Object>().and(arg1).and(arg2).and(arg3));
188     }
189 
190     static DesiredRateExpression<?> twoArgOp(String opName, DesiredRateExpression<?> arg1, DesiredRateExpression<?> arg2) {
191         return function(opName, new DesiredRateExpressionListImpl<Object>().and(arg1).and(arg2));
192     }
193 
194     static DesiredRateExpression<?> oneArgOp(String opName, DesiredRateExpression<?> arg) {
195         return function(opName, new DesiredRateExpressionListImpl<Object>().and(arg));
196     }
197     
198     static DesiredRateExpression<?> function(String function, DesiredRateExpressionList<?> args) {
199         Collection<FormulaFunction> matchedFunctions = FormulaRegistry.getDefault().findFunctions(function, args.getDesiredRateExpressions().size());
200         if (matchedFunctions.size() > 0) {
201             FormulaReadFunction readFunction = new FormulaReadFunction(Expressions.functionsOf(args), matchedFunctions);
202             List<String> argNames = new ArrayList<>(args.getDesiredRateExpressions().size());
203             for (DesiredRateExpression<? extends Object> arg : args.getDesiredRateExpressions()) {
204                 argNames.add(arg.getName());
205             }
206             return new FormulaFunctionReadExpression(args, readFunction, FormulaFunctions.format(function, argNames));
207         }
208         
209         throw new IllegalArgumentException("No function named '" + function + "' is defined");
210     }
211     
212     static <T> WriteExpression<T> readOnlyWriteExpression(String errorMessage) {
213         return new ReadOnlyWriteExpression<>(errorMessage, "");
214     }
215     
216     static <T> DesiredRateExpression<T> errorDesiredRateExpression(RuntimeException error) {
217         return new ErrorDesiredRateExpression<>(error, "");
218     }
219     
220     static <T> DesiredRateExpression<T> checkReturnType(final Class<T> clazz, final String argName, final DesiredRateExpression<?> arg1) {
221         return new DesiredRateExpressionImpl<T>(arg1, new ReadFunction<T>() {
222 
223             @Override
224             public T readValue() {
225                 Object obj = arg1.getFunction().readValue();
226                 if (obj == null) {
227                     return null;
228                 }
229                 
230                 if (clazz.isInstance(obj)) {
231                     return clazz.cast(obj);
232                 } else {
233                     throw new RuntimeException(argName + " must be a " + clazz.getSimpleName() + " (was " + ValueUtil.typeOf(obj).getSimpleName() + ")");
234                 }
235             }
236         }, arg1.getName());
237     }

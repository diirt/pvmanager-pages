5   package org.epics.pvmanager.formula;
6   
7   import java.util.Arrays;
8   import java.util.List;
9   import java.util.Objects;
10  import org.epics.pvmanager.expression.DesiredRateExpression;
11  import org.epics.vtype.VString;
12  
13  /**
14   * Formula function that returns the value of the pv matching the name
15   * of the argument.
16   *
17   * @author carcassi
18   */
19  class PvFormulaFunction extends DynamicFormulaFunction {
20  
21      @Override
22      public boolean isVarArgs() {
23          return false;
24      }
25  
26      @Override
27      public String getName() {
28          return "pv";
29      }
30  
31      @Override
32      public String getDescription() {
33          return "Returns the value of the given pv name";
34      }
35  
36      @Override
37      public List<Class<?>> getArgumentTypes() {
38          return Arrays.<Class<?>>asList(VString.class);
39      }
40  
41      @Override
42      public List<String> getArgumentNames() {
43          return Arrays.asList("pvName");
44      }
45  
46      @Override
47      public Class<?> getReturnType() {
48          return Object.class;
49      }
50      
51      private String previousName;
52      private DesiredRateExpression<?> currentExpression;
53  
54      @Override
55      public Object calculate(final List<Object> args) {
56          // Retrieve the new name
57          VString value = (VString) args.get(0);
58          String newName = null;
59          if (value != null) {
60              newName = value.getValue();
61          }
62          
63          // If the name does not match, disconnect and connect
64          if (!Objects.equals(newName, previousName)) {
65              // Disconnect previous
66              if (currentExpression != null) {
67                  getDirector().disconnectReadExpression(currentExpression);
68                  currentExpression = null;
69              }
70              
71              // Connect new
72              if (newName != null) {
73                  currentExpression = new LastOfChannelExpression<Object>(newName, Object.class);
74                  getDirector().connectReadExpression(currentExpression);
75              }
76              previousName = newName;
77          }
78  
79          // Return value
80          if (newName == null) {
81              return null;
82          }
83          return currentExpression.getFunction().readValue();
84      }
85  
86      @Override
87      public void dispose() {
88          // Disconnect everything on dispose
89          if (currentExpression != null) {
90              getDirector().disconnectReadExpression(currentExpression);
91          }
92          currentExpression = null;
93      }
94      
95  }

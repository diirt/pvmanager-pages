5   package org.epics.pvmanager.formula;
6   
7   import java.util.ArrayList;
8   import java.util.Collections;
9   import java.util.Comparator;
10  import java.util.List;
11  
12  /**
13   * Utility class to print an prefilled HTML table from a function set.
14   * 
15   * @author carcassi
16   */
17  public class PrintFunctionSetTable {
18      
19      /**
20       * Entry point for the utility.
21       * 
22       * @param args command-line arguments
23       */
24      public static void main(String[] args) {
25          FormulaFunctionSet set = new ArrayFunctionSet();
26          List<FormulaFunction> functions = new ArrayList<>(set.getFunctions());
27          Collections.sort(functions, new Comparator<FormulaFunction>() {
28  
29              @Override
30              public int compare(FormulaFunction o1, FormulaFunction o2) {
31                  return o1.getName().compareTo(o2.getName());
32              }
33          });
34          
35          System.out.println("        <table border=\"1\" style=\"border-collapse: collapse\">\n" +
36  "            <thead>\n" +
37  "                <tr>\n" +
38  "                    <th>Signature</th>\n" +
39  "                    <th>Description</th>\n" +
40  "                    <th>Null handling</th>\n" +
41  "                    <th>Alarm</th>\n" +
42  "                    <th>Time</th>\n" +
43  "                </tr>\n" +
44  "            </thead>\n" +
45  "            <tbody>");
46          
47          for (FormulaFunction formulaFunction : functions) {
48              System.out.println("                <tr>");
49              System.out.println("                    <td><code>" + FormulaFunctions.formatSignature(formulaFunction) + "</code></td>");
50              System.out.println("                    <td>" + formulaFunction.getDescription() + "</td>");
51              System.out.println("                    <td>Null if one of the arguments is null</td>");
52              System.out.println("                    <td>Highest alarm of the arguments</td>");
53              System.out.println("                    <td>Latest valid time of the arguments or now if no valid time is found</td>");
54              System.out.println("                </tr>");
55          }
56          
57          System.out.println("            </tbody>\n" +
58  "        </table>");
59      }
60  }

5   package org.epics.pvmanager.formula;
6   
7   import org.epics.pvmanager.PVDirector;
8   
9   /**
10   * Formula function that can add and remove dynamically access to
11   * pvs.
12   * <p>
13   * This formula function is given a director which can be used to open/close
14   * expressions that read real-time data.
15   *
16   * @author carcassi
17   */
18  public abstract class DynamicFormulaFunction extends StatefulFormulaFunction {
19      
20      private PVDirector<?> director;
21  
22      /**
23       * The director to use to connect/disconnect live data expressions.
24       * 
25       * @return the director
26       */
27      public final PVDirector<?> getDirector() {
28          return director;
29      }
30      
31      /**
32       * Changes the director. This is not part of the public API: the director
33       * is set by the infrastructure.
34       * 
35       * @param director the new director
36       */
37      void setDirector(PVDirector<?> director) {
38          this.director = director;
39      }
40      
41  }

5   package org.epics.pvmanager.service;
6   
7   import java.util.HashMap;
8   import java.util.Map;
9   import static org.epics.pvmanager.service.Service.namePattern;
10  
11  /**
12   *
13   * @author carcassi
14   */
15  public class ServiceMethodDescription {
16      
17      String name;
18      String description;
19      Map<String, Class<?>> argumentTypes = new HashMap<>();
20      Map<String, String> argumentDescriptions = new HashMap<>();
21      Map<String, Class<?>> resultTypes = new HashMap<>();
22      Map<String, String> resultDescriptions = new HashMap<>();
23  
24      public ServiceMethodDescription(String name, String description) {
25          this.name = name;
26          this.description = description;
27          if (!namePattern.matcher(name).matches()) {
28              throw new IllegalArgumentException("Name must start by a letter and only consist of letters and numbers");
29          }
30      }
31      
32      public ServiceMethodDescription addArgument(String name, String description, Class<?> type) {
33          if (!namePattern.matcher(name).matches()) {
34              throw new IllegalArgumentException("Name must start by a letter and only consist of letters and numbers");
35          }
36          argumentTypes.put(name, type);
37          argumentDescriptions.put(name, description);
38          return this;
39      }
40      
41      public ServiceMethodDescription addResult(String name, String description, Class<?> type) {
42          if (!namePattern.matcher(name).matches()) {
43              throw new IllegalArgumentException("Name must start by a letter and only consist of letters and numbers");
44          }
45          resultTypes.put(name, type);
46          resultDescriptions.put(name, description);
47          return this;
48      }
49  }

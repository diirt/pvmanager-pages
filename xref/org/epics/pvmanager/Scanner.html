6   package org.epics.pvmanager;
7   
8   /**
9    * Implements the strategy for event generation and event response.
10   * <p>
11   * An event should be always triggered in the following cases:
12   * <ul>
13   *   <li>If the expression changes even without external events (e.g. time average),
14   *       then the calculation needs to happen at the scan rate</li>
15   *   <li>Every time a channel connection or value changes</li>
16   *   <li>Every time the expression dynamically changes (e.g. dynamic map, indirect pv)</li>
17   * <ul>
18   *
19   * @author carcassi
20   */
21  interface Scanner {
22      
23      /**
24       * Starts the scanning. From this moment on, the pv may get notified.
25       */
26      void start();
27      
28      /**
29       * Pause the scanning. Events will be collected and delayed until a resume.
30       */
31      void pause();
32      
33      /**
34       * Resumes the scanning. If events were collected during the pause,
35       * they will be sent right away.
36       */
37      void resume();
38      
39      /**
40       * Stops the scanning. From this moment on, the pv will no longer be
41       * notified. Can't be restarted.
42       */
43      void stop();
44      
45      /**
46       * Called when a new notification reaches a collector.
47       */
48      void collectorChange();
49      
50      /**
51       * Called after a pv is notified.
52       */
53      void notifiedPv();
54  }

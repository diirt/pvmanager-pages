89     * Creates a new rpcservice method.
90     *
91     * @param rpcServiceMethodDescription a method description
92     */
93    RPCServiceMethod(RPCServiceMethodDescription rpcServiceMethodDescription, String hostName, String channelName,
94  		  String methodFieldName, boolean useNTQuery) {
95      super(rpcServiceMethodDescription.relaxedServiceMethodDescription);
96      //super(rpcServiceMethodDescription.serviceMethodDescription);
97      this.rpcServiceMethodDescription = rpcServiceMethodDescription;
98      this.parameterNames = rpcServiceMethodDescription.orderedParameterNames;
99      this.hostName = hostName;
100     this.channelName = channelName;
101     this.methodFieldName = methodFieldName;
102     this.useNTQuery = useNTQuery;
103 
104     this.requestStructure = createRequestStructure(rpcServiceMethodDescription.structureId);
105   }

107   private Structure createRequestStructure(String structureId) {
108 	  Structure paramStructure = createParametersRequestStructure(structureId);
109 
110 	  if (!useNTQuery)
111 		  return paramStructure;
112 	  else
113 	  {
114 		  return fieldCreate.createStructure("epics:nt/NTURI:1.0", 
115 						  new String[] { "scheme", "path", "query" },
116 						  new Field[] { 
117 						  	fieldCreate.createScalar(ScalarType.pvString),
118 						  	fieldCreate.createScalar(ScalarType.pvString),
119 						  	paramStructure,
120 				  });
121 	  }
122   }

124   private Structure createParametersRequestStructure(String structureId) {
125     if ((structureId == null) || (structureId.isEmpty())) {
126       return fieldCreate.createStructure(createRequestFieldNames(), createRequestFieldTypes());
127     } else
128       return fieldCreate.createStructure(structureId, createRequestFieldNames(), createRequestFieldTypes());
129   }

132   private String[] createRequestFieldNames() {
133 
134     //only operation name, if specified
135     if ((this.parameterNames == null) || (parameterNames.isEmpty())) {
136       return methodFieldName != null ? new String[]{ methodFieldName } : new String[0];
137     }
138 
139     //operation name (optional) + parameter names/fieldnames
140     List<String> fieldNames = new ArrayList<String>(this.parameterNames.size() + 1);
141     if (methodFieldName != null)
142     	fieldNames.add(methodFieldName);
143     for (String parameterName : this.parameterNames.keySet()) {
144       String fieldName = this.parameterNames.get(parameterName);
145       if (fieldName.equals(RPCServiceMethodDescription.FIELD_NAME_EQUALS_NAME)) {
146         fieldNames.add(parameterName);
147       } else {
148         fieldNames.add(fieldName);
149       }
150     }
151 
152     return fieldNames.toArray(new String[fieldNames.size()]);
153   }

156   private Field[] createRequestFieldTypes() {
157 
158     //only operation name type
159     if ((this.parameterNames == null) || (parameterNames.isEmpty())) {
160     	return methodFieldName != null ? new Field[]{fieldCreate.createScalar(ScalarType.pvString)} : new Field[0];
161     }
162 
163     //Map<String, Class<?>> argumentTypes = getArgumentTypes();
164     // create ServiceMethod to get access to argumentTypes of serviceMethodDescription (non-relaxed)
165     ServiceMethod sm = new ServiceMethod(rpcServiceMethodDescription.serviceMethodDescription) {
166 		@Override
167 		public void executeMethod(Map<String, Object> parameters,
168 				WriteFunction<Map<String, Object>> callback,
169 				WriteFunction<Exception> errorCallback) {
170 			// noop
171 		}
172     };
173     Map<String, Class<?>> argumentTypes = sm.getArgumentTypes();
174     
175     // operation name type + parameter types
176     List<Field> fieldList = new ArrayList<Field>(this.parameterNames.size() + 1);
177     if (methodFieldName != null)
178     	fieldList.add(fieldCreate.createScalar(ScalarType.pvString));
179     for (String parameterName : this.parameterNames.keySet()) {
180       fieldList.add(convertToPvType(argumentTypes.get(parameterName)));
181     }
182 
183     return fieldList.toArray(new Field[fieldList.size()]);
184   }

186   private Field convertToPvType(Class<?> argType) {
187 	  return NTUtils.vtypeToField(argType);

196   private boolean isResultQuery() {
197     return !getResultDescriptions().isEmpty();

201   @Override
202   public void executeMethod(final Map<String, Object> parameters, final WriteFunction<Map<String, Object>> callback, final WriteFunction<Exception> errorCallback) {
203 
204     RPCClient rpcClient = null;
205 
206     try {
207 
208       rpcClient = PooledRPCClientFactory.getRPCClient(this.hostName, this.channelName);
209 
210     } catch (RPCRequestException e) {
211       errorCallback.writeValue(e);
212       return;
213     }
214 
215 
216     try {
217       String methodName = this.rpcServiceMethodDescription.getOperationName() != null ?
218         this.rpcServiceMethodDescription.getOperationName() : this.rpcServiceMethodDescription.getName();
219 
220       PVStructure pvRequest = createPvRequest(parameters, methodName);
221       PVStructure pvResult = rpcClient.request(pvRequest, 3.0);
222 
223       VType vResult = createResult(pvResult);
224       if (vResult != null) {
225         Map<String, Object> resultMap = new HashMap<>();
226         String resultName = getResultDescriptions().keySet().toArray(new String[getResultDescriptions().size()])[0];
227         resultMap.put(resultName, vResult);
228         callback.writeValue(resultMap);
229       }
230     } catch (RPCRequestException rre) {
231       errorCallback.writeValue(rre);
232 
233     } finally {
234       //back to the pool
235       rpcClient.destroy();
236     }
237   }

240   PVStructure createPvRequest(final Map<String, Object> parameters, String methodName) {
241 
242     PVStructure pvRequest = PVDataFactory.getPVDataCreate().createPVStructure(this.requestStructure);
243     PVStructure retVal = pvRequest;
244     if (useNTQuery)
245     {
246     	pvRequest.getStringField("scheme").put("pva");
247     	pvRequest.getStringField("path").put(channelName);
248     	pvRequest = pvRequest.getStructureField("query");
249     }
250 
251     if (methodFieldName != null)
252     	pvRequest.getStringField(methodFieldName).put(methodName);
253 
254     if ((parameters == null) || (parameters.isEmpty())) {
255       return retVal;
256     }
257 
258     for (String parameterName : parameters.keySet()) {
259 
260       //check if the parameter name is only an alias (if we get a field name back, it's an alias)
261       String fieldName = this.parameterNames.get(parameterName);
262 
263       if ((fieldName == null) || (fieldName.equals(RPCServiceMethodDescription.FIELD_NAME_EQUALS_NAME))) {
264         fieldName = null;
265       }
266 
267       Object value = parameters.get(parameterName);
268       
269       // consider it as optional
270       if (value == null)
271     	  continue;
272 
273       PVField pvField = pvRequest.getSubField((fieldName != null ? fieldName : parameterName));
274       
275       
276       if (pvField instanceof PVScalar)
277       {
278     	  PVScalar pvScalar = (PVScalar)pvField;
279     	  if (value instanceof VDouble)
280     		  convert.fromDouble(pvScalar, ((VDouble)value).getValue());
281     	  else if (value instanceof VInt)
282     		  convert.fromInt(pvScalar, ((VInt)value).getValue());
283     	  else if (value instanceof VFloat)
284     		  convert.fromFloat(pvScalar, ((VFloat)value).getValue());
285     	  else if (value instanceof VLong)
286     		  convert.fromLong(pvScalar, ((VLong)value).getValue());
287     	  else if (value instanceof VShort)
288     		  convert.fromShort(pvScalar, ((VShort)value).getValue());
289     	  else if (value instanceof VByte)
290     		  convert.fromByte(pvScalar, ((VByte)value).getValue());
291     	  else if (value instanceof VBoolean)
292     		  // Convert is missing fromBoolean
293     	  	  convert.fromByte(pvScalar, ((VBoolean)value).getValue() ? (byte)1 : (byte)0);
294        	  else if (value instanceof VString)
295     		  convert.fromString(pvScalar, ((VString)value).getValue());
296       }
297       else if (pvField instanceof PVScalarArray)
298       {
299     	  PVScalarArray pvScalarArray = (PVScalarArray)pvField;
300     	  if (value instanceof VDoubleArray)
301     	  {
302     		  ListDouble list = ((VDoubleArray)value).getData();
303               double[] arr = CollectionNumbers.doubleArrayWrappedOrCopy(list);
304     		  convert.fromDoubleArray(pvScalarArray, 0, arr.length, arr, 0);
305     	  }
306     	  else if (value instanceof VIntArray)
307     	  {
308     		  ListInt list = ((VIntArray)value).getData();
309               int[] arr = CollectionNumbers.intArrayWrappedOrCopy(list);
310     		  convert.fromIntArray(pvScalarArray, 0, arr.length, arr, 0);
311     	  }
312     	  else if (value instanceof VFloatArray)
313     	  {
314     		  ListFloat list = ((VFloatArray)value).getData();
315               float[] arr = CollectionNumbers.floatArrayWrappedOrCopy(list);
316     		  convert.fromFloatArray(pvScalarArray, 0, arr.length, arr, 0);
317     	  }
318     	  else if (value instanceof VLongArray)
319     	  {
320     		  ListLong list = ((VLongArray)value).getData();
321               long[] arr = CollectionNumbers.longArrayWrappedOrCopy(list);
322     		  convert.fromLongArray(pvScalarArray, 0, arr.length, arr, 0);
323     	  }
324     	  else if (value instanceof VShortArray)
325     	  {
326     		  ListShort list = ((VShortArray)value).getData();
327               short[] arr = CollectionNumbers.shortArrayWrappedOrCopy(list);
328     		  convert.fromShortArray(pvScalarArray, 0, arr.length, arr, 0);
329     	  }
330     	  else if (value instanceof VByteArray)
331     	  {
332     		  ListByte list = ((VByteArray)value).getData();
333               byte[] arr = CollectionNumbers.byteArrayWrappedOrCopy(list);
334     		  convert.fromByteArray(pvScalarArray, 0, arr.length, arr, 0);
335     	  }
336        	  else if (value instanceof VStringArray)
337     	  {
338     		  List<String> list = ((VStringArray)value).getData();
339     		  String[] arr = list.toArray(new String[list.size()]);
340     		  convert.fromStringArray(pvScalarArray, 0, arr.length, arr, 0);
341     	  }
342        	  else if (value instanceof VBooleanArray)
343     	  {
344     		  ListBoolean list = ((VBooleanArray)value).getData();
345     		  int len = list.size();
346               byte[] arr = new byte[len];
347               for (int i = 0; i < len; i++)
348             	  arr[i] = list.getBoolean(i) ? (byte)1 : (byte)0;
349     		  convert.fromByteArray(pvScalarArray, 0, len, arr, 0);
350     	  }
351       }
352       else
353       {
354          // TODO complex types (do we need to support them?)
355         throw new RuntimeException("pvAccess RPC Service mapping support for " + value.getClass().getSimpleName() + " not implemented");
356       }
357     }
358     
359     return retVal;
360   }

363   private VType createResult(PVStructure pvResult) {
364 
365     if (pvResult == null) {
366       return null;
367     }
368 
369     if (!isResultQuery()) {
370       return null;
371     }
372 
373     Class<?> resultType = getResultTypes().values().toArray(new Class<?>[getResultTypes().size()])[0];
374 
375     if (this.rpcServiceMethodDescription.isResultStandalone) {
376       if (resultType.isAssignableFrom(VTable.class)) {
377 
378         if ("epics:nt/NTNameValue:1.0".equals(pvResult.getStructure().getID()))
379           return new PVFieldNTNameValueToVTable(pvResult, false);
380         else
381         	return new PVFieldToVTable(pvResult, false);
382         
383         // TODO !!!
384       //} else if (resultType.isAssignableFrom(VImage.class)) {
385       //  return new PVFieldToVImage(pvResult, false);
386       } else if (resultType.isAssignableFrom(VStatistics.class)) {
387           return new PVFieldToVStatistics(pvResult, false);
388       } else if (resultType.isAssignableFrom(VDoubleArray.class) &&
389     		  "epics:nt/NTMatrix:1.0".equals(pvResult.getStructure().getID())) {
390           return new PVFieldNTMatrixToVDoubleArray(pvResult, false);
391       }
392 
393       throw new IllegalArgumentException("Standalone result type " + resultType.getSimpleName() + " not supported in pvAccess RPC rpcservice");
394     }
395 
396     // TODO unsigned types, complex types 
397 
398     String resultName = getResultDescriptions().keySet().toArray(new String[getResultDescriptions().size()])[0];
399     String fieldName = this.rpcServiceMethodDescription.getFieldNames().get(resultName);
400 
401     // TODO missing ValueFactory.newVLong
402     
403     if (resultType.isAssignableFrom(VDouble.class)) {
404         return ValueFactory.newVDouble(pvResult.getDoubleField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
405     } else if (resultType.isAssignableFrom(VFloat.class)) {
406         return ValueFactory.newVFloat(pvResult.getFloatField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
407     } else if (resultType.isAssignableFrom(VString.class)) {
408         return ValueFactory.newVString(pvResult.getStringField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow());
409     } else if (resultType.isAssignableFrom(VInt.class)) {
410         return ValueFactory.newVInt(pvResult.getIntField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
411     } else if (resultType.isAssignableFrom(VShort.class)) {
412         return ValueFactory.newVShort(pvResult.getShortField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
413     } else if (resultType.isAssignableFrom(VByte.class)) {
414         return ValueFactory.newVByte(pvResult.getByteField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
415     } else if (resultType.isAssignableFrom(VBoolean.class)) {
416         return ValueFactory.newVBoolean(pvResult.getBooleanField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow());
417     } else if (resultType.isAssignableFrom(VDoubleArray.class)) {
418     	
419 		if ("epics:nt/NTMatrix:1.0".equals(pvResult.getStructure().getID()))
420 	      return new PVFieldNTMatrixToVDoubleArray(pvResult, false);
421 	    else
422           return new PVFieldToVDoubleArray(pvResult, fieldName != null ? fieldName : resultName, true);
423 		
424     } else if (resultType.isAssignableFrom(VFloatArray.class)) {
425         return new PVFieldToVFloatArray(pvResult, fieldName != null ? fieldName : resultName, true);
426     } else if (resultType.isAssignableFrom(VIntArray.class)) {
427         return new PVFieldToVIntArray(pvResult, fieldName != null ? fieldName : resultName, true);
428     } else if (resultType.isAssignableFrom(VShortArray.class)) {
429         return new PVFieldToVShortArray(pvResult, fieldName != null ? fieldName : resultName, true);
430     } else if (resultType.isAssignableFrom(VByteArray.class)) {
431         return new PVFieldToVByteArray(pvResult, fieldName != null ? fieldName : resultName, true);
432     } else if (resultType.isAssignableFrom(VStringArray.class)) {
433         return new PVFieldToVStringArray(pvResult, fieldName != null ? fieldName : resultName, true);
434     } else if (resultType.isAssignableFrom(VTable.class)) {
435         if ("epics:nt/NTNameValue:1.0".equals(pvResult.getStructure().getID()))
436             return new PVFieldNTNameValueToVTable(pvResult, false);
437           else
438           	return new PVFieldToVTable(pvResult, false);
439         // TODO !!!
440     //} else if (resultType.isAssignableFrom(VImage.class)) {
441      //   return new PVFieldToVImage(pvResult,false);
442     } else if (resultType.isAssignableFrom(VStatistics.class)) {
443         return new PVFieldToVStatistics(pvResult,false);
444     }
445     
446     throw new IllegalArgumentException("Result type " + resultType.getSimpleName() + " not supported in pvAccess RPC rpcservice");
447 
448   }

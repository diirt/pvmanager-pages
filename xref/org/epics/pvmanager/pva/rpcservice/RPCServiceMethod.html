88    /**
89     * Creates a new rpcservice method.
90     *
91     * @param rpcServiceMethodDescription a method description
92     */
93    RPCServiceMethod(RPCServiceMethodDescription rpcServiceMethodDescription, String hostName, String channelName,
94  		  String methodFieldName, boolean useNTQuery) {
95      super(rpcServiceMethodDescription.relaxedServiceMethodDescription);
96      //super(rpcServiceMethodDescription.serviceMethodDescription);
97      this.rpcServiceMethodDescription = rpcServiceMethodDescription;
98      this.parameterNames = rpcServiceMethodDescription.orderedParameterNames;
99      this.hostName = hostName;
100     this.channelName = channelName;
101     this.methodFieldName = methodFieldName;
102     this.useNTQuery = useNTQuery;
103 
104     this.requestStructure = createRequestStructure(rpcServiceMethodDescription.structureId);
105   }

107   private Structure createRequestStructure(String structureId) {
108 	  Structure paramStructure = createParametersRequestStructure(structureId);
109 
110 	  if (!useNTQuery)
111 		  return paramStructure;
112 	  else
113 	  {
114 		  return fieldCreate.createStructure("uri:ev4:nt/2012/pwd:NTURI", 
115 						  new String[] { "scheme", "path", "query" },
116 						  new Field[] { 
117 						  	fieldCreate.createScalar(ScalarType.pvString),
118 						  	fieldCreate.createScalar(ScalarType.pvString),
119 						  	paramStructure,
120 				  });
121 	  }
122   }
123   
124   private Structure createParametersRequestStructure(String structureId) {
125     if ((structureId == null) || (structureId.isEmpty())) {
126       return fieldCreate.createStructure(createRequestFieldNames(), createRequestFieldTypes());
127     } else
128       return fieldCreate.createStructure(structureId, createRequestFieldNames(), createRequestFieldTypes());
129   }
130

132   private String[] createRequestFieldNames() {
133 
134     //only operation name, if specified
135     if ((this.parameterNames == null) || (parameterNames.isEmpty())) {
136       return methodFieldName != null ? new String[]{ methodFieldName } : new String[0];
137     }
138 
139     //operation name (optional) + parameter names/fieldnames
140     List<String> fieldNames = new ArrayList<String>(this.parameterNames.size() + 1);
141     if (methodFieldName != null)
142     	fieldNames.add(methodFieldName);
143     for (String parameterName : this.parameterNames.keySet()) {
144       String fieldName = this.parameterNames.get(parameterName);
145       if (fieldName.equals(RPCServiceMethodDescription.FIELD_NAME_EQUALS_NAME)) {
146         fieldNames.add(parameterName);
147       } else {
148         fieldNames.add(fieldName);
149       }
150     }
151 
152     return fieldNames.toArray(new String[fieldNames.size()]);
153   }

156   private Field[] createRequestFieldTypes() {
157 
158     //only operation name type
159     if ((this.parameterNames == null) || (parameterNames.isEmpty())) {
160     	return methodFieldName != null ? new Field[]{fieldCreate.createScalar(ScalarType.pvString)} : new Field[0];
161     }
162 
163     //Map<String, Class<?>> argumentTypes = getArgumentTypes();
164     // create ServiceMethod to get access to argumentTypes of serviceMethodDescription (non-relaxed)
165     ServiceMethod sm = new ServiceMethod(rpcServiceMethodDescription.serviceMethodDescription) {
166 		@Override
167 		public void executeMethod(Map<String, Object> parameters,
168 				WriteFunction<Map<String, Object>> callback,
169 				WriteFunction<Exception> errorCallback) {
170 			// noop
171 		}
172     };
173     Map<String, Class<?>> argumentTypes = sm.getArgumentTypes();
174     
175     // operation name type + parameter types
176     List<Field> fieldList = new ArrayList<Field>(this.parameterNames.size() + 1);
177     if (methodFieldName != null)
178     	fieldList.add(fieldCreate.createScalar(ScalarType.pvString));
179     for (String parameterName : this.parameterNames.keySet()) {
180       fieldList.add(convertToPvType(argumentTypes.get(parameterName)));
181     }
182 
183     return fieldList.toArray(new Field[fieldList.size()]);
184   }

187   private Field convertToPvType(Class<?> argType) {
188 
189     if (argType == null) {
190       throw new IllegalArgumentException("Type not set: null");
191     }
192 
193     // TODO no unsigned and complex types
194     
195     if (argType.isAssignableFrom(VDouble.class)) {
196       return fieldCreate.createScalar(ScalarType.pvDouble);
197     } else if (argType.isAssignableFrom(VFloat.class)) {
198       return fieldCreate.createScalar(ScalarType.pvFloat);
199     } else if (argType.isAssignableFrom(VString.class)) {
200       return fieldCreate.createScalar(ScalarType.pvString);
201     } else if (argType.isAssignableFrom(VInt.class)) {
202       return fieldCreate.createScalar(ScalarType.pvInt);
203     } else if (argType.isAssignableFrom(VShort.class)) {
204       return fieldCreate.createScalar(ScalarType.pvShort);
205     } else if (argType.isAssignableFrom(VLong.class)) {
206       return fieldCreate.createScalar(ScalarType.pvLong);
207     } else if (argType.isAssignableFrom(VByte.class)) {
208       return fieldCreate.createScalar(ScalarType.pvByte);
209     } else if (argType.isAssignableFrom(VBoolean.class)) {
210       return fieldCreate.createScalar(ScalarType.pvBoolean);
211       
212     } else if (argType.isAssignableFrom(VDoubleArray.class)) {
213       return fieldCreate.createScalarArray(ScalarType.pvDouble);
214     } else if (argType.isAssignableFrom(VFloatArray.class)) {
215       return fieldCreate.createScalarArray(ScalarType.pvFloat);
216     } else if (argType.isAssignableFrom(VStringArray.class)) {
217       return fieldCreate.createScalarArray(ScalarType.pvString);
218     } else if (argType.isAssignableFrom(VIntArray.class)) {
219       return fieldCreate.createScalarArray(ScalarType.pvInt);
220     } else if (argType.isAssignableFrom(VLongArray.class)) {
221       return fieldCreate.createScalarArray(ScalarType.pvLong);
222     } else if (argType.isAssignableFrom(VShortArray.class)) {
223       return fieldCreate.createScalarArray(ScalarType.pvShort);
224     } else if (argType.isAssignableFrom(VByteArray.class)) {
225       return fieldCreate.createScalarArray(ScalarType.pvByte);
226     }
227     
228     throw new IllegalArgumentException("Argument class " + argType.getSimpleName() + " not supported in pvAccess RPC Service");
229   }

242   @Override
243   public void executeMethod(final Map<String, Object> parameters, final WriteFunction<Map<String, Object>> callback, final WriteFunction<Exception> errorCallback) {
244 
245     RPCClient rpcClient = null;
246 
247     try {
248 
249       rpcClient = PooledRPCClientFactory.getRPCClient(this.hostName, this.channelName);
250 
251     } catch (RPCRequestException e) {
252       errorCallback.writeValue(e);
253       return;
254     }
255 
256 
257     try {
258       String methodName = this.rpcServiceMethodDescription.getOperationName() != null ?
259         this.rpcServiceMethodDescription.getOperationName() : this.rpcServiceMethodDescription.getName();
260 
261       PVStructure pvRequest = createPvRequest(parameters, methodName);
262       PVStructure pvResult = rpcClient.request(pvRequest, 3.0);
263 
264       VType vResult = createResult(pvResult);
265       if (vResult != null) {
266         Map<String, Object> resultMap = new HashMap<>();
267         String resultName = getResultDescriptions().keySet().toArray(new String[getResultDescriptions().size()])[0];
268         resultMap.put(resultName, vResult);
269         callback.writeValue(resultMap);
270       }
271     } catch (RPCRequestException rre) {
272       errorCallback.writeValue(rre);
273 
274     } finally {
275       //back to the pool
276       rpcClient.destroy();
277     }
278   }

281   PVStructure createPvRequest(final Map<String, Object> parameters, String methodName) {
282 
283     PVStructure pvRequest = PVDataFactory.getPVDataCreate().createPVStructure(this.requestStructure);
284     PVStructure retVal = pvRequest;
285     if (useNTQuery)
286     {
287     	pvRequest.getStringField("scheme").put("pva");
288     	pvRequest.getStringField("path").put(channelName);
289     	pvRequest = pvRequest.getStructureField("query");
290     }
291 
292     if (methodFieldName != null)
293     	pvRequest.getStringField(methodFieldName).put(methodName);
294 
295     if ((parameters == null) || (parameters.isEmpty())) {
296       return retVal;
297     }
298 
299     for (String parameterName : parameters.keySet()) {
300 
301       //check if the parameter name is only an alias (if we get a field name back, it's an alias)
302       String fieldName = this.parameterNames.get(parameterName);
303 
304       if ((fieldName == null) || (fieldName.equals(RPCServiceMethodDescription.FIELD_NAME_EQUALS_NAME))) {
305         fieldName = null;
306       }
307 
308       Object value = parameters.get(parameterName);
309       
310       // consider it as optional
311       if (value == null)
312     	  continue;
313 
314       PVField pvField = pvRequest.getSubField((fieldName != null ? fieldName : parameterName));
315       
316       
317       if (pvField instanceof PVScalar)
318       {
319     	  PVScalar pvScalar = (PVScalar)pvField;
320     	  if (value instanceof VDouble)
321     		  convert.fromDouble(pvScalar, ((VDouble)value).getValue());
322     	  else if (value instanceof VInt)
323     		  convert.fromInt(pvScalar, ((VInt)value).getValue());
324     	  else if (value instanceof VFloat)
325     		  convert.fromFloat(pvScalar, ((VFloat)value).getValue());
326     	  else if (value instanceof VLong)
327     		  convert.fromLong(pvScalar, ((VLong)value).getValue());
328     	  else if (value instanceof VShort)
329     		  convert.fromShort(pvScalar, ((VShort)value).getValue());
330     	  else if (value instanceof VByte)
331     		  convert.fromByte(pvScalar, ((VByte)value).getValue());
332     	  else if (value instanceof VBoolean)
333     		  ((PVBoolean)pvScalar).put(((VBoolean)value).getValue());		// TODO Convert is missing fromBoolean
334        	  else if (value instanceof VString)
335     		  convert.fromString(pvScalar, ((VString)value).getValue());
336       }
337       else if (pvField instanceof PVScalarArray)
338       {
339     	  PVScalarArray pvScalarArray = (PVScalarArray)pvField;
340     	  if (value instanceof VDoubleArray)
341     	  {
342     		  ListDouble list = ((VDoubleArray)value).getData();
343               double[] arr = CollectionNumbers.doubleArrayWrappedOrCopy(list);
344     		  convert.fromDoubleArray(pvScalarArray, 0, arr.length, arr, 0);
345     	  }
346     	  else if (value instanceof VIntArray)
347     	  {
348     		  ListInt list = ((VIntArray)value).getData();
349               int[] arr = CollectionNumbers.intArrayWrappedOrCopy(list);
350     		  convert.fromIntArray(pvScalarArray, 0, arr.length, arr, 0);
351     	  }
352     	  else if (value instanceof VFloatArray)
353     	  {
354     		  ListFloat list = ((VFloatArray)value).getData();
355               float[] arr = CollectionNumbers.floatArrayWrappedOrCopy(list);
356     		  convert.fromFloatArray(pvScalarArray, 0, arr.length, arr, 0);
357     	  }
358     	  else if (value instanceof VLongArray)
359     	  {
360     		  ListLong list = ((VLongArray)value).getData();
361               long[] arr = CollectionNumbers.longArrayWrappedOrCopy(list);
362     		  convert.fromLongArray(pvScalarArray, 0, arr.length, arr, 0);
363     	  }
364     	  else if (value instanceof VShortArray)
365     	  {
366     		  ListShort list = ((VShortArray)value).getData();
367               short[] arr = CollectionNumbers.shortArrayWrappedOrCopy(list);
368     		  convert.fromShortArray(pvScalarArray, 0, arr.length, arr, 0);
369     	  }
370     	  else if (value instanceof VByteArray)
371     	  {
372     		  ListByte list = ((VByteArray)value).getData();
373               byte[] arr = CollectionNumbers.byteArrayWrappedOrCopy(list);
374     		  convert.fromByteArray(pvScalarArray, 0, arr.length, arr, 0);
375     	  }
376        	  else if (value instanceof VStringArray)
377     	  {
378     		  List<String> list = ((VStringArray)value).getData();
379     		  String[] arr = list.toArray(new String[list.size()]);
380     		  convert.fromStringArray(pvScalarArray, 0, arr.length, arr, 0);
381     	  }
382       }
383       else
384       {
385          // TODO complex types (do we need to support them?)
386         throw new RuntimeException("pvAccess RPC Service mapping support for " + value.getClass().getSimpleName() + " not implemented");
387       }
388     }
389     
390     return retVal;
391   }

394   private VType createResult(PVStructure pvResult) {
395 
396     if (pvResult == null) {
397       return null;
398     }
399 
400     if (!isResultQuery()) {
401       return null;
402     }
403 
404     Class<?> resultType = getResultTypes().values().toArray(new Class<?>[getResultTypes().size()])[0];
405 
406     if (this.rpcServiceMethodDescription.isResultStandalone) {
407       if (resultType.isAssignableFrom(VTable.class)) {
408 
409         if ("uri:ev4:nt/2012/pwd:NTNameValue".equals(pvResult.getStructure().getID()))
410           return new PVFieldNTNameValueToVTable(pvResult, false);
411         else
412         	return new PVFieldToVTable(pvResult, false);
413         
414       } else if (resultType.isAssignableFrom(VImage.class)) {
415         return new PVFieldToVImage(pvResult, false);
416       } else if (resultType.isAssignableFrom(VStatistics.class)) {
417           return new PVFieldToVStatistics(pvResult, false);
418       } else if (resultType.isAssignableFrom(VDoubleArray.class) &&
419     		  "uri:ev4:nt/2012/pwd:NTMatrix".equals(pvResult.getStructure().getID())) {
420           return new PVFieldNTMatrixToVDoubleArray(pvResult, false);
421       }
422 
423       throw new IllegalArgumentException("Standalone result type " + resultType.getSimpleName() + " not supported in pvAccess RPC rpcservice");
424     }
425 
426     // TODO unsigned types, complex types 
427 
428     String resultName = getResultDescriptions().keySet().toArray(new String[getResultDescriptions().size()])[0];
429     String fieldName = this.rpcServiceMethodDescription.getFieldNames().get(resultName);
430 
431     // TODO missing ValueFactory.newVLong
432     
433     if (resultType.isAssignableFrom(VDouble.class)) {
434         return ValueFactory.newVDouble(pvResult.getDoubleField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
435     } else if (resultType.isAssignableFrom(VFloat.class)) {
436         return ValueFactory.newVFloat(pvResult.getFloatField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
437     } else if (resultType.isAssignableFrom(VString.class)) {
438         return ValueFactory.newVString(pvResult.getStringField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow());
439     } else if (resultType.isAssignableFrom(VInt.class)) {
440         return ValueFactory.newVInt(pvResult.getIntField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
441     } else if (resultType.isAssignableFrom(VShort.class)) {
442         return ValueFactory.newVShort(pvResult.getShortField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
443     } else if (resultType.isAssignableFrom(VByte.class)) {
444         return ValueFactory.newVByte(pvResult.getByteField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow(), ValueFactory.displayNone());
445     } else if (resultType.isAssignableFrom(VBoolean.class)) {
446         return ValueFactory.newVBoolean(pvResult.getBooleanField(fieldName != null ? fieldName : resultName).get(), ValueFactory.alarmNone(), ValueFactory.timeNow());
447     } else if (resultType.isAssignableFrom(VDoubleArray.class)) {
448     	
449 		if ("uri:ev4:nt/2012/pwd:NTMatrix".equals(pvResult.getStructure().getID()))
450 	      return new PVFieldNTMatrixToVDoubleArray(pvResult, false);
451 	    else
452           return new PVFieldToVDoubleArray(pvResult, fieldName != null ? fieldName : resultName, true);
453 		
454     } else if (resultType.isAssignableFrom(VFloatArray.class)) {
455         return new PVFieldToVFloatArray(pvResult, fieldName != null ? fieldName : resultName, true);
456     } else if (resultType.isAssignableFrom(VIntArray.class)) {
457         return new PVFieldToVIntArray(pvResult, fieldName != null ? fieldName : resultName, true);
458     } else if (resultType.isAssignableFrom(VShortArray.class)) {
459         return new PVFieldToVShortArray(pvResult, fieldName != null ? fieldName : resultName, true);
460     } else if (resultType.isAssignableFrom(VByteArray.class)) {
461         return new PVFieldToVByteArray(pvResult, fieldName != null ? fieldName : resultName, true);
462     } else if (resultType.isAssignableFrom(VStringArray.class)) {
463         return new PVFieldToVStringArray(pvResult, fieldName != null ? fieldName : resultName, true);
464     } else if (resultType.isAssignableFrom(VTable.class)) {
465         if ("uri:ev4:nt/2012/pwd:NTNameValue".equals(pvResult.getStructure().getID()))
466             return new PVFieldNTNameValueToVTable(pvResult, false);
467           else
468           	return new PVFieldToVTable(pvResult, false);
469     } else if (resultType.isAssignableFrom(VImage.class)) {
470         return new PVFieldToVImage(pvResult,false);
471     } else if (resultType.isAssignableFrom(VStatistics.class)) {
472         return new PVFieldToVStatistics(pvResult,false);
473     }
474     
475     throw new IllegalArgumentException("Result type " + resultType.getSimpleName() + " not supported in pvAccess RPC rpcservice");
476 
477   }

22  package org.epics.pvmanager.pva.rpcservice.rpcclient;
23  
24  /**
25   * A list of properties that are configurable for a rpc client pool.
26   * @author fhanik (tomcat connection pool)
27   * @author dkumar (modified for a rpc client pool implementation)
28   */
29  
30  public interface PoolConfiguration {
31  
32    public static final int DEFAULT_MAX_ACTIVE = 100;
33  
34    /**
35     * rpc clients that have been abandoned (timed out) wont get closed and reported up unless the number of rpc clients in use are
36     * above the percentage defined by abandonWhenPercentageFull.
37     * The value should be between 0-100.
38     * The default value is 0, which implies that rpc clients are eligible for
39     * closure as soon as removeAbandonedTimeout has been reached.
40     * @param percentage a value between 0 and 100 to indicate when rpc clients that have been abandoned/timed out are considered abandoned
41     */
42    public void setAbandonWhenPercentageFull(int percentage);
43  
44  
45    /**
46     * rpc clients that have been abandoned (timed out) wont get closed and reported up unless the number of rpc clients in use are
47     * above the percentage defined by abandonWhenPercentageFull.
48     * The value should be between 0-100.
49     * The default value is 0, which implies that rpc clients are eligible for
50     * closure as soon as removeAbandonedTimeout has been reached.
51     * @return percentage - a value between 0 and 100 to indicate when rpc clients that have been abandoned/timed out are considered abandoned
52     */
53    public int getAbandonWhenPercentageFull();
54  
55  
56    /**
57     * Returns the number of rpc clients that will be established when the rpc client pool is started.
58     * Default value is 10
59     * @return number of rpc clients to be started when pool is started
60     */
61    public int getInitialSize();
62  
63  
64    /**
65     * Set the number of rpc clients that will be established when the rpc client pool is started.
66     * Default value is 10.
67     * If this value exceeds {@link #setMaxActive(int)} it will automatically be lowered.
68     * @param initialSize the number of rpc clients to be established.
69     *
70     */
71    public void setInitialSize(int initialSize);
72  
73  
74    /**
75     * boolean flag to set if stack traces should be logged for application code which abandoned a rpc client.
76     * Logging of abandoned rpc clients adds overhead for every rpc client borrow because a stack trace has to be generated.
77     * The default value is false.
78     * @return true if the rpc client pool logs stack traces when rpc clients are borrowed from the pool.
79     */
80    public boolean isLogAbandoned();
81  
82  
83    /**
84     * boolean flag to set if stack traces should be logged for application code which abandoned a rpc client.
85     * Logging of abandoned rpc clients adds overhead for every rpc client borrow because a stack trace has to be generated.
86     * The default value is false.
87     * @param logAbandoned set to true if stack traces should be recorded.
88     */
89    public void setLogAbandoned(boolean logAbandoned);
90  
91  
92    /**
93     * The maximum number of active rpc clients that can be allocated from this pool at the same time. The default value is 100
94     * @return the maximum number of rpc clients used by this pool
95     */
96    public int getMaxActive();
97  
98    /**
99     * The maximum number of active rpc clients that can be allocated from this pool at the same time. The default value is 100
100    * @param maxActive hard limit for number of managed rpc clients by this pool
101    */
102   public void setMaxActive(int maxActive);
103 
104 
105   /**
106    * The maximum number of rpc clients that should be kept in the idle pool if {@link #isPoolSweeperEnabled()} returns true.
107    * If the If {@link #isPoolSweeperEnabled()} returns true, then the idle pool can grow up to {@link #getMaxActive}
108    * and will be shrunk according to {@link #getMinEvictableIdleTimeMillis()} setting.
109    * Default value is maxActive:100
110    * @return the maximum number of idle rpc clients.
111    */
112   public int getMaxIdle();
113 
114   /**
115    * The maximum number of rpc clients that should be kept in the idle pool if {@link #isPoolSweeperEnabled()} returns false.
116    * If the If {@link #isPoolSweeperEnabled()} returns true, then the idle pool can grow up to {@link #getMaxActive}
117    * and will be shrunk according to {@link #getMinEvictableIdleTimeMillis()} setting.
118    * Default value is maxActive:100
119    * @param maxIdle the maximum size of the idle pool
120    */
121   public void setMaxIdle(int maxIdle);
122 
123 
124   /**
125    * The maximum number of milliseconds that the pool will wait (when there are no available rpc clients) and the
126    * {@link #getMaxActive} has been reached) for a rpc client to be returned
127    * before throwing an exception. Default value is 30000 (30 seconds)
128    * @return the number of milliseconds to wait for a rpc client to become available if the pool is maxed out.
129    */
130   public int getMaxWait();
131 
132 
133   /**
134    * The maximum number of milliseconds that the pool will wait (when there are no available rpc clients and the
135    * {@link #getMaxActive} has been reached) for a rpc client to be returned
136    * before throwing an exception. Default value is 30000 (30 seconds)
137    * @param maxWait the maximum number of milliseconds to wait.
138    */
139   public void setMaxWait(int maxWait);
140 
141   /**
142    * The minimum amount of time an object must sit idle in the pool before it is eligible for eviction.
143    * The default value is 60000 (60 seconds).
144    * @return the minimum amount of idle time in milliseconds before a rpc client is considered idle and eligible for eviction.
145    */
146   public int getMinEvictableIdleTimeMillis();
147 
148 
149   /**
150    * The minimum amount of time an object must sit idle in the pool before it is eligible for eviction.
151    * The default value is 60000 (60 seconds).
152    * @param minEvictableIdleTimeMillis the number of milliseconds a rpc client must be idle to be eligible for eviction.
153    */
154   public void setMinEvictableIdleTimeMillis(int minEvictableIdleTimeMillis);
155 
156 
157   /**
158    * The minimum number of established rpc clients that should be kept in the pool at all times.
159    * The rpc client pool can shrink below this number if validation fail and rpc clients get closed.
160    * Default value is derived from {@link #getInitialSize()} (also see {@link #setTestWhileIdle(boolean)}
161    * The idle pool will not shrink below this value during an eviction run, hence the number of actual rpc clients
162    * can be between {@link #getMinIdle()} and somewhere between {@link #getMaxIdle()} and {@link #getMaxActive()}
163    * @return the minimum number of idle or established rpc clients
164    */
165   public int getMinIdle();
166 
167 
168   /**
169    * The minimum number of established rpc clients that should be kept in the pool at all times.
170    * The rpc client pool can shrink below this number if validation fail and rpc clients get closed.
171    * Default value is derived from {@link #getInitialSize()} (also see {@link #setTestWhileIdle(boolean)}
172    * The idle pool will not shrink below this value during an eviction run, hence the number of actual rpc clients
173    * can be between {@link #getMinIdle()} and somewhere between {@link #getMaxIdle()} and {@link #getMaxActive()}
174    *
175    * @param minIdle the minimum number of idle or established rpc clients
176    */
177   public void setMinIdle(int minIdle);
178 
179   /**
180    * boolean flag to remove abandoned rpc clients if they exceed the removeAbandonedTimout.
181    * If set to true a rpc client is considered abandoned and eligible for removal if it has
182    * been in use longer than the {@link #getRemoveAbandonedTimeout()} and the condition for
183    * {@link #getAbandonWhenPercentageFull()} is met.
184    * Setting this to true can recover rpc clients from applications that fail to close a rpc client.
185    * See also {@link #isLogAbandoned()} The default value is false.
186    * @return true if abandoned rpc clients can be closed and expelled out of the pool
187    */
188   public boolean isRemoveAbandoned();
189 
190 
191   /**
192    * boolean flag to remove abandoned rpc clients if they exceed the removeAbandonedTimout.
193    * If set to true a rpc client is considered abandoned and eligible for removal if it has
194    * been in use longer than the {@link #getRemoveAbandonedTimeout()} and the condition for
195    * {@link #getAbandonWhenPercentageFull()} is met.
196    * Setting this to true can recover rpc clients from applications that fail to close a rpc client.
197    * See also {@link #isLogAbandoned()} The default value is false.
198    * @param removeAbandoned set to true if abandoned rpc clients can be closed and expelled out of the pool
199    */
200   public void setRemoveAbandoned(boolean removeAbandoned);
201 
202 
203   /**
204    * The time in seconds before a rpc client can be considered abandoned.
205    * @param removeAbandonedTimeout the time in seconds before a used rpc client can be considered abandoned
206    */
207   public void setRemoveAbandonedTimeout(int removeAbandonedTimeout);
208 
209 
210   /**
211    * The time in seconds before a rpc client can be considered abandoned.
212    * @return the time in seconds before a used rpc client can be considered abandoned
213    */
214   public int getRemoveAbandonedTimeout();
215 
216 
217   /**
218    * The indication of whether objects will be validated before being borrowed from the pool.
219    * If the object fails to validate, it will be dropped from the pool, and we will attempt to borrow another.
220    * Default value is false
221    * In order to have a more efficient validation, see {@link #setValidationInterval(long)}
222    * @return true if the rpc client is to be validated upon borrowing a rpc client from the pool
223    * @see #getValidationInterval()
224    */
225   public boolean isTestOnBorrow();
226 
227 
228   /**
229    * The indication of whether objects will be validated before being borrowed from the pool.
230    * If the object fails to validate, it will be dropped from the pool, and we will attempt to borrow another.
231    * Default value is false
232    * In order to have a more efficient validation, see {@link #setValidationInterval(long)}
233    * @param testOnBorrow set to true if validation should take place before a rpc client is handed out to the application
234    * @see #getValidationInterval()
235    */
236   public void setTestOnBorrow(boolean testOnBorrow);
237 
238 
239   /**
240    * The indication of whether objects will be validated after being returned to the pool.
241    * If the object fails to validate, it will be dropped from the pool.
242    * Default value is false
243    * In order to have a more efficient validation, see {@link #setValidationInterval(long)}
244    * @return true if validation should take place after a rpc client is returned to the pool
245    * @see #getValidationInterval()
246    */
247   public boolean isTestOnReturn();
248 
249 
250   /**
251    * The indication of whether objects will be validated after being returned to the pool.
252    * If the object fails to validate, it will be dropped from the pool.
253    * Default value is false
254    * In order to have a more efficient validation, see {@link #setValidationInterval(long)}
255    * @param testOnReturn true if validation should take place after a rpc client is returned to the pool
256    * @see #getValidationInterval()
257    */
258   public void setTestOnReturn(boolean testOnReturn);
259 
260 
261   /**
262    * Set to true if validation should take place while the rpc client is idle.
263    * @return true if validation should take place during idle checks
264    * @see #setTimeBetweenEvictionRunsMillis(int)
265    */
266   public boolean isTestWhileIdle();
267 
268 
269   /**
270    * Set to true if validation should take place while the rpc client is idle.
271    * @param testWhileIdle true if validation should take place during idle checks
272    * @see #setTimeBetweenEvictionRunsMillis(int)
273    */
274   public void setTestWhileIdle(boolean testWhileIdle);
275 
276 
277   /**
278    * The number of milliseconds to sleep between runs of the idle rpc client validation, abandoned cleaner
279    * and idle pool resizing. This value should not be set under 1 second.
280    * It dictates how often we check for idle, abandoned rpc clients, and how often we validate idle rpc client and resize the idle pool.
281    * The default value is 5000 (5 seconds)
282    * @return the sleep time in between validations in milliseconds
283    */
284   public int getTimeBetweenEvictionRunsMillis();
285 
286 
287   /**
288    * The number of milliseconds to sleep between runs of the idle rpc client validation, abandoned cleaner
289    * and idle pool resizing. This value should not be set under 1 second.
290    * It dictates how often we check for idle, abandoned rpc clients, and how often we validate idle rpc client and resize the idle pool.
291    * The default value is 5000 (5 seconds)
292    * @param timeBetweenEvictionRunsMillis the sleep time in between validations in milliseconds
293    */
294   public void setTimeBetweenEvictionRunsMillis(int timeBetweenEvictionRunsMillis);
295 
296 
297   /**
298    * Avoid excess validation, only run validation at most at this frequency - time in milliseconds.
299    * If a rpc client is due for validation, but has been validated previously
300    * within this interval, it will not be validated again.
301    * The default value is 30000 (30 seconds).
302    * @return the validation interval in milliseconds
303    */
304   public long getValidationInterval();
305 
306 
307   /**
308    * avoid excess validation, only run validation at most at this frequency - time in milliseconds.
309    * If a rpc client is due for validation, but has been validated previously
310    * within this interval, it will not be validated again.
311    * The default value is 30000 (30 seconds).
312    * @param validationInterval the validation interval in milliseconds
313    */
314   public void setValidationInterval(long validationInterval);
315 
316 
317    /**
318    * Returns true if the pool sweeper is enabled for the rpc client pool.
319    * The pool sweeper is enabled if any settings that require async intervention in the pool are turned on
320    * <pre>
321    boolean result = getTimeBetweenEvictionRunsMillis()&gt;0;
322    result = result &amp;&amp; (isRemoveAbandoned() &amp;&amp; getRemoveAbandonedTimeout()&gt;0);
323    result = result || (isTestWhileIdle();
324    return result;
325    </pre>
326    *
327    * @return true if a background thread is or will be enabled for this pool
328    */
329   public boolean isPoolSweeperEnabled();
330 
331 
332   /**
333    * Time in milliseconds to keep this rpc client  alive even when used.
334    * When a rpc client is returned to the pool, the pool will check to see if the
335    * ((now - time-when-connected) &gt; maxAge) has been reached, and if so,
336    * it closes the rpc client rather than returning it to the pool.
337    * The default value is 0, which implies that rpc clients will be left open and no
338    * age check will be done upon returning the rpc client to the pool.
339    * @return the time in milliseconds a rpc client will be open for when used
340    */
341   public long getMaxAge();
342 
343 
344   /**
345    * Time in milliseconds to keep this rpc client alive even when used.
346    * When a rpc client is returned to the pool, the pool will check to see if the
347    * ((now - time-when-connected) &gt; maxAge) has been reached, and if so,
348    * it closes the rpc client rather than returning it to the pool.
349    * The default value is 0, which implies that rpc clients will be left open and no
350    * age check will be done upon returning the rpc client to the pool.
351    * @param maxAge the time in milliseconds a rpc client will be open for when used
352    */
353   public void setMaxAge(long maxAge);
354 
355 
356   /**
357    * Return true if a lock should be used when operations are performed on the rpc client object.
358    * Should be set to false unless you plan to have a background thread of your own doing idle and abandon checking
359    * such as JMX clients. If the pool sweeper is enabled, then the lock will automatically be used regardless of this setting.
360    * @return true if a lock is used.
361    */
362   public boolean getUseLock();
363 
364 
365   /**
366    * Set to true if a lock should be used when operations are performed on the rpc client object.
367    * Should be set to false unless you plan to have a background thread of your own doing idle and abandon checking
368    * such as JMX clients. If the pool sweeper is enabled, then the lock will automatically be used regardless of this setting.
369    * @param useLock set to true if a lock should be used on rpc client operations
370    */
371   public void setUseLock(boolean useLock);
372 
373 
374   /**
375    * Similar to {@link #setRemoveAbandonedTimeout(int)} but instead of treating the rpc client
376    * as abandoned, and potentially closing the rpc client, this simply logs the warning if
377    * {@link #isLogAbandoned()} returns true. If this value is equal or less than 0, no suspect
378    * checking will be performed. Suspect checking only takes place if the timeout value is larger than 0 and
379    * the rpc client was not abandoned or if abandon check is disabled. If a rpc client is suspect a WARN message gets
380    * logged.
381    * @param seconds - the amount of time in seconds that has to pass before a rpc client is marked suspect.
382    */
383   public void setSuspectTimeout(int seconds);
384 
385 
386   /**
387    * Returns the time in seconds to pass before a rpc client is marked an abanoned suspect.
388    * Any value lesser than or equal to 0 means the check is disabled.
389    * @return Returns the time in seconds to pass before a rpc client is marked an abanoned suspect.
390    */
391   public int getSuspectTimeout();

6   
7   import org.epics.pvdata.pv.PVStructure;
8   import org.epics.vtype.VDouble;
9   import org.epics.vtype.VTypeToString;
10  
11  /**
12   * @author msekoranja
13   *
14   */
15  public class PVFieldToVDouble extends AlarmTimeDisplayExtractor implements VDouble {
16  
17  	protected final Double value;
18  	
19  	/**
20  	 * @param pvField
21  	 * @param disconnected
22  	 */
23  	public PVFieldToVDouble(PVStructure pvField, boolean disconnected) {
24  		super(pvField, disconnected);
25  		
26  		value = getDoubleValue(pvField, "value", null);
27  	}
28  
29      /* (non-Javadoc)
30       * @see org.epics.pvmanager.pva.adapters.PVFieldToVNumber#getValue()
31       */
32  	@Override
33      public Double getValue()
34      {
35      	return value;
36      }
37      
38      @Override
39      public String toString() {
40          return VTypeToString.toString(this);
41      }
42  
43  }

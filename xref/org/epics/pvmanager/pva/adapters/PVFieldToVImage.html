26  public class PVFieldToVImage implements VImage {
27  
28  	private final int width;
29  	private final int height;
30  	private final byte[] data;
31  	
32  	private enum NDColorMode
33  	{
34  		NDColorModeMono,    /** Monochromatic image */
35  		NDColorModeBayer,   /** Bayer pattern image,
36  										1 value per pixel but with color filter on detector */
37  		NDColorModeRGB1,    /** RGB image with pixel color interleave,
38  										data array is [3, NX, NY] */
39  		NDColorModeRGB2,    /** RGB image with row color interleave,
40  										data array is [NX, 3, NY]  */
41  		NDColorModeRGB3,    /** RGB image with plane color interleave,
42  										data array is [NX, NY, 3]  */
43  		NDColorModeYUV444,  /** YUV image, 3 bytes encodes 1 RGB pixel */
44  		NDColorModeYUV422,  /** YUV image, 4 bytes encodes 2 RGB pixel */
45  		NDColorModeYUV411;   /** YUV image, 6 bytes encodes 4 RGB pixels */
46  	};
47  	
48  			
49  	/**
50  	 * @param pvImage
51  	 * @param disconnected
52  	 */
53  	public PVFieldToVImage(PVStructure pvImage, boolean disconnected) {
54  		
55  		PVScalarArray valueArray = (PVScalarArray)pvImage.getSubField("value");
56  		if (valueArray == null)
57  			throw new IllegalArgumentException("given pvImage structure is missing scalar_t[] value field");
58  
59  		PVInt colorMode = pvImage.getIntField("colorMode");
60  		if (colorMode == null)
61  			throw new IllegalArgumentException("given pvImage structure is missing int colorMode field");
62  		int cmOrdinal = colorMode.get();
63  		NDColorMode[] cmValues = NDColorMode.values();
64  		if (cmOrdinal < 0 || cmOrdinal >= cmValues.length)
65  			throw new IllegalArgumentException("given pvImage structure field int colorMode is invalid");
66  		NDColorMode mode = cmValues[cmOrdinal];
67  			
68  		
69  		PVIntArray dim = (PVIntArray)pvImage.getScalarArrayField("dim", ScalarType.pvInt);
70  		if (dim == null)
71  			throw new IllegalArgumentException("given pvImage structure is missing int[] dim field");
72  
73  		int dimLen = dim.getLength();
74  		if (dimLen != 2 && dimLen != 3)
75  			throw new IllegalArgumentException("given pvImage structure field int[] dim array size is not valid");
76  
77  		IntArrayData dimData = new IntArrayData();
78  		dim.get(0, dim.getLength(), dimData);
79  		
80  		int nx = dimData.data[0];
81  		int ny = dimData.data[1];
82  		int nz = (dimLen == 3) ? dimData.data[2] : 1;
83  		
84  		int valueArraySize = nx * ny * nz;
85  		if (valueArraySize <= 0)
86  			throw new IllegalArgumentException("given pvImage structure field int[] dim values are not valid");
87  		
88  		if (valueArray.getLength() != valueArraySize)
89  			throw new IllegalArgumentException("scalar_t[] value array length does not match given dimensions");
90  		
91  		switch (mode)
92  		{
93  			case NDColorModeMono:
94  			case NDColorModeBayer:
95  			{
96  				
97  				// 2D value array
98  				if (nz != 1)
99  					throw new IllegalArgumentException("invalid dimensions for given color mode");
100 				
101 				width = nx;
102 				height = ny;
103 				
104 				switch (valueArray.getScalarArray().getElementType())
105 				{
106 					case pvBoolean:
107 					{
108 						BooleanArrayData bad = new BooleanArrayData();
109 						((PVBooleanArray)valueArray).get(0, valueArraySize, bad);
110 						// convert b&w to BGR
111 						data = new byte[3*valueArraySize];
112 						int p = 0;
113 						for (int i = 0; i < valueArraySize; i++)
114 						{
115 							byte c = bad.data[i] ? (byte)-1 : (byte)0;
116 							// B = G = R = c
117 							data[p++] = c; 
118 							data[p++] = c; 
119 							data[p++] = c;
120 						}
121 						break;
122 					}
123 					
124 					case pvByte:
125 					{
126 						ByteArrayData bad = new ByteArrayData();
127 						((PVByteArray)valueArray).get(0, valueArraySize, bad);
128 						// convert grayscale to BGR
129 						data = new byte[3*valueArraySize];
130 						int p = 0;
131 						for (int i = 0; i < valueArraySize; i++)
132 						{
133 							byte c = bad.data[i];
134 							// B = G = R = c
135 							data[p++] = c; 
136 							data[p++] = c; 
137 							data[p++] = c;
138 						}
139 						break;
140 					}
141 					
142 					case pvShort:
143 					{
144 						ShortArrayData bad = new ShortArrayData();
145 						((PVShortArray)valueArray).get(0, valueArraySize, bad);
146 						// convert grayscale to BGR
147 						data = new byte[3*valueArraySize];
148 						int p = 0;
149 						for (int i = 0; i < valueArraySize; i++)
150 						{
151 							byte c = (byte)(bad.data[i] >>> 8);
152 							// B = G = R = c
153 							data[p++] = c; 
154 							data[p++] = c; 
155 							data[p++] = c;
156 						}
157 						break;
158 					}
159 
160 					case pvInt:
161 					{
162 						IntArrayData bad = new IntArrayData();
163 						((PVIntArray)valueArray).get(0, valueArraySize, bad);
164 						// convert grayscale to BGR
165 						data = new byte[3*valueArraySize];
166 						int p = 0;
167 						for (int i = 0; i < valueArraySize; i++)
168 						{
169 							byte c = (byte)(bad.data[i] >>> 24);
170 							// B = G = R = c
171 							data[p++] = c; 
172 							data[p++] = c; 
173 							data[p++] = c;
174 						}
175 						break;
176 					}
177 
178 					/*
179 					case pvFloat:
180 					{
181 						FloatArrayData bad = new FloatArrayData();
182 						((PVFloatArray)valueArray).get(0, valueArraySize, bad);
183 						// convert grayscale to BGR
184 						data = new byte[3*valueArraySize];
185 						int p = 0;
186 						for (int i = 0; i < valueArraySize; i++)
187 						{
188 							int tc = (int)(bad.data[i] * 255);   // assuming [0 - 1.0]
189 							byte c = (tc > 127) ? (byte)(tc-256) : (byte)tc;
190 							// B = G = R = c
191 							data[p++] = c; 
192 							data[p++] = c; 
193 							data[p++] = c;
194 						}
195 						break;
196 					}
197 
198 					case pvDouble:
199 					{
200 						DoubleArrayData bad = new DoubleArrayData();
201 						((PVDoubleArray)valueArray).get(0, valueArraySize, bad);
202 						// convert grayscale to BGR
203 						data = new byte[3*valueArraySize];
204 						int p = 0;
205 						for (int i = 0; i < valueArraySize; i++)
206 						{
207 							int tc = (int)(bad.data[i] * 255);	// assuming [0 - 1.0]
208 							byte c = (tc > 127) ? (byte)(tc-256) : (byte)tc;
209 							// B = G = R = c
210 							data[p++] = c; 
211 							data[p++] = c; 
212 							data[p++] = c;
213 						}
214 						break;
215 					}
216 					*/
217 					
218 					default:
219 						throw new IllegalArgumentException("unsupported scalar_t[] value type");
220 				
221 				}
222 				
223 				break;
224 			}
225 			
226 			case NDColorModeRGB1:
227 			{
228 				if (nx != 3)
229 					throw new IllegalArgumentException("dim[0] should be 3 for NDColorModeRGB1");
230 				width = ny;
231 				height = nz;
232 				
233 				ByteArrayData bad = new ByteArrayData();
234 				((PVByteArray)valueArray).get(0, valueArraySize, bad);
235 				// convert to BGR
236 				data = new byte[valueArraySize];
237 				int out = 0; int in = 0;
238 				while (in < valueArraySize)
239 				{
240 					byte r = bad.data[in++]; 
241 					byte g = bad.data[in++]; 
242 					byte b = bad.data[in++];
243 					data[out++] = b; 
244 					data[out++] = g; 
245 					data[out++] = r;
246 				}
247 				break;
248 			}
249 
250 			case NDColorModeRGB2:
251 			{
252 				// ny = 3
253 				if (ny != 3)
254 					throw new IllegalArgumentException("dim[1] should be 3 for NDColorModeRGB2");
255 				width = nx;
256 				height = nz;
257 				
258 				ByteArrayData bad = new ByteArrayData();
259 				((PVByteArray)valueArray).get(0, valueArraySize, bad);
260 				// convert to BGR
261 				data = new byte[valueArraySize];
262 				int out = 0; 
263 				int nCols = width;
264 				int nRows = height;
265 				for (int row = 0; row < nRows; row++)
266 				{
267 					int redIn = row * nCols * 3;
268 					int greenIn = redIn + nCols;
269 					int blueIn = greenIn + nCols;
270 					for (int col = 0; col < nCols; col++)
271 					{
272 						byte r = bad.data[redIn++]; 
273 						byte g = bad.data[greenIn++]; 
274 						byte b = bad.data[blueIn++];
275 						data[out++] = b; 
276 						data[out++] = g; 
277 						data[out++] = r;
278 					}
279 				}
280 				break;
281 			}
282 			
283 			case NDColorModeRGB3:
284 			{
285 				// nz = 3
286 				if (nz != 3)
287 					throw new IllegalArgumentException("dim[2] should be 3 for NDColorModeRGB3");
288 				width = nx;
289 				height = ny;
290 				
291 				int imageSize = width * height;
292 
293 				ByteArrayData bad = new ByteArrayData();
294 				((PVByteArray)valueArray).get(0, valueArraySize, bad);
295 				// convert to BGR
296 				data = new byte[valueArraySize];
297 				int out = 0; 
298 				int redIn = 0;
299 				int greenIn = imageSize;
300 				int blueIn = imageSize * 2;
301 				while (redIn < imageSize)
302 				{
303 					byte r = bad.data[redIn++]; 
304 					byte g = bad.data[greenIn++]; 
305 					byte b = bad.data[blueIn++];
306 					data[out++] = b; 
307 					data[out++] = g; 
308 					data[out++] = r;
309 				}
310 				break;
311 			}
312 		
313 			case NDColorModeYUV444:
314 			case NDColorModeYUV422:
315 			case NDColorModeYUV411:
316 			default:
317 				throw new IllegalArgumentException("unsupported colorMode");
318 		}				
319 
320 	}

322 	@Override
323 	public int getHeight() {
324 		return height;
325 	}
326 
327 	@Override
328 	public int getWidth() {
329 		return width;
330 	}
331 
332 	@Override
333 	public byte[] getData() {
334 		return data;
335 	}
336 
337 	@Override
338     public String toString() {
339         StringBuilder builder = new StringBuilder();
340         Class<?> type = ValueUtil.typeOf(this);
341         builder.append(type.getSimpleName())
342                 .append("[height=")
343                 .append(getHeight())
344                 .append(", width=")
345                 .append(getWidth());
346         builder.append(']');
347         return builder.toString();
348     }

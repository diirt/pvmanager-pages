5   package org.epics.pvmanager.pva.adapters;
6   
7   import org.epics.pvdata.pv.PVField;
8   import org.epics.pvdata.pv.PVScalar;
9   import org.epics.pvdata.pv.PVStructure;
10  import org.epics.vtype.VInt;
11  import org.epics.vtype.VTypeToString;
12  
13  /**
14   * @author msekoranja
15   *
16   */
17  public class PVFieldToVInt extends AlarmTimeDisplayExtractor implements VInt {
18  
19  	protected final Integer value;
20  	
21  	/**
22  	 * @param pvField
23  	 * @param disconnected
24  	 */
25  	public PVFieldToVInt(PVStructure pvField, boolean disconnected) {
26  		super(pvField, disconnected);
27  		
28  		PVField field = pvField.getSubField("value");
29  		if (field instanceof PVScalar)
30  			value = convert.toInt((PVScalar)field);
31  		else
32  			value = null;
33  	}
34  
35  	/* (non-Javadoc)
36  	 * @see org.epics.pvmanager.pva.adapters.PVFieldToVNumber#getValue()
37  	 */
38  	@Override
39      public Integer getValue()
40      {
41      	return value;
42      }
43      
44      @Override
45      public String toString() {
46          return VTypeToString.toString(this);
47      }
48  
49  }

5   package org.epics.pvmanager.pva.adapters;
6   
7   import org.epics.pvdata.pv.PVBoolean;
8   import org.epics.pvdata.pv.PVStructure;
9   import org.epics.vtype.VBoolean;
10  import org.epics.vtype.VTypeToString;
11  
12  /**
13   * @author dkumar
14   */
15  public class PVFieldToVBoolean extends AlarmTimeDisplayExtractor implements VBoolean {
16   
17    protected final Boolean value;
18  
19  
20    /**
21     * @param pvField
22     * @param disconnected
23     */
24    public PVFieldToVBoolean(PVStructure pvField, boolean disconnected) {
25      super(pvField, disconnected);
26  
27      PVBoolean booleanField = pvField.getBooleanField("value");
28      if (booleanField != null) {
29        value = booleanField.get();
30      } else {
31        value = null;
32      }
33    }
34  
35  
36    @Override
37    public Boolean getValue() {
38      return value;
39    }
40  
41  
42    @Override
43    public String toString() {
44      return VTypeToString.toString(this);
45    }
46  }

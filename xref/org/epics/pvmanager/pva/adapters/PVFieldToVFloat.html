5   package org.epics.pvmanager.pva.adapters;
6   
7   import org.epics.pvdata.pv.PVField;
8   import org.epics.pvdata.pv.PVScalar;
9   import org.epics.pvdata.pv.PVStructure;
10  import org.epics.vtype.VFloat;
11  import org.epics.vtype.VTypeToString;
12  
13  /**
14   * @author dkumar
15   */
16  public class PVFieldToVFloat extends AlarmTimeDisplayExtractor implements VFloat {
17  
18  	protected final Float value;
19  
20  	public PVFieldToVFloat(PVStructure pvField, boolean disconnected) {
21  		this("value", pvField, disconnected);
22  	}
23  
24  	public PVFieldToVFloat(String fieldName, PVStructure pvField, boolean disconnected) {
25  		this(pvField.getSubField(fieldName), pvField, disconnected);
26  	}
27  
28  	public PVFieldToVFloat(PVField field, PVStructure pvParent, boolean disconnected) {
29  		super(pvParent, disconnected);
30  
31  		if (field instanceof PVScalar)
32  			value = convert.toFloat((PVScalar)field);
33  	    else
34  	    	value = null;
35  	}
36  
37  	@Override
38  	public Float getValue() {
39  		return value;
40  	}
41  
42  	@Override
43  	public String toString() {
44  		return VTypeToString.toString(this);
45  	}
46  }

45  public class PVFieldNTNameValueToVTable implements VTable {
46  
47    private Class<?> valueType;
48    private List<String> names;
49    private List<Object> values;
50  
51    private static final Convert convert = ConvertFactory.getConvert();
52  
53    /**
54     * @param pvField
55     * @param disconnected
56     */
57  	public PVFieldNTNameValueToVTable(PVStructure pvField, boolean disconnected) {
58  
59  		PVStringArray namesField = (PVStringArray) pvField.getScalarArrayField("name", ScalarType.pvString);
60  
61  		PVScalarArray scalarArray = (PVScalarArray) pvField.getSubField("value");
62  
63  		if ((namesField == null) || (scalarArray == null)) {
64  			this.names = null;
65  			valueType = null;
66  			values = null;
67  			return;
68  		}
69  
70  		StringArrayData namesData = new StringArrayData();
71  		namesField.get(0, namesField.getLength(), namesData);
72  		this.names = Arrays.asList(namesData.data);
73  
74  		int numCols = this.names.size();
75  
76  		this.values = new ArrayList<Object>(numCols);
77  
78  		if (scalarArray instanceof PVDoubleArray) {
79  			
80  			valueType = double.class;
81  			DoubleArrayData data = new DoubleArrayData();
82  			((PVDoubleArray) scalarArray).get(0, numCols, data);
83  			for (int i = 0; i < numCols; i++)
84  				values.add(new ArrayDouble(data.data[i]));
85  
86  		} else if (scalarArray instanceof PVFloatArray) {
87  
88  			valueType = float.class;
89  			FloatArrayData data = new FloatArrayData();
90  			((PVFloatArray) scalarArray).get(0, numCols, data);
91  			for (int i = 0; i < numCols; i++)
92  				values.add(new ArrayFloat(data.data[i]));
93  
94  		} else if (scalarArray instanceof PVIntArray) {
95  
96  			valueType = int.class;
97  			IntArrayData data = new IntArrayData();
98  			((PVIntArray) scalarArray).get(0, numCols, data);
99  			for (int i = 0; i < numCols; i++)
100 				values.add(new ArrayInt(data.data[i]));
101 
102 		} else if (scalarArray instanceof PVUIntArray) {
103 
104 			valueType = int.class;
105 			IntArrayData data = new IntArrayData();
106 			((PVUIntArray) scalarArray).get(0, numCols, data);
107 			for (int i = 0; i < numCols; i++)
108 				values.add(new ArrayInt(data.data[i]));
109 
110 		} else if (scalarArray instanceof PVByteArray) {
111 
112 			valueType = byte.class;
113 			ByteArrayData data = new ByteArrayData();
114 			((PVByteArray) scalarArray).get(0, numCols, data);
115 			for (int i = 0; i < numCols; i++)
116 				values.add(new ArrayByte(data.data[i]));
117 
118 		} else if (scalarArray instanceof PVUByteArray) {
119 
120 			valueType = byte.class;
121 			ByteArrayData data = new ByteArrayData();
122 			((PVUByteArray) scalarArray).get(0, numCols, data);
123 			for (int i = 0; i < numCols; i++)
124 				values.add(new ArrayByte(data.data[i]));
125 
126 		} else if (scalarArray instanceof PVLongArray
127 				|| scalarArray instanceof PVULongArray) {
128 
129 			valueType = int.class;
130 			int[] intArr = new int[numCols];
131 			convert.toIntArray(scalarArray, 0, numCols, intArr, 0);
132 			for (int i = 0; i < numCols; i++)
133 				values.add(new ArrayInt(intArr[i]));
134 
135 		} else if (scalarArray instanceof PVShortArray) {
136 
137 			valueType = short.class;
138 			ShortArrayData data = new ShortArrayData();
139 			((PVShortArray) scalarArray).get(0, numCols, data);
140 			for (int i = 0; i < numCols; i++)
141 				values.add(new ArrayShort(data.data[i]));
142 
143 		} else if (scalarArray instanceof PVUShortArray) {
144 
145 			valueType = short.class;
146 			ShortArrayData data = new ShortArrayData();
147 			((PVUShortArray) scalarArray).get(0, numCols, data);
148 			for (int i = 0; i < numCols; i++)
149 				values.add(new ArrayShort(data.data[i]));
150 
151 		} else if (scalarArray instanceof PVStringArray) {
152 
153 			valueType = String.class;
154 			StringArrayData data = new StringArrayData();
155 			((PVStringArray) scalarArray).get(0, numCols, data);
156 			for (int i = 0; i < numCols; i++)
157 				values.add(Arrays.asList(data.data[i]));
158 
159 		} else {
160 			throw new IllegalArgumentException("Unsupported type for NTNameValue.value array field");
161 		}
162 	}
163 
164   @Override
165   public int getColumnCount() {
166     return this.names.size();
167   }
168 
169   @Override
170   public int getRowCount() {
171     return 1;
172   }
173 
174   @Override
175   public Class<?> getColumnType(int column) {
176     return this.valueType;
177   }
178 
179   @Override
180   public String getColumnName(int column) {
181     return this.names.get(column);
182   }
183 
184   @Override
185   public Object getColumnData(int column) {
186     return this.values.get(column);
187   }
188 
189   @Override
190   public String toString() {
191     return VTypeToString.toString(this);
192   }

5   package org.epics.pvmanager.pva.adapters;
6   
7   import java.util.Arrays;
8   import java.util.Collections;
9   import java.util.List;
10  
11  import org.epics.pvdata.pv.StringArrayData;
12  import org.epics.pvdata.pv.PVStringArray;
13  import org.epics.pvdata.pv.PVStructure;
14  import org.epics.pvdata.pv.ScalarType;
15  import org.epics.vtype.VStringArray;
16  import org.epics.vtype.VTypeToString;
17  import org.epics.util.array.ArrayInt;
18  import org.epics.util.array.ListInt;
19  
20  /**
21   * @author msekoranja
22   *
23   */
24  public class PVFieldToVStringArray extends AlarmTimeDisplayExtractor implements VStringArray {
25  
26  	private final ListInt size;
27  	private final List<String> array;
28  	
29  	/**
30  	 * @param pvField
31  	 * @param disconnected
32  	 */
33  	public PVFieldToVStringArray(PVStructure pvField, String fieldName, boolean disconnected) {
34  		super(pvField, disconnected);
35  		
36  		PVStringArray valueField =
37  			(PVStringArray)pvField.getScalarArrayField(fieldName, ScalarType.pvString);
38  		if (valueField != null)
39  		{
40  			StringArrayData data = new StringArrayData();
41  			valueField.get(0, valueField.getLength(), data);
42  			
43  			this.size = new ArrayInt(data.data.length);
44  			this.array = Collections.unmodifiableList(Arrays.asList(data.data));
45  		}
46  		else
47  		{
48  			this.size = null;
49  			this.array = null;
50  		}
51  	}
52  
53  	public PVFieldToVStringArray(PVStructure pvField, boolean disconnected) {
54  		this(pvField, "value", disconnected);
55  	}
56  	
57  	/* (non-Javadoc)
58  	 * @see org.epics.pvmanager.data.Array#getSizes()
59  	 */
60  	@Override
61  	public ListInt getSizes() {
62  		return size;
63  	}
64  
65  	/* (non-Javadoc)
66  	 * @see org.epics.pvmanager.data.VStringArray#getArray()
67  	 */
68  	@Override
69  	public List<String> getData() {
70  		return array;
71  	}
72      
73      @Override
74      public String toString() {
75          return VTypeToString.toString(this);
76      }
77  
78  }

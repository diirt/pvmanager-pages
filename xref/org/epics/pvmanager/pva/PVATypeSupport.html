5   package org.epics.pvmanager.pva;
6   
7   import org.epics.pvmanager.DataSourceTypeSupport;
8   import org.epics.pvmanager.ValueCache;
9   
10  /**
11   * 
12   * Given a set of {@link PVATypeAdapter} prepares type support for the 
13   * JCA data source.
14   *
15   * @author carcassi
16   */
17  public class PVATypeSupport extends DataSourceTypeSupport {
18      
19      private final PVATypeAdapterSet adapters;
20  
21      /**
22       * A new type support for the pva type support.
23       * 
24       * @param adapters a set of pva adapters
25       */
26      public PVATypeSupport(PVATypeAdapterSet adapters) {
27          this.adapters = adapters;
28      }
29      
30      /**
31       * Returns a matching type adapter for the given
32       * cache and channel.
33       * 
34       * @param cache the cache that will store the data
35       * @param channel the pva channel
36       * @return the matched type adapter
37       */
38      protected PVATypeAdapter find(ValueCache<?> cache, PVAChannelHandler channel) {
39          return find(adapters.getAdapters(), cache, channel);
40      }
41      
42  }

5   package org.epics.pvmanager;
6   
7   /**
8    * Represent a building block that can store a particular value
9    *
10   * @param <T> the type of the value held by the cache
11   * @author carcassi
12   */
13  public class ValueCacheImpl<T> implements ValueCache<T> {
14  
15      /**
16       * The value cached
17       */
18      private T value;
19  
20      /**
21       * The type of objects this cache can store
22       */
23      private final Class<T> type;
24  
25      /**
26       * Creates a new cache for the given data type.
27       *
28       * @param dataType class token for the data type
29       */
30      public ValueCacheImpl(Class<T> dataType) {
31          this.value = null;
32          type = dataType;
33      }
34  
35      /**
36       * Returns the value in the cache.
37       *
38       * @return value in the cache
39       */
40      @Override
41      public T readValue() {
42          return value;
43      }
44  
45      /**
46       * Changes the value in the cache.
47       *
48       * @param newValue the new value
49       */
50      @Override
51      public void writeValue(Object newValue) {
52          // TODO should add type check
53          this.value = type.cast(newValue);
54      }
55  
56      /**
57       * The type of objects that this cache can contain.
58       *
59       * @return the class token
60       */
61      @Override
62      public Class<T> getType() {
63          return type;
64      }
65  
66  }

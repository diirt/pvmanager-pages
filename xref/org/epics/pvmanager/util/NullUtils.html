5   package org.epics.pvmanager.util;
6   
7   import java.util.Collection;
8   
9   /**
10   * Utility class to handle null values.
11   *
12   * @author carcassi
13   */
14  public class NullUtils {
15  
16      /**
17       * Returns true if both objects are null or they are equal.
18       *
19       * @param obj1 first object
20       * @param obj2 second object
21       * @return true if equals or both are null
22       * @deprecated use Java7 method java.util.Objects.equals
23       */
24      @Deprecated
25      public static boolean equalsOrBothNull(Object obj1, Object obj2) {
26          if (obj1 == null)
27              return obj2 == null;
28          else
29              return obj1.equals(obj2);
30      }
31      
32      /**
33       * Checks whether the collection contains a null value.
34       * 
35       * @param args a collection; can't be null
36       * @return true if one of the value is null
37       */
38      public static boolean containsNull(Collection<Object> args) {
39          for (Object object : args) {
40              if (object == null) {
41                  return true;
42              }
43          }
44          return false;
45      }
46  }

5   package org.epics.pvmanager.timecache.query;

7   import java.util.List;
8   
9   import org.epics.util.time.TimeInterval;
10  import org.epics.util.time.Timestamp;
11  import org.epics.vtype.VType;
12  
13  /**
14   * Represents a chunk of data.
15   * @author Fred Arnaud (Sopra Group) - ITER
16   */
17  public interface QueryData {
18  
19  	/**
20  	 * The time range where the data is defined
21  	 */
22  	public TimeInterval getTimeInterval();
23  
24  	/**
25  	 * The number of elements. <p> Both data and timestamps will have this
26  	 * number of elements.
27  	 */
28  	public int getCount();
29  
30  	/**
31  	 * The time for each element.
32  	 */
33  	public List<Timestamp> getTimestamps();
34  
35  	public List<VType> getData();
36  
37  }

5   package org.epics.pvmanager.timecache;
6   
7   import java.util.Map.Entry;
8   import java.util.SortedMap;
9   import java.util.TreeMap;
10  
11  /**
12   * Statistics of {@link Cache}.
13   * @author Fred Arnaud (Sopra Group) - ITER
14   */
15  public class CacheStatistics {
16  
17  	private SortedMap<String, PVCacheStatistics> pvStats = new TreeMap<String, PVCacheStatistics>();
18  
19  	public void addStats(String channelName, PVCacheStatistics stats) {
20  		pvStats.put(channelName, stats);
21  	}
22  
23  	public String print() {
24  		StringBuilder sb = new StringBuilder();
25  		for (Entry<String, PVCacheStatistics> statsEntry : pvStats.entrySet()) {
26  			sb.append("-----------------------------------------------\n");
27  			sb.append(">>>>" + statsEntry.getKey() + "<<<<\n");
28  			sb.append(statsEntry.getValue().print() + "\n");
29  			sb.append("-----------------------------------------------\n");
30  		}
31  		return sb.toString();
32  	}
33  }

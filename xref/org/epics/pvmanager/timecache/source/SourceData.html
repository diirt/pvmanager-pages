5   package org.epics.pvmanager.timecache.source;
6   
7   import org.epics.pvmanager.timecache.Data;
8   import org.epics.util.time.Timestamp;
9   import org.epics.vtype.VType;
10  
11  /**
12   * {@link Data} from {@link DataSource}.
13   * @author Fred Arnaud (Sopra Group) - ITER
14   */
15  public class SourceData implements Data {
16  
17  	private final Timestamp timestamp;
18  	private final VType value;
19  
20  	public SourceData(Timestamp timestamp, VType value) {
21  		this.timestamp = timestamp;
22  		this.value = value;
23  	}
24  
25  	/** {@inheritDoc} */
26  	@Override
27  	public Timestamp getTimestamp() {
28  		return timestamp;
29  	}
30  
31  	/** {@inheritDoc} */
32  	@Override
33  	public VType getValue() {
34  		return value;
35  	}
36  
37  	/** {@inheritDoc} */
38  	@Override
39  	public int hashCode() {
40  		final int prime = 31;
41  		int result = 1;
42  		result = prime * result
43  				+ ((timestamp == null) ? 0 : timestamp.hashCode());
44  		return result;
45  	}
46  
47  	/** {@inheritDoc} */
48  	@Override
49  	public boolean equals(Object obj) {
50  		if (this == obj)
51  			return true;
52  		if (obj == null)
53  			return false;
54  		if (getClass() != obj.getClass())
55  			return false;
56  		SourceData other = (SourceData) obj;
57  		if (timestamp == null) {
58  			if (other.timestamp != null)
59  				return false;
60  		} else if (!timestamp.equals(other.timestamp))
61  			return false;
62  		return true;
63  	}
64  
65  	/** {@inheritDoc} */
66  	@Override
67  	public int compareTo(Data d) {
68  		return timestamp.compareTo(d.getTimestamp());
69  	}
70  
71  	/** {@inheritDoc} */
72  	@Override
73  	public String toString() {
74  		return "SourceData [timestamp=" + timestamp + ", value=" + value + "]";
75  	}
76  
77  }

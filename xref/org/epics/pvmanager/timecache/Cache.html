11  /**
12   * Cache main interface: creates and manages all {@link PVCache}. Initialises
13   * {@link Query} with the corresponding {@link PVCache}.
14   * @author Fred Arnaud (Sopra Group) - ITER
15   */
16  public interface Cache {
17  
18  	/**
19  	 * Creates a new {@link Query}.
20  	 * @param channelName channel to be requested.
21  	 * @param type VType of data to be pulled by the query.
22  	 * @param parameters {@link QueryParameters}.
23  	 * @return initialised query without data.
24  	 */
25  	public <V extends VType> Query createQuery(String channelName,
26  			Class<V> type, QueryParameters parameters);
27  
28  	public void setStatisticsEnabled(boolean enabled);
29  
30  	public CacheStatistics getStatistics();
31  
32  }

5   package org.epics.pvmanager.extra;
6   
7   /**
8    * Defines a way to interpolate some scattered points to an equally distanced
9    * grid.
10   * 
11   * @author carcassi
12   */
13  public interface Interpolation {
14      
15      /**
16       * Returns a set of interpolated y, equally spaced.
17       * 
18       * @param x an array of position value, assumed to be monotonically increasing
19       * @param y an array of values, assumed to be the same length of x
20       * @param nSamples the number of samples to be returned
21       * @return a new array with nSamples interpolated values
22       */
23      public double[] interpolate(double[] x, double[] y, int nSamples);
24  }

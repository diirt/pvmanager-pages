5   package org.epics.pvmanager;

7   /**
8    * Callback for any change in the PV value. Cannot simply use a PropertyChangedListener
9    * because the payload of the PV will be typically updated in place for complex
10   * data structures, and therefore the data object is the same and would not
11   * trigger a PropertyChangedEvent.
12   *
13   * @param <T> the type of reader for the listener
14   * @author carcassi
15   */
16  public interface PVReaderListener<T> {
17  
18      /**
19       * Notified when the value of the PV has changed.
20       * 
21       * @param event the reader event
22       */
23      void pvChanged(PVReaderEvent<T> event);
24  
25  }

7   import java.util.ArrayList;
8   import java.util.List;
9   import java.util.concurrent.Executor;
10  import org.epics.pvmanager.expression.DesiredRateExpression;

14  /**
15   * An expression used to set the final parameters on how the pv expression
16   * should be read.
17   * 
18   * @param <T> the type of the expression
19   * @author carcassi
20   */
21  public class PVReaderConfiguration<T> extends CommonConfiguration {
22  
23      @Override
24      public PVReaderConfiguration<T> from(DataSource dataSource) {
25          super.from(dataSource);
26          return this;
27      }
28  
29      @Override
30      public PVReaderConfiguration<T> notifyOn(Executor onThread) {
31          super.notifyOn(onThread);
32          return this;
33      }
34  
35      /**
36       * Sets a timeout for no values received.
37       * <p>
38       * For more details, consult {@link #timeout(org.epics.util.time.TimeDuration, java.lang.String) }.
39       *
40       * @param timeout the duration of the timeout; can't be null
41       * @return this expression
42       */
43      @Override
44      public PVReaderConfiguration<T> timeout(TimeDuration timeout) {
45          super.timeout(timeout);
46          return this;
47      }
48  
49      /**
50       * Sets a timeout for no values received with the given message.
51       * <p>
52       * If no value is received before the given time, a {@link TimeoutException}
53       * is notified through the listener. Note the difference: the timeout is
54       * not on the connection but on the value itself. This allows to use timeouts
55       * when creating combined expressions that can produce data even if not
56       * all elements have values. For single channels, this means that if the
57       * channel is connected, but no value has been processed, a timeout
58       * exception is still sent.
59       *
60       * @param timeout the duration of the timeout; can't be null
61       * @param timeoutMessage the message for the reported timeout
62       * @return this expression
63       */
64      @Override
65      public PVReaderConfiguration<T> timeout(TimeDuration timeout, String timeoutMessage) {
66          super.timeout(timeout, timeoutMessage);
67          return this;
68      }
69      
70      private DesiredRateExpression<T> aggregatedPVExpression;
71      private ExceptionHandler exceptionHandler;
72      private List<PVReaderListener<T>> readListeners = new ArrayList<>();
73  
74      PVReaderConfiguration(DesiredRateExpression<T> aggregatedPVExpression) {
75          this.aggregatedPVExpression = aggregatedPVExpression;
76      }
77      
78      /**
79       * Adds a listener notified for any reader event (values, connection and errors).
80       * <p>
81       * Registering a listener here guarantees that no event is ever missed.
82       * 
83       * @param listener the listener to register
84       * @return this expression
85       */
86      public PVReaderConfiguration<T> readListener(PVReaderListener<? super T> listener) {
87          @SuppressWarnings("unchecked")
88          PVReaderListener<T> convertedListener = (PVReaderListener<T>) listener;
89          readListeners.add(convertedListener);
90          return this;
91      }
92  
93      /**
94       * Forwards exception to the given exception handler. No thread switch
95       * is done, so the handler is notified on the thread where the exception
96       * was thrown.
97       * <p>
98       * Giving a custom exception handler will disable the default handler,
99       * so {@link PVReader#lastException() } is no longer set and no notification
100      * is done.
101      *
102      * @param exceptionHandler an exception handler
103      * @return this
104      */
105     public PVReaderConfiguration<T> routeExceptionsTo(ExceptionHandler exceptionHandler) {
106         if (this.exceptionHandler != null) {
107             throw new IllegalArgumentException("Exception handler already set");
108         }
109         this.exceptionHandler = ExceptionHandler.safeHandler(exceptionHandler);
110         return this;
111     }
112     
113     /**
114      * Sets the rate of scan of the expression and creates the actual {@link PVReader}
115      * object that can be monitored through listeners.
116      * 
117      * @param rate the minimum time distance (i.e. the maximum rate) between two different notifications
118      * @return the PVReader
119      */
120     public PVReader<T> maxRate(TimeDuration rate) {
121         //int scanPeriodMs = (int) (period.getNanoSec() / 1000000);
122         
123         if (rate.getSec() < 0 && rate.getNanoSec() < 5000000) {
124             throw new IllegalArgumentException("Current implementation limits the rate to >5ms or <200Hz (requested " + rate + "s)");
125         }
126 
127         checkDataSourceAndThreadSwitch();
128 
129         // Create PVReader and connect
130         PVReaderImpl<T> pv = new PVReaderImpl<T>(aggregatedPVExpression.getName(), Executors.localThread() == notificationExecutor);
131         for (PVReaderListener<T> pVReaderListener : readListeners) {
132             pv.addPVReaderListener(pVReaderListener);
133         }
134         ReadFunction<T> aggregatedFunction = aggregatedPVExpression.getFunction();
135         
136         PVReaderDirector<T> director = new PVReaderDirector<T>(pv, aggregatedFunction, PVManager.getReadScannerExecutorService(),
137                 notificationExecutor, dataSource, exceptionHandler);
138         ScannerParameters scannerParameters = new ScannerParameters()
139                 .readerDirector(director)
140                 .scannerExecutor(PVManager.getReadScannerExecutorService())
141                 .maxDuration(rate);
142         if (timeout != null) {
143             if (timeoutMessage == null)
144                 timeoutMessage = "Read timeout";
145             director.timeout(timeout, timeoutMessage);
146         }
147 //        if (aggregatedFunction instanceof Collector || aggregatedFunction instanceof ValueCache) {
148 //            scannerParameters.type(ScannerParameters.Type.PASSIVE);
149 //        } else {
150             scannerParameters.type(ScannerParameters.Type.ACTIVE);
151 //        }
152         SourceDesiredRateDecoupler rateDecoupler = scannerParameters.build();
153         pv.setDirector(director);
154         director.setScanner(rateDecoupler);
155         director.connectExpression(aggregatedPVExpression);
156         rateDecoupler.start();
157 
158         return pv;
159     }

5   package org.epics.pvmanager;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   import java.util.concurrent.Executor;
10  import org.epics.pvmanager.expression.DesiredRateExpression;
11  import org.epics.pvmanager.util.Executors;
12  import org.epics.util.time.TimeDuration;
13  
14  /**
15   * An expression used to set the final parameters on how the pv expression
16   * should be read.
17   * 
18   * @param <T> the type of the expression
19   * @author carcassi
20   */
21  public class PVReaderConfiguration<T> extends CommonConfiguration {
22  
23      @Override
24      public PVReaderConfiguration<T> from(DataSource dataSource) {
25          super.from(dataSource);
26          return this;
27      }
28  
29      @Override
30      public PVReaderConfiguration<T> notifyOn(Executor onThread) {
31          super.notifyOn(onThread);
32          return this;
33      }
34  
35      /**
36       * Sets a timeout for no values received.
37       * <p>
38       * For more details, consult {@link #timeout(org.epics.util.time.TimeDuration, java.lang.String) }.
39       *
40       * @param timeout the duration of the timeout; can't be null
41       * @return this expression
42       */
43      @Override
44      public PVReaderConfiguration<T> timeout(TimeDuration timeout) {
45          super.timeout(timeout);
46          return this;
47      }
48  
49      /**
50       * Sets a timeout for no values received with the given message.
51       * <p>
52       * If no value is received before the given time, a {@link TimeoutException}
53       * is notified through the listener. Note the difference: the timeout is
54       * not on the connection but on the value itself. This allows to use timeouts
55       * when creating combined expressions that can produce data even if not
56       * all elements have values. For single channels, this means that if the
57       * channel is connected, but no value has been processed, a timeout
58       * exception is still sent.
59       *
60       * @param timeout the duration of the timeout; can't be null
61       * @param timeoutMessage the message for the reported timeout
62       * @return this expression
63       */
64      @Override
65      public PVReaderConfiguration<T> timeout(TimeDuration timeout, String timeoutMessage) {
66          super.timeout(timeout, timeoutMessage);
67          return this;
68      }
69      
70      private final DesiredRateExpression<T> aggregatedPVExpression;
71      private final List<PVReaderListener<T>> readListeners = new ArrayList<>();
72      private ExceptionHandler exceptionHandler;
73      private TimeDuration maxRate;
74      PVReaderImpl<T> pv;
75      ReadFunction<T> aggregatedFunction;
76  
77      PVReaderConfiguration(DesiredRateExpression<T> aggregatedPVExpression) {
78          this.aggregatedPVExpression = aggregatedPVExpression;
79      }
80      
81      /**
82       * Adds a listener notified for any reader event (values, connection and errors).
83       * <p>
84       * Registering a listener here guarantees that no event is ever missed.
85       * 
86       * @param listener the listener to register
87       * @return this expression
88       */
89      public PVReaderConfiguration<T> readListener(PVReaderListener<? super T> listener) {
90          @SuppressWarnings("unchecked")
91          PVReaderListener<T> convertedListener = (PVReaderListener<T>) listener;
92          readListeners.add(convertedListener);
93          return this;
94      }
95  
96      /**
97       * Forwards exception to the given exception handler. No thread switch
98       * is done, so the handler is notified on the thread where the exception
99       * was thrown.
100      * <p>
101      * Giving a custom exception handler will disable the default handler,
102      * so {@link PVReader#lastException() } is no longer set and no notification
103      * is done.
104      *
105      * @param exceptionHandler an exception handler
106      * @return this
107      */
108     public PVReaderConfiguration<T> routeExceptionsTo(ExceptionHandler exceptionHandler) {
109         if (this.exceptionHandler != null) {
110             throw new IllegalArgumentException("Exception handler already set");
111         }
112         this.exceptionHandler = ExceptionHandler.safeHandler(exceptionHandler);
113         return this;
114     }
115     
116     /**
117      * Sets the rate of scan of the expression and creates the actual {@link PVReader}
118      * object that can be monitored through listeners.
119      * 
120      * @param rate the minimum time distance (i.e. the maximum rate) between two different notifications
121      * @return the PVReader
122      */
123     public PVReader<T> maxRate(TimeDuration rate) {
124         maxRateAndValidate(rate);
125         
126         preparePvReader();
127         
128         PVDirector<T> director = prepareDirector(this);
129         prepareDecoupler(director, this);
130 
131         return pv;
132     }
133     
134     void maxRateAndValidate(TimeDuration rate) {
135         this.maxRate = rate;
136         validateReaderConfiguration();
137     }
138     
139     static <T> PVDirector<T> prepareDirector(PVReaderConfiguration<T> readConfiguration) {
140         PVDirector<T> director = new PVDirector<>(readConfiguration.pv, readConfiguration.aggregatedFunction, PVManager.getReadScannerExecutorService(),
141                 readConfiguration.notificationExecutor, readConfiguration.dataSource, readConfiguration.exceptionHandler);
142         if (readConfiguration.timeout != null) {
143             if (readConfiguration.timeoutMessage == null)
144                 readConfiguration.timeoutMessage = "Read timeout";
145             director.readTimeout(readConfiguration.timeout, readConfiguration.timeoutMessage);
146         }
147         return director;
148     }
149     
150     static <T> void prepareDecoupler(PVDirector<T> director, PVReaderConfiguration<T> readConfiguration) {
151         ScannerParameters scannerParameters = new ScannerParameters()
152                 .readerDirector(director)
153                 .scannerExecutor(PVManager.getReadScannerExecutorService())
154                 .maxDuration(readConfiguration.maxRate);
155         if (readConfiguration.aggregatedFunction instanceof Collector || readConfiguration.aggregatedFunction instanceof ValueCache) {
156             scannerParameters.type(ScannerParameters.Type.PASSIVE);
157         } else {
158             scannerParameters.type(ScannerParameters.Type.ACTIVE);
159         }
160         SourceDesiredRateDecoupler rateDecoupler = scannerParameters.build();
161         
162         readConfiguration.pv.setDirector(director);
163         director.setScanner(rateDecoupler);
164         director.connectReadExpression(readConfiguration.aggregatedPVExpression);
165         rateDecoupler.start();
166     }
167     
168     private void validateReaderConfiguration() {
169         if (maxRate.getSec() < 0 && maxRate.getNanoSec() < 5000000) {
170             throw new IllegalArgumentException("Current implementation limits the rate to >5ms or <200Hz (requested " + maxRate + "s)");
171         }
172 
173         checkDataSourceAndThreadSwitch();
174     }
175     
176     void preparePvReader() {
177         pv = new PVReaderImpl<>(aggregatedPVExpression.getName(), Executors.localThread() == notificationExecutor);
178         for (PVReaderListener<T> pVReaderListener : readListeners) {
179             pv.addPVReaderListener(pVReaderListener);
180         }
181         aggregatedFunction = aggregatedPVExpression.getFunction();
182     }
183 }

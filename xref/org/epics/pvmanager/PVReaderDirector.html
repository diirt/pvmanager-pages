20  /**
21   * Orchestrates the different elements of pvmanager to make a reader functional.
22   * <p>
23   * This class is responsible for the correct read operation, including:
24   * <ul>
25   * <li>Setting up the collector for notifications</li>
26   * <li>Setting up the collector for connection notification</li>
27   * <li>Building connection recipes and forwarding them to the datasource<li>
28   * <li>Managing the scanning task and notification for new values, connection status
29   * or errors</li>
30   * <li>Disconnecting the expressions from the datasources if the reader is closed
31   * or if it's garbage collected</li>
32   * </ul>
33   *
34   * @param <T> value type for the reader managed by this director
35   * @author carcassi
36   */
37  public class PVReaderDirector<T> {
38      
39      private static final Logger log = Logger.getLogger(PVReaderDirector.class.getName());
40  
41      // Required for connection and exception notification
42  
43      /** Executor used to notify of new values/connection/exception */
44      private final Executor notificationExecutor;
45      /** Executor used to scan the connection/exception queues */
46      private final ScheduledExecutorService scannerExecutor;
47      private volatile ScheduledFuture<?> scanTaskHandle;
48      /** PVReader to update during the notification */
49      private final WeakReference<PVReaderImpl<T>> pvRef;
50      /** Function for the new value */
51      private final ReadFunction<T> function;
52      
53      // Required to connect/disconnect expressions
54      private final DataSource dataSource;
55      private final Object lock = new Object();
56      private final Map<DesiredRateExpression<?>, ReadRecipe> recipes =
57              new HashMap<>();
58  
59      // Required for multiple operations
60      /** Connection collector required to connect/disconnect expressions and for connection notification */
61      private final ConnectionCollector connCollector =
62              new ConnectionCollector();
63      /** Exception queue to be used to connect/disconnect expression and for exception notification */
64      private final QueueCollector<Exception> exceptionCollector;
65      
66      
67      ReadRecipe getCurrentReadRecipe() {
68          ReadRecipeBuilder builder = new ReadRecipeBuilder();
69          for (Map.Entry<DesiredRateExpression<?>, ReadRecipe> entry : recipes.entrySet()) {
70              ReadRecipe readRecipe = entry.getValue();
71              for (ChannelReadRecipe channelReadRecipe : readRecipe.getChannelReadRecipes()) {
72                  builder.addChannel(channelReadRecipe.getChannelName(), channelReadRecipe.getReadSubscription().getValueCache());
73              }
74          }
75          return builder.build(exceptionCollector, connCollector);
76      }
77      
78      /**
79       * Connects the given expression.
80       * <p>
81       * This can be used for dynamic expression to add and connect child expressions.
82       * The added expression will be automatically closed when the associated
83       * reader is closed, if it's not disconnected first.
84       * 
85       * @param expression the expression to connect
86       */
87      public void connectExpression(DesiredRateExpression<?> expression) {
88          ReadRecipeBuilder builder = new ReadRecipeBuilder();
89          expression.fillReadRecipe(this, builder);
90          ReadRecipe recipe = builder.build(exceptionCollector, connCollector);
91          synchronized(lock) {
92              recipes.put(expression, recipe);
93          }
94          if (!recipe.getChannelReadRecipes().isEmpty()) {
95              try {
96                  dataSource.connectRead(recipe);
97              } catch(Exception ex) {
98                  recipe.getChannelReadRecipes().iterator().next().getReadSubscription().getExceptionWriteFunction().writeValue(ex);
99              }
100         }
101     }
102     
103     /**
104      * Simulate a static connection in which the channel has one exception
105      * and the connection will never change.
106      * <p>
107      * This is a temporary method an will be subject to change in the future.
108      * The aim is to allow to connect expressions that are not channels
109      * but can influence exception and connection state. For example,
110      * to report problems encountered during expression creation as runtime
111      * problems through the normal exception/connection methods.
112      * <p>
113      * In the future, this should be generalized to allow fully fledged expressions
114      * that connect/disconnect and can report errors.
115      * 
116      * @param ex the exception to queue
117      * @param connection the connection flag
118      * @param channelName the channel name
119      */
120     public void connectStatic(Exception ex, boolean connection, String channelName) {
121         exceptionCollector.writeValue(ex);
122         connCollector.addChannel(channelName).writeValue(connection);
123     }
124     
125     /**
126      * Disconnects the given expression.
127      * <p>
128      * This can be used for dynamic expression, to remove and disconnects child
129      * expressions.
130      *
131      * @param expression the expression to disconnect
132      */
133     public void disconnectExpression(DesiredRateExpression<?> expression) {
134         ReadRecipe recipe;
135         synchronized(lock) {
136             recipe = recipes.remove(expression);
137         }
138         if (recipe == null) {
139             log.log(Level.SEVERE, "Director was asked to disconnect expression '" + expression + "' which was not found.");
140         }
141         
142         if (!recipe.getChannelReadRecipes().isEmpty()) {
143             try {
144                 dataSource.disconnectRead(recipe);
145             } catch(Exception ex) {
146                 recipe.getChannelReadRecipes().iterator().next().getReadSubscription().getExceptionWriteFunction().writeValue(ex);
147             }
148         }
149     }
150     
151     private volatile boolean closed = false;
152     
153     void close() {
154         closed = true;
155     }
156 
157     /**
158      * Closed and disconnects all the child expressions.
159      */
160     private void disconnect() {
161         synchronized(lock) {
162             while (!recipes.isEmpty()) {
163                 DesiredRateExpression<?> expression = recipes.keySet().iterator().next();
164                 disconnectExpression(expression);
165             }
166         }
167     }
168 
169     /**
170      * Creates a new notifier. The new notifier will notifier the given pv
171      * with new values calculated by the function, and will use onThread to
172      * perform the notifications.
173      * <p>
174      * After construction, one MUST set the pvRecipe, so that the
175      * dataSource is appropriately closed.
176      *
177      * @param pv the pv on which to notify
178      * @param function the function used to calculate new values
179      * @param notificationExecutor the thread switching mechanism
180      */
181     PVReaderDirector(PVReaderImpl<T> pv, ReadFunction<T> function, ScheduledExecutorService scannerExecutor,
182             Executor notificationExecutor, DataSource dataSource, ExceptionHandler exceptionHandler) {
183         this.pvRef = new WeakReference<>(pv);
184         this.function = function;
185         this.notificationExecutor = notificationExecutor;
186         this.scannerExecutor = scannerExecutor;
187         this.dataSource = dataSource;
188         if (exceptionHandler == null) {
189             exceptionCollector = new QueueCollector<>(1);
190         } else {
191             exceptionCollector = new LastExceptionCollector(1, exceptionHandler);
192         }
193     }
194 
195     /**
196      * Determines whether the notifier is active or not.
197      * <p>
198      * The notifier becomes inactive if the PVReader is closed or is garbage collected.
199      * The first time this function determines that the notifier is inactive,
200      * it will ask the data source to close all channels relative to the
201      * pv.
202      *
203      * @return true if new notification should be performed
204      */
205     boolean isActive() {
206         // Making sure to get the reference once for thread safety
207         final PVReader<T> pv = pvRef.get();
208         if (pv != null && !pv.isClosed()) {
209             return true;
210         } else if (pv == null && closed != true) {
211             log.warning("PVReader wasn't properly closed and it was garbage collected. Closing the associated connections...");
212             return false;
213         } else {
214             return false;
215         }
216     }
217     
218     /**
219      * Checks whether the pv is paused
220      * 
221      * @return true if paused
222      */
223     boolean isPaused() {
224         final PVReader<T> pv = pvRef.get();
225         if (pv == null || pv.isPaused()) {
226             return true;
227         } else {
228             return false;
229         }
230     }
231     
232     private volatile boolean notificationInFlight = false;
233     
234     /**
235      * Notifies the PVReader of a new value.
236      */
237     void notifyPv() {
238         // Don't even calculate if notification is in flight.
239         // This makes pvManager automatically throttle back if the consumer
240         // is slower than the producer.
241         if (notificationInFlight)
242             return;
243         
244         // Calculate new value
245         T newValue = null;
246         boolean calculationSucceeded = false;
247         try {
248             // Tries to calculate the value
249             newValue = function.readValue();
250             if (newValue != null) {
251                 NotificationSupport.findNotificationSupportFor(newValue);
252             }
253             calculationSucceeded = true;
254         } catch (RuntimeException ex) {
255             // Calculation failed
256             exceptionCollector.writeValue(ex);
257         } catch (Throwable ex) {
258             log.log(Level.SEVERE, "Unrecoverable error during scanning", ex);
259         }
260         
261         // Calculate new connection
262         final boolean connected = connCollector.readValue();
263         List<Exception> exceptions = exceptionCollector.readValue();
264         final Exception lastException;
265         if (exceptions.isEmpty()) {
266             lastException = null;
267         } else {
268             lastException = exceptions.get(exceptions.size() - 1);
269         }
270         
271         // TODO: if payload is immutable, the difference test should be done here
272         // and not in the runnable (to save SWT time)
273         
274         // Prepare values to ship to the other thread.
275         // The data will be shipped as part of the task,
276         // which is properly synchronized by the executor
277         final T finalValue = newValue;
278         final boolean finalCalculationSucceeded = calculationSucceeded;
279         notificationInFlight = true;
280         notificationExecutor.execute(new Runnable() {
281 
282             @Override
283             public void run() {
284                 try {
285                     PVReaderImpl<T> pv = pvRef.get();
286                     // Proceed with notification only if PVReader was not garbage
287                     // collected
288                     if (pv != null) {
289                         
290                         // Atomicity guaranteed by:
291                         //  - all the modification on the PVReader
292                         //    are done here, on the same thread where the listeners will be called.
293                         //    This means the callbacks are guaranteed to run after all
294                         //    changes are done
295                         //  - notificationInFlight guarantees that no other notification
296                         //    will run while one notification is running. This means
297                         //    the next event is serialized after the end of this one.
298                         pv.setConnected(connected);
299                         if (lastException != null) {
300                             pv.setLastException(lastException);
301                         }
302                         
303                         // XXX Are we sure that we should skip notifications if values are null?
304                         if (finalCalculationSucceeded && finalValue != null) {
305                             Notification<T> notification =
306                                     NotificationSupport.notification(pv.getValue(), finalValue);
307                             // Remember to notify anyway if an exception need to be notified
308                             if (notification.isNotificationNeeded()) {
309                                 pv.setValue(notification.getNewValue());
310                             } else if (pv.isLastExceptionToNotify() || pv.isReadConnectionToNotify()) {
311                                 pv.firePvValueChanged();
312                             }
313                         } else {
314                             // Remember to notify anyway if an exception need to be notified
315                             if (pv.isLastExceptionToNotify() || pv.isReadConnectionToNotify()) {
316                                 pv.firePvValueChanged();
317                             }
318                         }
319                     }
320                 } finally {
321                     notificationInFlight = false;
322                 }
323             }
324         });
325     }
326     
327     void startScan(TimeDuration duration) {
328         scanTaskHandle = scannerExecutor.scheduleWithFixedDelay(new Runnable() {
329 
330             @Override
331             public void run() {
332                 if (isActive()) {
333                     // If paused, simply skip without stopping the scan
334                     if (!isPaused()) {
335                         notifyPv();
336                     }
337                 } else {
338                     stopScan();
339                     disconnect();
340                 }
341             }
342         }, 0, duration.toNanosLong(), TimeUnit.NANOSECONDS);
343     }
344     
345     void timeout(TimeDuration timeout, final String timeoutMessage) {
346         scannerExecutor.schedule(new Runnable() {
347 
348             @Override
349             public void run() {
350                 PVReaderImpl<T> pv = pvRef.get();
351                 if (pv != null && !pv.isSentFirsEvent()) {
352                     exceptionCollector.writeValue(new TimeoutException(timeoutMessage));
353                 }
354             }
355         }, timeout.toNanosLong(), TimeUnit.NANOSECONDS);
356     }
357     
358     void stopScan() {
359         if (scanTaskHandle != null) {
360             scanTaskHandle.cancel(false);
361             scanTaskHandle = null;
362         } else {
363             throw new IllegalStateException("Scan was never started");
364         }
365     }

5   package org.epics.pvmanager.sample.graphene;
6   
7   import java.awt.event.MouseEvent;
8   import org.epics.graphene.AreaGraph2DRendererUpdate;
9   import static org.epics.pvmanager.formula.ExpressionLanguage.formula;
10  import static org.epics.pvmanager.graphene.ExpressionLanguage.*;
11  import org.epics.pvmanager.graphene.HistogramGraph2DExpression;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  public class HistogramGraphApp extends BaseGraphApp<AreaGraph2DRendererUpdate> {
18      
19      private boolean highlightFocusValue;
20  
21      public HistogramGraphApp() {
22          dataFormulaField.setModel(new javax.swing.DefaultComboBoxModel<String>(
23                  new String[] { "sim://gaussianWaveform",
24                      "=histogramOf('sim://noiseWaveform')",
25                      "=arrayWithBoundaries(arrayOf(1,3,2,4,3,5), range(-10,10))",
26                      "=caHistogram(\"histo\")"}));
27      }
28  
29      @Override
30      protected HistogramGraph2DExpression createExpression(String dataFormula) {
31          HistogramGraph2DExpression plot = histogramGraphOf(formula(dataFormula));
32          plot.update(plot.newUpdate().highlightFocusValue(highlightFocusValue));
33          return plot;
34      }
35  
36      @Override
37      protected void onMouseMove(MouseEvent e) {
38          graph.update(graph.newUpdate().focusPixel(e.getX()));
39      }
40  
41      public boolean isHighlightFocusValue() {
42          return highlightFocusValue;
43      }
44  
45      public void setHighlightFocusValue(boolean highlightFocusValue) {
46          this.highlightFocusValue = highlightFocusValue;
47          graph.update(graph.newUpdate().highlightFocusValue(highlightFocusValue));
48      }
49  
50      @Override
51      protected void openConfigurationDialog() {
52          HistogramGraphDialog dialog = new HistogramGraphDialog(new javax.swing.JFrame(), true, this);
53          dialog.setTitle("Configure...");
54          dialog.setLocationRelativeTo(this);
55          dialog.setVisible(true);
56      }
57      
58      public static void main(String[] args) {
59          main(HistogramGraphApp.class);
60      }
61      
62  }

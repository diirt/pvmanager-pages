74      private void bindResize(final int i) {
75          imagePanels.get(i).addComponentListener(new ComponentAdapter() {
76  
77              @Override
78              public void componentResized(ComponentEvent e) {
79                  if (plots.get(i) != null) {
80                      plots.get(i).update(plots.get(i).newUpdate().imageHeight(imagePanels.get(i).getHeight()).imageWidth(imagePanels.get(i).getWidth()));
81                  }
82              }
83          });
84      }
85      
86      private void bindPvChange(final int i) {
87          pvFields.get(i).addActionListener(new ActionListener() {
88  
89              @Override
90              public void actionPerformed(ActionEvent e) {
91                  pvChange(i);
92              }
93          });
94          pvChange(i);
95      }
96      
97      private void pvChange(final int i) {
98          if (pvs.get(i) != null) {
99              pvs.get(i).close();
100             imagePanels.get(i).setImage(null);
101             plots.set(i, null);
102         }
103         
104         if (pvFields.get(i).getText() == null || pvFields.get(i).getText().trim().isEmpty()) {
105             return;
106         }
107 
108         SparklineGraph2DExpression plot = sparklineGraphOf(formula(pvFields.get(i).getText()),
109                     null,
110                     null);
111         
112         plots.set(i, plot);
113         
114         plots.get(i).update(plots.get(i).newUpdate().imageHeight(imagePanels.get(i).getHeight())
115                 .imageWidth(imagePanels.get(i).getWidth()));
116         pvs.set(i, PVManager.read(plots.get(i))
117                 .notifyOn(swingEDT())
118                 .readListener(new PVReaderListener<Graph2DResult>() {
119 
120                     @Override
121                     public void pvChanged(PVReaderEvent<Graph2DResult> event) {
122                         setLastError(event.getPvReader().lastException());
123                         if (event.getPvReader().getValue() != null) {
124                             BufferedImage image = ValueUtil.toImage(event.getPvReader().getValue().getImage());
125                             imagePanels.get(i).setImage(image);
126                         }
127                     }
128                 })
129                 .maxRate(ofHertz(50)));
130     }

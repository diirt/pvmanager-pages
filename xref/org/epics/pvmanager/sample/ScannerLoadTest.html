7   import java.util.ArrayList;
8   import java.util.List;
9   import org.epics.pvmanager.PV;
10  import org.epics.pvmanager.PVManager;

12  import static org.epics.pvmanager.ExpressionLanguage.*;
13  import org.epics.util.time.TimeDuration;

15  /**
16   * Load test to see the effect on the performance of the scanning.
17   * <p>
18   * The test opens a number of local pvs. Since there is no load associated
19   * on keeping the local pv open, the total cpu load can be assumed to be
20   * the scanning. If can also be used to optimize memory consumption.
21   * <p>
22   * 2012/11/12 - 1,000 pvs, 1.8%; 10,000 pvs, 18.9%; 100,000 pvs, 34%. GC time ~0%
23   *
24   * @author carcassi
25   */
26  public class ScannerLoadTest {
27      public static void main(String[] args) throws Exception {
28          PVManager.setDefaultDataSource(new LocalDataSource());
29          
30          List<PV<Object, Object>> pvs = new ArrayList<>();
31          int nPvs = 100000;
32          for (int i = 0; i < nPvs; i++) {
33              PV<Object, Object> pv = PVManager.readAndWrite(channel("channel " + i)).asynchWriteAndMaxReadRate(TimeDuration.ofHertz(50));
34              pvs.add(pv);
35              
36          }
37          System.out.println("Started");
38          
39          Thread.sleep(60000);
40          
41          for (PV<Object, Object> pv : pvs) {
42              pv.close();
43          }
44          
45      }
46  }

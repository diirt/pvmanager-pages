72      private void bindResize(final int i) {
73          imagePanels.get(i).addComponentListener(new ComponentAdapter() {
74  
75              @Override
76              public void componentResized(ComponentEvent e) {
77                  if (plots.get(i) != null) {
78                      plots.get(i).update(plots.get(i).newUpdate().imageHeight(imagePanels.get(i).getHeight()).imageWidth(imagePanels.get(i).getWidth()));
79                  }
80              }
81          });
82      }
83      
84      private void bindPvChange(final int i) {
85          pvFields.get(i).addActionListener(new ActionListener() {
86  
87              @Override
88              public void actionPerformed(ActionEvent e) {
89                  pvChange(i);
90              }
91          });
92          pvChange(i);
93      }
94      
95      private void pvChange(final int i) {
96          if (pvs.get(i) != null) {
97              pvs.get(i).close();
98              imagePanels.get(i).setImage(null);
99              plots.set(i, null);
100         }
101         
102         if (pvFields.get(i).getText() == null || pvFields.get(i).getText().trim().isEmpty()) {
103             return;
104         }
105 
106         SparklineGraph2DExpression plot = sparklineGraphOf(formula(pvFields.get(i).getText()),
107                     null,
108                     null);
109         
110         plots.set(i, plot);
111         
112         plots.get(i).update(plots.get(i).newUpdate().imageHeight(imagePanels.get(i).getHeight())
113                 .imageWidth(imagePanels.get(i).getWidth()));
114         pvs.set(i, PVManager.read(plots.get(i))
115                 .notifyOn(swingEDT())
116                 .readListener(new PVReaderListener<Graph2DResult>() {
117 
118                     @Override
119                     public void pvChanged(PVReaderEvent<Graph2DResult> event) {
120                         setLastError(event.getPvReader().lastException());
121                         if (event.getPvReader().getValue() != null) {
122                             BufferedImage image = ValueUtil.toImage(event.getPvReader().getValue().getImage());
123                             imagePanels.get(i).setImage(image);
124                         }
125                     }
126                 })
127                 .maxRate(ofHertz(50)));
128     }

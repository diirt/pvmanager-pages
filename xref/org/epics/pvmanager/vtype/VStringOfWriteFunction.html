5   package org.epics.pvmanager.vtype;
6   
7   import org.epics.vtype.ValueFormat;
8   import org.epics.vtype.VType;
9   import org.epics.pvmanager.ReadFunction;
10  import org.epics.pvmanager.WriteFunction;
11  
12  /**
13   * Converts the value of the argument to a VString.
14   *
15   * @author carcassi
16   */
17  class VStringOfWriteFunction implements WriteFunction<String> {
18      
19      private final ReadFunction<? extends VType> reference;
20      private final WriteFunction<Object> argument;
21      private final ValueFormat format;
22  
23      public VStringOfWriteFunction(ReadFunction<? extends VType> reference, ValueFormat format, WriteFunction<Object> argument) {
24          this.reference = reference;
25          this.format = format;
26          this.argument = argument;
27      }
28  
29      @Override
30      public void writeValue(String newValue) {
31          argument.writeValue(format.parseObject(newValue, reference.readValue()));
32      }
33      
34  }

178     /**
179      * Transforms a list of numeric scalar into an array.
180      * 
181      * @param expressions a list of numeric expressions
182      * @return a new numeric array expression
183      */
184     public static DesiredRateExpression<VNumberArray>
185             vNumberArrayOf(DesiredRateExpressionList<? extends VNumber> expressions) {
186         // TODO - there should be a common function to extract the list of functions
187         List<ReadFunction<? extends VNumber>> functions = new ArrayList<ReadFunction<? extends VNumber>>();
188         for (DesiredRateExpression<? extends VNumber> expression : expressions.getDesiredRateExpressions()) {
189             functions.add(expression.getFunction());
190         }
191         VNumbersToVNumberArrayConverter converter =
192                 new VNumbersToVNumberArrayConverter(functions);
193         return new DesiredRateExpressionImpl<VNumberArray>(expressions, converter, "numberArrayOf");
194     }

448      * @param valueFormat the format to use for the conversion
449      * @return an expression with the string representation of the argument

451     public static DesiredRateReadWriteExpression<VString, String> vStringOf(DesiredRateReadWriteExpression<? extends VType, ? extends Object> expression,
452             ValueFormat valueFormat) {
453         LatestValueCollector<VType> forward = new LatestValueCollector<>();
454         DesiredRateExpression<VString> readExp = new DesiredRateExpressionImpl<>(expression,
455                 new VStringOfFunction(expression.getFunction(), valueFormat, forward)
456                 , expression.getName());
457         @SuppressWarnings("unchecked")
458         WriteFunction<Object> writeFunction = (WriteFunction<Object>) (WriteFunction) expression.getWriteFunction();
459         WriteExpression<String> writeExp = new WriteExpressionImpl<>(expression,
460                 new VStringOfWriteFunction(forward, valueFormat, writeFunction), expression.getName());
461         return new DesiredRateReadWriteExpressionImpl<>(readExp, writeExp);
462     }

515     /**
516      * A synchronized array from the given expression.
517      *
518      * @param tolerance maximum time difference between samples in the
519      * reconstructed array
520      * @param cacheDepth maximum time difference between samples in the caches
521      * used to reconstruct the array
522      * @param expressions the expressions from which to reconstruct the array
523      * @return an expression for the array
524      */
525     public static DesiredRateExpression<VMultiDouble>
526             synchronizedArrayOf(TimeDuration tolerance, TimeDuration cacheDepth, SourceRateExpressionList<VDouble> expressions) {
527         if (cacheDepth.equals(TimeDuration.ofMillis(0)) && cacheDepth.getSec() > 0)
528             throw new IllegalArgumentException("Distance between samples must be non-zero and positive");
529         List<String> names = new ArrayList<String>();
530         List<ReadFunction<List<VDouble>>> collectors = new ArrayList<ReadFunction<List<VDouble>>>();
531         DesiredRateExpressionList<List<VDouble>> desiredRateExpressions = new DesiredRateExpressionListImpl<List<VDouble>>();
532         for (SourceRateExpression<VDouble> expression : expressions.getSourceRateExpressions()) {
533             DesiredRateExpression<List<VDouble>> collectorExp = timedCacheOf(expression, cacheDepth);
534             desiredRateExpressions.and(collectorExp);
535             collectors.add(collectorExp.getFunction());
536             names.add(expression.getName());
537         }
538         SynchronizedVDoubleAggregator aggregator =
539                 new SynchronizedVDoubleAggregator(names, collectors, tolerance);
540         return new DesiredRateExpressionImpl<VMultiDouble>(desiredRateExpressions,
541                 (ReadFunction<VMultiDouble>) aggregator, "syncArray");
542     }

571     /**
572      * Creates a vTable by aggregating different values from different pvs.
573      * 
574      * @param columns columns of the table
575      * @return an expression for the table
576      */
577     public static DesiredRateExpression<VTable> vTable(VTableColumn... columns) {
578         DesiredRateExpressionListImpl<Object> list = new DesiredRateExpressionListImpl<Object>();
579         List<List<ReadFunction<?>>> functions = new ArrayList<List<ReadFunction<?>>>();
580         List<String> names = new ArrayList<String>();
581         for (VTableColumn column : columns) {
582             functions.add(Expressions.functionsOf(column.getValueExpressions()));
583             list.and(column.getValueExpressions());
584             names.add(column.getName());
585         }
586         return new DesiredRateExpressionImpl<VTable>(list, new VTableAggregationFunction(functions, names), "table");
587     }

5   package org.epics.pvmanager.sys;
6   
7   import java.net.InetAddress;
8   import java.net.UnknownHostException;
9   import java.util.Objects;
10  import java.util.logging.Level;
11  import java.util.logging.Logger;
12  import org.epics.vtype.Alarm;
13  import org.epics.vtype.AlarmSeverity;
14  import static org.epics.vtype.ValueFactory.*;
15  
16  /**
17   *
18   * @author carcassi
19   */
20  class QualifiedHostnameChannelHandler extends SystemChannelHandler {
21      
22      private String previousValue = null;
23  
24      public QualifiedHostnameChannelHandler(String channelName) {
25          super(channelName);
26      }
27  
28      @Override
29      protected Object createValue() {
30          String hostname;
31          Alarm alarm;
32          try {
33              hostname = InetAddress.getLocalHost().getCanonicalHostName();
34              alarm = alarmNone();
35          } catch (UnknownHostException ex) {
36              hostname = "Unknown host";
37              alarm = newAlarm(AlarmSeverity.INVALID, "Undefined");
38          }
39          if (!Objects.equals(hostname, previousValue)) {
40              previousValue = hostname;
41              return newVString(hostname, alarm, timeNow());
42          } else {
43              return null;
44          }
45      }
46      
47  }

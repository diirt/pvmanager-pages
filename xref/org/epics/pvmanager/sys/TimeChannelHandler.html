5   package org.epics.pvmanager.sys;
6   
7   import org.epics.util.time.Timestamp;
8   import org.epics.util.time.TimestampFormat;
9   import static org.epics.vtype.ValueFactory.*;
10  
11  /**
12   *
13   * @author carcassi
14   */
15  class TimeChannelHandler extends SystemChannelHandler {
16  
17      private static final TimestampFormat timeFormat = new TimestampFormat("yyyy/MM/dd HH:mm:ss.SSS");    
18  
19      public TimeChannelHandler(String channelName) {
20          super(channelName);
21      }
22  
23      @Override
24      protected Object createValue() {
25          Timestamp time = Timestamp.now();
26          String formatted = timeFormat.format(time);
27          return newVString(formatted, alarmNone(), newTime(time));
28      }
29      
30  }

5   package org.epics.pvmanager.integration;
6   
7   import org.epics.vtype.VTypeToString;
8   import org.epics.vtype.VTypeValueEquals;
9   import org.epics.vtype.ValueUtil;
10  
11  /**
12   *
13   * @author carcassi
14   */
15  public enum VTypeMatchMask {
16  
17      VALUE(true, false, false, false),
18      ALARM(false, true, false, false),
19      TIME(false, false, true, false),
20  //    DISPLAY(false, false, false, true),
21      ALL(true, true, true, true),
22      ALL_EXCEPT_TIME(true, true, false, true);
23      
24      private final boolean value;
25      private final boolean alarm;
26      private final boolean time;
27      private final boolean display;
28  
29      private VTypeMatchMask(boolean value, boolean alarm, boolean time, boolean display) {
30          this.value = value;
31          this.alarm = alarm;
32          this.time = time;
33          this.display = display;
34      }
35  
36      public String match(Object expectedValue, Object actualValue) {
37          if (value && !VTypeValueEquals.typeEquals(actualValue, expectedValue)) {
38              return "TYPE mismatch: was " + actualValue + " (expected " + expectedValue + ")";
39          } else if (value && !VTypeValueEquals.valueEquals(actualValue, expectedValue)) {
40              return "VALUE mismatch: was " + actualValue + " (expected " + expectedValue + ")";
41          } else if (alarm && !VTypeValueEquals.alarmEquals(ValueUtil.alarmOf(actualValue), ValueUtil.alarmOf(expectedValue))) {
42              return "ALARM mismatch: was " + VTypeToString.alarmToString(ValueUtil.alarmOf(actualValue))
43                      + " (expected " + VTypeToString.alarmToString(ValueUtil.alarmOf(expectedValue)) + ")";
44          } else if (time && !VTypeValueEquals.timeEquals(ValueUtil.timeOf(actualValue), ValueUtil.timeOf(expectedValue))) {
45              return "TIME mismatch: was " + actualValue + " (expected " + expectedValue + ")";
46          }
47          return null;
48      }
49      
50      
51  }

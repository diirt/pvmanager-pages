8   import org.epics.vtype.VImage;
9   import org.epics.vtype.ValueUtil;
10  import java.awt.image.BufferedImage;
11  import java.util.List;
12  import org.epics.graphene.*;
13  import org.epics.pvmanager.QueueCollector;
14  import org.epics.pvmanager.ReadFunction;
15  import static org.epics.pvmanager.graphene.ArgumentExpressions.stringArgument;
16  import org.epics.vtype.VTable;
17  import org.epics.vtype.VType;
18  
19  /**
20   *
21   * @author carcassi
22   */
23  class SparklineGraph2DFunction implements ReadFunction<Graph2DResult> {
24      
25      private ReadFunction<VType> tableData;
26      private ReadFunctionArgument<String> xColumnName;
27      private ReadFunctionArgument<String> yColumnName;
28      
29      private SparklineGraph2DRenderer renderer = new SparklineGraph2DRenderer(300, 200);
30      
31      private VImage previousImage;
32      private final QueueCollector<SparklineGraph2DRendererUpdate> rendererUpdateQueue = new QueueCollector<>(100);
33  
34      SparklineGraph2DFunction(ReadFunction<?> tableData,
35  	    ReadFunction<?> xColumnName,
36  	    ReadFunction<?> yColumnName) {
37          this.tableData = new CheckedReadFunction<VType>(tableData, "Data", VTable.class, VNumberArray.class);
38          this.xColumnName = stringArgument(xColumnName, "X Column");
39          this.yColumnName = stringArgument(yColumnName, "Y Column");
40      }
41  
42      public QueueCollector<SparklineGraph2DRendererUpdate> getRendererUpdateQueue() {
43          return rendererUpdateQueue;
44      }
45  
46      @Override
47      public Graph2DResult readValue() {
48          VType vType = tableData.readValue();
49          xColumnName.readNext();
50          yColumnName.readNext();
51          
52          // Table and columns must be available
53          if (vType == null || xColumnName.isMissing() || yColumnName.isMissing()) {
54              return null;
55          }
56  
57          // Prepare new dataset
58          Point2DDataset dataset;
59          if (vType instanceof VNumberArray) {
60              dataset = Point2DDatasets.lineData(((VNumberArray) vType).getData());
61          } else {
62              dataset = DatasetConversions.point2DDatasetFromVTable((VTable) vType, xColumnName.getValue(), yColumnName.getValue());
63          }
64          
65          // Process all renderer updates
66          List<SparklineGraph2DRendererUpdate> updates = rendererUpdateQueue.readValue();
67          for (SparklineGraph2DRendererUpdate rendererUpdate : updates) {
68              renderer.update(rendererUpdate);
69          }
70          
71          // If no size is set, don't calculate anything
72          if (renderer.getImageHeight() == 0 && renderer.getImageWidth() == 0)
73              return null;
74          
75          BufferedImage image = new BufferedImage(renderer.getImageWidth(), renderer.getImageHeight(), BufferedImage.TYPE_3BYTE_BGR);
76          renderer.draw(image.createGraphics(), dataset);
77          
78          previousImage = ValueUtil.toVImage(image);
79          return new Graph2DResult(vType, previousImage,
80                  new GraphDataRange(renderer.getXPlotRange(), dataset.getXStatistics(), renderer.getXAggregatedRange()),
81                  new GraphDataRange(renderer.getYPlotRange(), dataset.getYStatistics(), renderer.getYAggregatedRange()),
82                  -1);
83      }
84      
85  }

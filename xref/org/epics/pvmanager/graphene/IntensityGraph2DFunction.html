32  class IntensityGraph2DFunction implements ReadFunction<Graph2DResult> {
33      
34      private ReadFunction<VNumberArray> arrayData;
35      
36      private IntensityGraph2DRenderer renderer = new IntensityGraph2DRenderer(300, 200);
37      
38      private VNumberArray oldData;
39      private Graph2DResult previousResult;
40      private final QueueCollector<IntensityGraph2DRendererUpdate> rendererUpdateQueue = new QueueCollector<>(100);
41  
42      public IntensityGraph2DFunction(ReadFunction<?> arrayData) {
43          this.arrayData = new CheckedReadFunction<VNumberArray>(arrayData, "Data", VNumberArray.class);
44      }
45      
46      public QueueCollector<IntensityGraph2DRendererUpdate> getUpdateQueue() {
47          return rendererUpdateQueue;
48      }
49  
50      @Override
51      public Graph2DResult readValue() {
52          VNumberArray data = arrayData.readValue();
53          
54          // Data must be available
55          if (data == null) {
56              return null;
57          }
58  
59          List<IntensityGraph2DRendererUpdate> updates = getUpdateQueue().readValue();
60          
61          // If data is old and no updates, return the previous result
62          if (data == oldData && updates.isEmpty()) {
63              return previousResult;
64          }
65          
66          oldData = data;
67          
68          // TODO: check array is one dimensional
69  
70          Cell2DDataset dataset = DatasetConversions.cell2DDatasetsFromVNumberArray(data);
71          
72          // Process all renderer updates
73          for (IntensityGraph2DRendererUpdate rendererUpdate : updates) {
74              renderer.update(rendererUpdate);
75          }
76          
77          // If no size is set, don't calculate anything
78          if (renderer.getImageHeight() == 0 && renderer.getImageWidth() == 0)
79              return null;
80          
81          GraphBuffer buffer = new GraphBuffer(renderer);
82          renderer.draw(buffer, dataset);
83          
84          return new Graph2DResult(data, ValueUtil.toVImage(buffer.getImage()),
85                  new GraphDataRange(renderer.getXPlotRange(), dataset.getXRange(), dataset.getXRange()),
86                  new GraphDataRange(renderer.getYPlotRange(), dataset.getYRange(), dataset.getYRange()),
87                  -1, selectionData(data, renderer));
88      }
89      
90      private VNumberArray selectionData(VNumberArray data, IntensityGraph2DRenderer renderer) {
91          if (renderer.getXIndexSelectionRange() == null ||
92                  renderer.getYIndexSelectionRange() == null) {
93              return null;
94          }
95          
96          ArrayDimensionDisplay xDisplay = data.getDimensionDisplay().get(1);
97          int leftIndex = renderer.getXIndexSelectionRange().getMinimum().intValue();
98          int rightIndex = renderer.getXIndexSelectionRange().getMaximum().intValue();
99          final int xSize = rightIndex - leftIndex + 1;
100         if (xDisplay.isReversed()) {
101             leftIndex = data.getSizes().getInt(1) - 1 - leftIndex;
102             rightIndex = data.getSizes().getInt(1) - 1 - rightIndex;
103         }
104         
105         ArrayDimensionDisplay yDisplay = data.getDimensionDisplay().get(0);
106         int bottomIndex = renderer.getYIndexSelectionRange().getMinimum().intValue();
107         int topIndex = renderer.getYIndexSelectionRange().getMaximum().intValue();
108         if (yDisplay.isReversed()) {
109             bottomIndex = data.getSizes().getInt(0) - 1 - bottomIndex;
110             topIndex = data.getSizes().getInt(0) - 1 - topIndex;
111         }
112         final int ySize = topIndex - bottomIndex + 1;
113 
114         return ValueFactory.newVNumberArray(new ListDouble() {
115 
116             @Override
117             public double getDouble(int index) {
118                 throw new UnsupportedOperationException("Not supported yet.");
119             }
120 
121             @Override
122             public int size() {
123                 return xSize * ySize;
124             }
125         }, new ArrayInt(ySize, xSize), Arrays.asList(ValueFactory.newDisplay(ListMath.limit(yDisplay.getCellBoundaries(), bottomIndex, topIndex + 1), yDisplay.getUnits()),
126                 ValueFactory.newDisplay(ListMath.limit(xDisplay.getCellBoundaries(), leftIndex, rightIndex + 1), yDisplay.getUnits())), data, data, data);
127     }

5   package org.epics.pvmanager.graphene;
6   
7   import java.util.List;
8   import org.epics.pvmanager.ReadFunction;
9   import org.epics.vtype.VStringArray;
10  
11  /**
12   *
13   * @author carcassi
14   */
15  public class VStringArrayToListStringReadFunction implements ReadFunction<List<String>> {
16      
17      private final ReadFunction<VStringArray> function;
18  
19      public VStringArrayToListStringReadFunction(ReadFunction<VStringArray> function) {
20          this.function = function;
21      }
22  
23      @Override
24      public List<String> readValue() {
25          VStringArray value = function.readValue();
26          if (value == null) {
27              return null;
28          } else {
29              return value.getData();
30          }
31      }
32  
33  }

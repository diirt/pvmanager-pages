25  class HistogramGraph2DFunction implements ReadFunction<Graph2DResult> {
26      
27      private ReadFunction<VNumberArray> arrayData;
28      
29      private AreaGraph2DRenderer renderer = new AreaGraph2DRenderer(300, 200);
30      
31      private Graph2DResult previousImage;
32      private final QueueCollector<AreaGraph2DRendererUpdate> rendererUpdateQueue = new QueueCollector<>(100);
33  
34      public HistogramGraph2DFunction(ReadFunction<?> arrayData) {
35          this.arrayData = new CheckedReadFunction<VNumberArray>(arrayData, "Data", VNumberArray.class);
36      }
37      
38      public QueueCollector<AreaGraph2DRendererUpdate> getUpdateQueue() {
39          return rendererUpdateQueue;
40      }
41  
42      @Override
43      public Graph2DResult readValue() {
44          VNumberArray data = arrayData.readValue();
45          
46          // Data must be available
47          if (data == null) {
48              return null;
49          }
50          
51          // TODO: check array is one dimensional
52  
53          Cell1DDataset dataset = null;
54          try {
55              dataset = Cell1DDatasets.datasetFrom(data.getData(), data.getDimensionDisplay().get(0).getCellBoundaries());
56          } catch (Throwable e) {
57              e.printStackTrace();
58          }
59          // Process all renderer updates
60          for (AreaGraph2DRendererUpdate rendererUpdate : getUpdateQueue().readValue()) {
61              renderer.update(rendererUpdate);
62          }
63          
64          // If no size is set, don't calculate anything
65          if (renderer.getImageHeight() == 0 && renderer.getImageWidth() == 0)
66              return null;
67          
68          BufferedImage image = new BufferedImage(renderer.getImageWidth(), renderer.getImageHeight(), BufferedImage.TYPE_3BYTE_BGR);
69          renderer.draw(image.createGraphics(), dataset);
70          int index = -1;
71          if (renderer.getFocusValueIndex() != null) {
72              index = renderer.getFocusValueIndex();
73          }
74          
75          return new Graph2DResult(data, ValueUtil.toVImage(image),
76                  new GraphDataRange(renderer.getXPlotRange(), dataset.getXRange(), dataset.getXRange()),
77                  new GraphDataRange(renderer.getYPlotRange(), dataset.getStatistics(), renderer.getYAggregatedRange()),
78                  index);
79      }
80      
81  }

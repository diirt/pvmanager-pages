5   package org.epics.pvmanager.graphene;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   import org.epics.pvmanager.ReadFunction;
10  import org.epics.vtype.ValueUtil;
11  
12  /**
13   *
14   * @author carcassi
15   */
16  public class CheckedReadFunction<T> implements ReadFunction<T> {
17      
18      private final Class<T> clazz;
19      private final Class<? extends T>[] classes;
20      private final ReadFunction<?> arg;
21      private final String argName;
22  
23      public CheckedReadFunction(ReadFunction<?> arg, String argName, Class<T> clazz) {
24          if (clazz == null) {
25              throw new NullPointerException("Class can't be null");
26          }
27          this.clazz = clazz;
28          this.classes = null;
29          this.arg = arg;
30          this.argName = argName;
31      }
32  
33      @SafeVarargs
34      public CheckedReadFunction(ReadFunction<?> arg, String argName, Class<? extends T>... classes) {
35          if (classes == null) {
36              throw new NullPointerException("Classes can't be null");
37          }
38          this.clazz = null;
39          this.classes = classes;
40          this.arg = arg;
41          this.argName = argName;
42      }
43      
44      @Override
45      public T readValue() {
46          Object obj = arg.readValue();
47          if (obj == null) {
48              return null;
49          }
50  
51          if (clazz != null) {
52              if (clazz.isInstance(obj)) {
53                  return clazz.cast(obj);
54              } else {
55                  throw new RuntimeException(argName + " must be a " + clazz.getSimpleName() + " (was " + ValueUtil.typeOf(obj).getSimpleName() + ")");
56              }
57          } else {
58              for (Class<? extends T> aClass : classes) {
59                  if (aClass.isInstance(obj)) {
60                      return aClass.cast(obj);
61                  }
62              }
63              StringBuilder sb = new StringBuilder();
64              List<String> names = new ArrayList<>();
65              for (Class<? extends T> aClass : classes) {
66                  names.add(aClass.getSimpleName());
67              }
68              throw new RuntimeException(argName + " must be one of " + names.toString() + " (was " + ValueUtil.typeOf(obj).getSimpleName() + ")");
69          }
70      }
71  }

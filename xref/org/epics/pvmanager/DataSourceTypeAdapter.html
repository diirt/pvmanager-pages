9   package org.epics.pvmanager;
10  
11  /**
12   * Matches and fills a cache with the data from connection and message payloads.
13   * This optional class helps the writer of a datasource to manage the
14   * type matching and conversions.
15   *
16   * @param <ConnectionPayload> the type of payload given at connection
17   * @param <MessagePayload> the type of payload for each message
18   * @author carcassi
19   */
20  public interface DataSourceTypeAdapter<ConnectionPayload, MessagePayload> {
21      
22      /**
23       * Determines whether the converter can take values from the channel
24       * described by the connection payload and transform them in a 
25       * type required by the cache.
26       * 
27       * @param cache the cache where data will need to be written
28       * @param connection the connection information
29       * @return zero if there is no match, or the position of the type matched
30       */
31      int match(ValueCache<?> cache, ConnectionPayload connection);
32      
33      /**
34       * The parameters required to open a monitor for the channel. The
35       * type of the parameters will be datasource specific
36       * <p>
37       * For channels multiplexed on a single subscription, this method
38       * is never used.
39       * 
40       * @param cache the cache where data will need to be written
41       * @param connection the connection information
42       * @return datasource specific subscription information
43       */
44      Object getSubscriptionParameter(ValueCache<?> cache, ConnectionPayload connection);
45      
46      /**
47       * Takes the information in the message and updates the cache. 
48       * 
49       * @param cache cache to be updated
50       * @param connection the connection information
51       * @param message the payload of each message
52       * @return true if a new value was stored
53       */
54      boolean updateCache(ValueCache<?> cache, ConnectionPayload connection, MessagePayload message);
55  }

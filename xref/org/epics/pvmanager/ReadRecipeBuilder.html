5   package org.epics.pvmanager;
6   
7   import java.util.Collection;
8   import java.util.HashMap;
9   import java.util.HashSet;
10  import java.util.Map;
11  import java.util.Set;
12  
13  /**
14   * Builder class for {@link ReadRecipe}.
15   *
16   * @author carcassi
17   */
18  public class ReadRecipeBuilder {
19  
20      private final Map<String, Collection<ValueCache<?>>> channelCaches
21              = new HashMap<>();
22  
23      /**
24       * Adds a channel and its read cache to the recipe.
25       * 
26       * @param channelName the name of the channel
27       * @param cache the cache that contains the value
28       * @return this builder
29       */
30      public ReadRecipeBuilder addChannel(String channelName, ValueCache<?> cache) {
31          Collection<ValueCache<?>> cachesForChannel = channelCaches.get(channelName);
32          if (cachesForChannel == null) {
33              cachesForChannel = new HashSet<>();
34              channelCaches.put(channelName, cachesForChannel);
35          }
36          cachesForChannel.add(cache);
37          return this;
38      }
39  
40      /**
41       * Builds the recipe.
42       * <p>
43       * To finish building the recipe, one needs to specify where to send errors
44       * and where to send the connection status changes.
45       * 
46       * @param exceptionWriteFunction where exception should be routed
47       * @param connectionCollector where connection status should be routed
48       * @return a new recipe
49       */
50      public ReadRecipe build(WriteFunction<Exception> exceptionWriteFunction, ConnectionCollector connectionCollector) {
51          Set<ChannelReadRecipe> recipes = new HashSet<>();
52          for (Map.Entry<String, Collection<ValueCache<?>>> entry : channelCaches.entrySet()) {
53              String channelName = entry.getKey();
54              Collection<ValueCache<?>> valueCaches = entry.getValue();
55              for (ValueCache<?> valueCache : valueCaches) {
56                  recipes.add(new ChannelReadRecipe(channelName, 
57                         new ChannelHandlerReadSubscription(valueCache, exceptionWriteFunction, connectionCollector.addChannel(channelName))));
58             }
59          }
60          return new ReadRecipe(recipes);
61      }
62  }

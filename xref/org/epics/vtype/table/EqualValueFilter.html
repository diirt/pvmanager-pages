5   package org.epics.vtype.table;
6   
7   import java.util.List;
8   import java.util.Objects;
9   import org.epics.util.array.ListNumber;
10  import org.epics.vtype.VTable;
11  
12  /**
13   *
14   * @author carcassi
15   */
16  public class EqualValueFilter {
17      private final List<VTable> tables;
18      private final int[] columnIndexes;
19      private final Class<?> type;
20  
21      public EqualValueFilter(List<VTable> tables, int[] columnIndexes) {
22          this.tables = tables;
23          this.columnIndexes = columnIndexes;
24          Class<?> firstTableType = tables.get(0).getColumnType(columnIndexes[0]);
25          if (firstTableType.isPrimitive()) {
26              type = double.class;
27          } else if (firstTableType.equals(String.class)) {
28              type = String.class;
29          } else {
30              throw new UnsupportedOperationException("Natural join only supports numbers and Strings");
31          }
32          for (int i = 1; i < columnIndexes.length; i++) {
33              Class<?> tableType = tables.get(i).getColumnType(columnIndexes[i]);
34              if (type == double.class) {
35                  if (!tableType.isPrimitive()) {
36                      throw new IllegalArgumentException("Column types must match for natural join");
37                  }
38              }
39              if (type == String.class) {
40                  if (tableType != String.class) {
41                      throw new IllegalArgumentException("Column types must match for natural join");
42                  }
43              }
44              
45          }
46      }
47      
48      public boolean filterRow(int[] rowIndexes) {
49          if (type == double.class) {
50              double value = ((ListNumber) tables.get(0).getColumnData(columnIndexes[0])).getDouble(rowIndexes[0]);
51              for (int i = 1; i < rowIndexes.length; i++) {
52                  if (value != ((ListNumber) tables.get(i).getColumnData(columnIndexes[i])).getDouble(rowIndexes[i])) {
53                      return false;
54                  }
55              }
56          } else {
57              Object value = ((List) tables.get(0).getColumnData(columnIndexes[0])).get(rowIndexes[0]);
58              for (int i = 1; i < rowIndexes.length; i++) {
59                  if (!Objects.equals(value, ((List) tables.get(i).getColumnData(columnIndexes[i])).get(rowIndexes[i]))) {
60                      return false;
61                  }
62              }
63          }
64          return true;
65      }
66      
67  }

5   package org.epics.vtype.table;
6   
7   import java.util.List;
8   import java.util.Objects;
9   import org.epics.util.array.ListNumber;
10  import org.epics.vtype.VNumber;
11  import org.epics.vtype.VString;
12  import org.epics.vtype.VTable;
13  
14  /**
15   *
16   * @author carcassi
17   */
18  class ValueFilter {
19      private final VTable table;
20      private final int columnIndex;
21      private final Object value;
22  
23      public ValueFilter(VTable table, String columnName, Object value) {
24          this.table = table;
25          columnIndex = VTableFactory.columnNames(table).indexOf(columnName);
26          if (columnIndex == -1) {
27              throw new IllegalArgumentException("Table does not contain column '" + columnName + "'");
28          }
29          Class<?> columnType = table.getColumnType(columnIndex);
30          if (columnType.isPrimitive()) {
31              if (!(value instanceof VNumber)) {
32                  throw new IllegalArgumentException("Column '" + columnName + "' is a number but not value '" + value + "'");
33              }
34          } else if (columnType.equals(String.class)) {
35              if (!(value instanceof VString)) {
36                  throw new IllegalArgumentException("Column '" + columnName + "' is a string but not value '" + value + "'");
37              }
38          } else {
39              throw new UnsupportedOperationException("Equal value filter only works on numbers and strings");
40          }
41          this.value = value;
42      }
43      
44      public boolean filterRow(int rowIndex) {
45          if (value instanceof VNumber) {
46              double columnValue = ((ListNumber) table.getColumnData(columnIndex)).getDouble(rowIndex);
47              return columnValue == ((VNumber) value).getValue().doubleValue();
48          } else if (value instanceof VString) {
49              @SuppressWarnings("unchecked")
50              List<String> columnData = (List<String>) table.getColumnData(columnIndex);
51              return Objects.equals(columnData.get(rowIndex), ((VString) value).getValue());
52          }
53          throw new IllegalStateException("Unexpected error");
54      }
55      
56  }

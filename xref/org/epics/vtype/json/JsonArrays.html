29  /**
30   * Utility classes to convert JSON arrays to and from Lists and ListNumbers.
31   *
32   * @author carcassi
33   */
34  public class JsonArrays {
35      
36      /**
37       * Checks whether the array contains only numbers.
38       * 
39       * @param array a JSON array
40       * @return true if all elements are JSON numbers
41       */
42      public static boolean isNumericArray(JsonArray array) {
43          for (JsonValue value : array) {
44              if (!(value instanceof JsonNumber)) {
45                  return false;
46              }
47          }
48          return true;
49      }
50      
51      /**
52       * Checks whether the array contains only strings.
53       * 
54       * @param array a JSON array
55       * @return true if all elements are JSON strings
56       */
57      public static boolean isStringArray(JsonArray array) {
58          for (JsonValue value : array) {
59              if (!(value instanceof JsonString)) {
60                  return false;
61              }
62          }
63          return true;
64      }
65      
66      /**
67       * Converts the given numeric JSON array to a ListDouble.
68       * 
69       * @param array an array of numbers
70       * @return a new ListDouble
71       */
72      public static ListDouble toListDouble(JsonArray array) {
73          double[] values = new double[array.size()];
74          for (int i = 0; i < values.length; i++) {
75              if (array.isNull(i)) {
76                  values[i] = Double.NaN;
77              } else {
78                  values[i] = array.getJsonNumber(i).doubleValue();
79              }
80          }
81          return new ArrayDouble(values);
82      }
83      
84      /**
85       * Converts the given numeric JSON array to a ListFloat.
86       * 
87       * @param array an array of numbers
88       * @return a new ListFloat
89       */
90      public static ListFloat toListFloat(JsonArray array) {
91          float[] values = new float[array.size()];
92          for (int i = 0; i < values.length; i++) {
93              if (array.isNull(i)) {

158     /**
159      * Converts the given string JSON array to a List of Strings.
160      * 
161      * @param array an array of strings
162      * @return a new List of Strings
163      */
164     public static List<String> toListString(JsonArray array) {
165         List<String> strings = new ArrayList<>();
166         for (int i = 0; i < array.size(); i++) {
167             strings.add(array.getString(i));
168         }
169         return strings;
170     }
171 
172     /**
173      * Converts the given List of String to a string JSON array.
174      * 
175      * @param list a List of Strings
176      * @return an array of strings
177      */
178     public static JsonArrayBuilder fromListString(List<String> list) {
179         JsonArrayBuilder b = Json.createArrayBuilder();
180         for (String element : list) {
181             b.add(element);
182         }
183         return b;
184     }
185 
186     /**
187      * Converts the given ListNumber to a number JSON array.
188      * 
189      * @param list a list of numbers
190      * @return an array of numbers
191      */
192     public static JsonArrayBuilder fromListNumber(ListNumber list) {
193         JsonArrayBuilder b = Json.createArrayBuilder();
194         if (list instanceof ListByte || list instanceof ListShort || list instanceof ListInt) {
195             for (int i = 0; i < list.size(); i++) {
196                 b.add(list.getInt(i));
197             }
198         } else if (list instanceof ListLong) {
199             for (int i = 0; i < list.size(); i++) {
200                 b.add(list.getLong(i));
201             }
202         } else {
203             for (int i = 0; i < list.size(); i++) {
204                 double value = list.getDouble(i);
205                 if (Double.isNaN(value) || Double.isInfinite(value)) {
206                     b.addNull();
207                 } else {
208                     b.add(value);
209                 }
210             }
211         }
212         return b;
213     }

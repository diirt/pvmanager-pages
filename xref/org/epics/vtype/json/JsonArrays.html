30  /**
31   * Utility classes to convert JSON arrays to and from Lists and ListNumbers.
32   *
33   * @author carcassi
34   */
35  public class JsonArrays {
36      
37      /**
38       * Checks whether the array contains only numbers.
39       * 
40       * @param array a JSON array
41       * @return true if all elements are JSON numbers
42       */
43      public static boolean isNumericArray(JsonArray array) {
44          for (JsonValue value : array) {
45              if (!(value instanceof JsonNumber)) {
46                  return false;
47              }
48          }
49          return true;
50      }
51      
52      /**
53       * Checks whether the array contains only strings.
54       * 
55       * @param array a JSON array
56       * @return true if all elements are JSON strings
57       */
58      public static boolean isStringArray(JsonArray array) {
59          for (JsonValue value : array) {
60              if (!(value instanceof JsonString)) {
61                  return false;
62              }
63          }
64          return true;
65      }
66      
67      /**
68       * Converts the given numeric JSON array to a ListDouble.
69       * 
70       * @param array an array of numbers
71       * @return a new ListDouble
72       */
73      public static ListDouble toListDouble(JsonArray array) {
74          double[] values = new double[array.size()];
75          for (int i = 0; i < values.length; i++) {
76              if (array.isNull(i)) {
77                  values[i] = Double.NaN;
78              } else {
79                  values[i] = array.getJsonNumber(i).doubleValue();
80              }
81          }
82          return new ArrayDouble(values);
83      }

159     /**
160      * Converts the given string JSON array to a List of Strings.
161      * 
162      * @param array an array of strings
163      * @return a new List of Strings
164      */
165     public static List<String> toListString(JsonArray array) {
166         List<String> strings = new ArrayList<>();
167         for (int i = 0; i < array.size(); i++) {
168             strings.add(array.getString(i));

170         return strings;
171     }

174     /**
175      * Converts the given JSON array to a List of Timestamp.
176      * 
177      * @param array an array
178      * @return a new List of Timestamps
179      */
180     public static List<Timestamp> toListTimestamp(JsonArray array) {
181         List<Timestamp> timestamps = new ArrayList<>();
182         for (int i = 0; i < array.size(); i++) {
183             timestamps.add(Timestamp.of(array.getJsonNumber(i).longValue(), 0));
184         }
185         return timestamps;
186     }

188     /**
189      * Converts the given List of String to a string JSON array.
190      * 
191      * @param list a List of Strings
192      * @return an array of strings
193      */
194     public static JsonArrayBuilder fromListString(List<String> list) {
195         JsonArrayBuilder b = Json.createArrayBuilder();
196         for (String element : list) {
197             b.add(element);
198         }
199         return b;
200     }
201 
202     /**
203      * Converts the given List of Timestamp to a JSON array.
204      * 
205      * @param list a List of Timestamps
206      * @return an array
207      */
208     public static JsonArrayBuilder fromListTimestamp(List<Timestamp> list) {
209         JsonArrayBuilder b = Json.createArrayBuilder();
210         for (Timestamp element : list) {
211             b.add(element.getSec());
212         }
213         return b;
214     }
215 
216     /**
217      * Converts the given ListNumber to a number JSON array.
218      * 
219      * @param list a list of numbers
220      * @return an array of numbers
221      */
222     public static JsonArrayBuilder fromListNumber(ListNumber list) {
223         JsonArrayBuilder b = Json.createArrayBuilder();
224         if (list instanceof ListByte || list instanceof ListShort || list instanceof ListInt) {
225             for (int i = 0; i < list.size(); i++) {
226                 b.add(list.getInt(i));
227             }
228         } else if (list instanceof ListLong) {
229             for (int i = 0; i < list.size(); i++) {
230                 b.add(list.getLong(i));
231             }
232         } else {
233             for (int i = 0; i < list.size(); i++) {
234                 double value = list.getDouble(i);
235                 if (Double.isNaN(value) || Double.isInfinite(value)) {
236                     b.addNull();
237                 } else {
238                     b.add(value);
239                 }
240             }
241         }
242         return b;
243     }

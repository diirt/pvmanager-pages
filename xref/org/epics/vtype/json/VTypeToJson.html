5   package org.epics.vtype.json;
6   
7   import javax.json.JsonObject;
8   import org.epics.vtype.VType;
9   
10  /**
11   * Utility to serialize and de-serialize vTypes to and from JSON objects.
12   * These methods convert vTypes to and from standard JSONP objects. One
13   * can then use the standard library to serialize/de-serialize text streams.
14   * 
15   * @author carcassi
16   */
17  public class VTypeToJson {
18  
19      /**
20       * Converts the given JsonObject to a vType.
21       * 
22       * @param json a JSON object
23       * @return the corresponding vType
24       */
25      public static VType toVType(JsonObject json) {
26          return VTypeToJsonV1.toVType(json);
27      }
28      
29      /**
30       * Converts the given vType to a JsonObject.
31       * 
32       * @param vType a vType
33       * @return the corresponding JsonObject
34       */
35      public static JsonObject toJson(VType vType) {
36          return VTypeToJsonV1.toJson(vType);
37      }
38  }

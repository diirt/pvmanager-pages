5   package org.epics.vtype;
6   
7   /**
8    * Immutable VInt implementation.
9    *
10   * @author carcassi
11   */
12  class IVLong extends IVNumeric implements VLong {
13      
14      private final Long value;
15  
16      IVLong(Long value, Alarm alarm, Time time, Display display) {
17          super(alarm, time, display);
18          this.value = value;
19      }
20  
21      @Override
22      public Long getValue() {
23          return value;
24      }
25  
26      @Override
27      public String toString() {
28          return VTypeToString.toString(this);
29      }
30  
31  }

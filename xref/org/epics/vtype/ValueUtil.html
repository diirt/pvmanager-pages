43      /**
44       * Returns the type of the object by returning the class object of one
45       * of the VXxx interfaces. The getClass() methods returns the
46       * concrete implementation type, which is of little use. If no
47       * super-interface is found, Object.class is used.
48       * 
49       * @param obj an object implementing a standard type
50       * @return the type is implementing
51       */
52      public static Class<?> typeOf(Object obj) {
53          if (obj == null)
54              return null;
55  
56          return typeOf(obj.getClass());
57      }
58  
59      private static Class<?> typeOf(Class<?> clazz) {
60          if (clazz.equals(Object.class))
61              return Object.class;
62  
63          for (int i = 0; i < clazz.getInterfaces().length; i++) {
64              Class<?> interf = clazz.getInterfaces()[i];
65              if (types.contains(interf))
66                  return interf;
67          }
68  
69          return typeOf(clazz.getSuperclass());
70      }

87      /**
88       * Extracts the alarm information if present, based on value
89       * and connection status.
90       * 
91       * @param value a value
92       * @param connected the connection status
93       * @return the alarm information
94       */
95      public static Alarm alarmOf(Object value, boolean connected) {
96          if (value != null) {
97              if (value instanceof Alarm) {
98                  return (Alarm) value;
99              } else {
100                 return ValueFactory.alarmNone();
101             }
102         } else if (connected) {
103             return ValueFactory.newAlarm(AlarmSeverity.INVALID, "No value");
104         } else {
105             return ValueFactory.newAlarm(AlarmSeverity.UNDEFINED, "Disconnected");
106         }
107     }

109     /**
110      * Returns the alarm with highest severity. null values can either be ignored or
111      * treated as UNDEFINED severity.
112      * 
113      * @param args a list of values
114      * @param considerNull whether to consider null values
115      * @return the highest alarm; can't be null
116      */
117     public static Alarm highestSeverityOf(final List<Object> args, final boolean considerNull) {
118         Alarm finalAlarm = ValueFactory.alarmNone();
119         for (Object object : args) {
120             Alarm newAlarm;
121             if (object == null && considerNull) {
122                 newAlarm = ValueFactory.newAlarm(AlarmSeverity.UNDEFINED, "No Value");
123             } else {
124                 newAlarm = ValueUtil.alarmOf(object);
125                 if (newAlarm == null) {
126                     newAlarm = ValueFactory.alarmNone();
127                 }
128             }
129             if (newAlarm.getAlarmSeverity().compareTo(finalAlarm.getAlarmSeverity()) > 0) {
130                 finalAlarm = newAlarm;
131             }
132         }
133         
134         return finalAlarm;
135     }

137     /**
138      * Returns the time with latest timestamp.
139      * 
140      * @param args a list of values
141      * @return the latest time; can be null
142      */
143     public static Time latestTimeOf(final List<Object> args) {
144         Time finalTime = null;
145         for (Object object : args) {
146             Time newTime;
147             if (object != null)  {
148                 newTime = ValueUtil.timeOf(object);
149                 if (newTime != null && (finalTime == null || newTime.getTimestamp().compareTo(finalTime.getTimestamp()) > 0)) {
150                     finalTime = newTime;
151                 }
152             }
153         }
154         
155         return finalTime;
156     }

158     /**
159      * Returns the time with latest valid timestamp or now.
160      * 
161      * @param args a list of values
162      * @return the latest time; can't be null
163      */
164     public static Time latestValidTimeOrNowOf(final List<Object> args) {
165         Time finalTime = null;
166         for (Object object : args) {
167             Time newTime;
168             if (object != null)  {
169                 newTime = ValueUtil.timeOf(object);
170                 if (newTime != null && newTime.isTimeValid() && (finalTime == null || newTime.getTimestamp().compareTo(finalTime.getTimestamp()) > 0)) {
171                     finalTime = newTime;
172                 }
173             }
174         }
175         
176         if (finalTime == null) {
177             finalTime = ValueFactory.timeNow();
178         }
179         
180         return finalTime;
181     }

196      * Extracts the display information if present.
197      *
198      * @param obj an object implementing a standard type
199      * @return the display information for the object
200      */
201     public static Display displayOf(Object obj) {
202         if (obj instanceof VBoolean) {
203             return ValueFactory.displayBoolean();
204         }
205         if (!(obj instanceof Display))
206             return null;
207         Display display = (Display) obj;
208         if (display.getLowerAlarmLimit() == null || display.getLowerDisplayLimit() == null)
209             return null;
210         return display;
211     }

216      * @param display a display

219     public static boolean displayHasValidDisplayLimits(Display display) {

224             return false;

226         return true;

241     /**
242      * Normalizes the given value according to the given display information.
243      *
244      * @param value a value
245      * @param display the display information
246      * @return the normalized value, or null of either value or display is null
247      */
248     public static Double normalize(Number value, Display display) {
249         if (value == null || display == null) {
250             return null;
251         }
252 
253         return (value.doubleValue() - display.getLowerDisplayLimit()) / (display.getUpperDisplayLimit() - display.getLowerDisplayLimit());
254     }
255 
256     /**
257      * Normalizes the given value according to the given range;
258      *
259      * @param value a value
260      * @param lowValue the lowest value in the range
261      * @param highValue the highest value in the range
262      * @return the normalized value, or null if any value is null
263      */
264     public static Double normalize(Number value, Number lowValue, Number highValue) {
265         if (value == null || lowValue == null || highValue == null) {
266             return null;
267         }
268 
269         return (value.doubleValue() - lowValue.doubleValue()) / (highValue.doubleValue() - lowValue.doubleValue());
270     }

272     /**
273      * Extracts a numeric value for the object. If it's a numeric scalar,
274      * the value is returned. If it's a numeric array, the first element is
275      * returned. If it's a numeric multi array, the value of the first
276      * element is returned.
277      *
278      * @param obj an object implementing a standard type
279      * @return the numeric value
280      */
281     public static Double numericValueOf(Object obj) {
282         if (obj instanceof VNumber) {
283             Number value = ((VNumber) obj).getValue();
284             if (value != null) {
285                 return value.doubleValue();
286             }
287         }
288         
289         if (obj instanceof VBoolean) {
290             return (double) (((VBoolean) obj).getValue() ? 1 : 0);
291         }
292         
293         if (obj instanceof VEnum) {
294             return (double) ((VEnum) obj).getIndex();
295         }
296 
297         if (obj instanceof VNumberArray) {
298             ListNumber data = ((VNumberArray) obj).getData();
299             if (data != null && data.size() != 0) {
300                 return data.getDouble(0);
301             }
302         }
303         
304         if (obj instanceof VEnumArray) {
305             ListNumber data = ((VEnumArray) obj).getIndexes();
306             if (data != null && data.size() != 0) {
307                 return data.getDouble(0);
308             }
309         }
310 
311         if (obj instanceof MultiScalar) {
312             List values = ((MultiScalar) obj).getValues();
313             if (!values.isEmpty())
314                 return numericValueOf(values.get(0));
315         }
316 
317         return null;
318     }

320     /**
321      * Converts a VImage to an AWT BufferedImage, so that it can be displayed.
322      * The content of the vImage buffer is copied, so further changes
323      * to the VImage will not modify the BufferedImage.
324      *
325      * @param vImage the image to be converted
326      * @return a new BufferedImage
327      */
328     public static BufferedImage toImage(VImage vImage) {
329         BufferedImage image = new BufferedImage(vImage.getWidth(), vImage.getHeight(), BufferedImage.TYPE_3BYTE_BGR);
330         System.arraycopy(vImage.getData(), 0, ((DataBufferByte) image.getRaster().getDataBuffer()).getData(), 0,
331                 vImage.getWidth() * vImage.getHeight() * 3);
332         return image;
333     }
334 
335     /**
336      * Converts an AWT BufferedImage to a VImage.
337      * <p>
338      * Currently, only TYPE_3BYTE_BGR is supported
339      * 
340      * @param image
341      * @return a new image
342      */
343     public static VImage toVImage(BufferedImage image) {
344         if (image.getType() != BufferedImage.TYPE_3BYTE_BGR) {
345             BufferedImage newImage = new BufferedImage(image.getWidth(), image.getHeight(), BufferedImage.TYPE_3BYTE_BGR);
346             newImage.getGraphics().drawImage(image, 0, 0, null);
347             image = newImage;
348         }
349 
350         byte[] buffer = ((DataBufferByte) image.getRaster().getDataBuffer()).getData();
351         return ValueFactory.newVImage(image.getHeight(), image.getWidth(), buffer);
352     }

354     /**
355      * Returns true if the two displays contain the same information.
356      * 
357      * @param d1 the first display
358      * @param d2 the second display
359      * @return true if they match
360      */
361     public static boolean displayEquals(Display d1, Display d2) {
362         if (d1 == d2) {
363             return true;
364         }
365         
366         if (Objects.equals(d1.getFormat(), d2.getFormat()) &&
367                 Objects.equals(d1.getUnits(), d2.getUnits()) &&
368                 Objects.equals(d1.getLowerDisplayLimit(), d2.getLowerDisplayLimit()) &&
369                 Objects.equals(d1.getLowerAlarmLimit(), d2.getLowerAlarmLimit()) &&
370                 Objects.equals(d1.getLowerWarningLimit(), d2.getLowerWarningLimit()) &&
371                 Objects.equals(d1.getUpperWarningLimit(), d2.getUpperWarningLimit()) &&
372                 Objects.equals(d1.getUpperAlarmLimit(), d2.getUpperAlarmLimit()) &&
373                 Objects.equals(d1.getUpperDisplayLimit(), d2.getUpperDisplayLimit()) &&
374                 Objects.equals(d1.getLowerCtrlLimit(), d2.getLowerCtrlLimit()) &&
375                 Objects.equals(d1.getUpperCtrlLimit(), d2.getUpperCtrlLimit())) {
376             return true;
377         }
378         
379         return false;
380     }

387     private static Map<AlarmSeverity, Integer> createDefaultSeverityColorMap() {
388         Map<AlarmSeverity, Integer> colorMap = new EnumMap<>(AlarmSeverity.class);
389         colorMap.put(AlarmSeverity.NONE, 0xFF00FF00); // Color.GREEN
390         colorMap.put(AlarmSeverity.MINOR, 0xFFFFFF00); // Color.YELLOW
391         colorMap.put(AlarmSeverity.MAJOR, 0xFFFF0000); // Color.RED
392         colorMap.put(AlarmSeverity.INVALID, 0xFFFF00FF); // Color.MAGENTA
393         colorMap.put(AlarmSeverity.UNDEFINED, 0xFF404040); // Color.DARK_GRAY
394         return colorMap;
395     }

397     /**
398      * Changes the color map for AlarmSeverity. The new color map must be complete
399      * and not null;
400      * 
401      * @param map the new color map
402      */
403     public static void setAlarmSeverityColorMap(Map<AlarmSeverity, Integer> map) {
404         if (map == null) {
405             throw new IllegalArgumentException("Alarm severity color map can't be null");
406         }
407         
408         for (AlarmSeverity alarmSeverity : AlarmSeverity.values()) {
409             if (!map.containsKey(alarmSeverity)) {
410                 throw new IllegalArgumentException("Missing color for AlarmSeverity." + alarmSeverity);
411             }
412         }
413         
414         Map<AlarmSeverity, Integer> colorMap = new EnumMap<>(AlarmSeverity.class);
415         colorMap.putAll(map);
416         rgbSeverityColor = colorMap;
417     }

483     /**
484      * Extracts the values of a column, making sure it contains
485      * numeric values.
486      * 
487      * @param table a table
488      * @param columnName the name of the column to extract
489      * @return the values; null if the columnName is null or is not found
490      * @throws IllegalArgumentException if the column is found but does not contain numeric values
491      */
492     public static ListNumber numericColumnOf(VTable table, String columnName) {
493         if (columnName == null) {
494             return null;
495         }
496         
497         for (int i = 0; i < table.getColumnCount(); i++) {
498             if (columnName.equals(table.getColumnName(i))) {
499                 if (table.getColumnType(i).isPrimitive()) {
500                     return (ListNumber) table.getColumnData(i);
501                 } else {
502                     throw new IllegalArgumentException("Column '" + columnName +"' is not numeric (contains " + table.getColumnType(i).getSimpleName() + ")");
503                 }
504             }
505         }
506         
507         throw new IllegalArgumentException("Column '" + columnName +"' was not found");
508     }

510     /**
511      * Extracts the values of a column, making sure it contains
512      * strings.
513      * 
514      * @param table a table
515      * @param columnName the name of the column to extract
516      * @return the values; null if the columnName is null or is not found
517      * @throws IllegalArgumentException if the column is found but does not contain string values
518      */
519     public static List<String> stringColumnOf(VTable table, String columnName) {
520         if (columnName == null) {
521             return null;
522         }
523         
524         for (int i = 0; i < table.getColumnCount(); i++) {
525             if (columnName.equals(table.getColumnName(i))) {
526                 if (table.getColumnType(i).equals(String.class)) {
527                     @SuppressWarnings("unchecked")
528                     List<String> result = (List<String>) table.getColumnData(i);
529                     return result;
530                 } else {
531                     throw new IllegalArgumentException("Column '" + columnName +"' is not string (contains " + table.getColumnType(i).getSimpleName() + ")");
532                 }
533             }
534         }
535         
536         throw new IllegalArgumentException("Column '" + columnName +"' was not found");
537     }

565     /**
566      * Filters an element of a one-dimensional array.
567      * 
568      * @param array a 1D array
569      * @param index a valid index
570      * @return the trimmed array to that one index
571      */
572     public static VNumberArray subArray(VNumberArray array, int index) {
573         if (array.getSizes().size() != 1) {
574             throw new IllegalArgumentException("Array was not one-dimensional");
575         }
576         if (index < 0 || array.getData().size() <= index) {
577             throw new IllegalArgumentException("Index not in the array range");
578         }
579         
580         ArrayDimensionDisplay display = array.getDimensionDisplay().get(0);
581         return ValueFactory.newVNumberArray(new ArrayDouble(array.getData().getDouble(index)),
582                 new ArrayInt(1), Arrays.asList(ValueFactory.newDisplay(new ArrayDouble(display.getCellBoundaries().getDouble(index), display.getCellBoundaries().getDouble(index+1)), display.getUnits())),
583                 array, array, array);

5   package org.epics.vtype;
6   
7   import java.util.ArrayList;
8   import java.util.List;
9   import org.epics.util.array.ListInt;
10  
11  /**
12   *
13   * @author carcassi
14   */
15  class IVEnumArray extends IVMetadata implements VEnumArray {
16      
17      private final ListInt indexes;
18      private final List<String> labels;
19      private final ListInt sizes;
20      private final List<String> array;
21  
22      public IVEnumArray(ListInt indexes, List<String> labels, ListInt sizes, Alarm alarm, Time time) {
23          super(alarm, time);
24          List<String> tempArray = new ArrayList<>(indexes.size());
25          for (int i = 0; i < indexes.size(); i++) {
26              int index = indexes.getInt(i);
27              if (index < 0 || index >= labels.size()) {
28                  throw new IndexOutOfBoundsException("VEnumArray indexes must be within the label range");
29              }
30              tempArray.add(labels.get(index));
31          }
32          this.array = tempArray;
33          this.indexes = indexes;
34          this.labels = labels;
35          this.sizes = sizes;
36      }
37  
38      @Override
39      public List<String> getLabels() {
40          return labels;
41      }
42  
43      @Override
44      public List<String> getData() {
45          return array;
46      }
47  
48      @Override
49      public ListInt getIndexes() {
50          return indexes;
51      }
52  
53      @Override
54      public ListInt getSizes() {
55          return sizes;
56      }
57  
58      @Override
59      public String toString() {
60          return VTypeToString.toString(this);
61      }
62      
63  }

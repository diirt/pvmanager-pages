5   package org.epics.vtype;
6   
7   import java.util.List;
8   
9   /**
10   * Extended alarm information. This supplements the alarm information
11   * given with scalar and array pvs, by showing all the possible alarm
12   * conditions specific to a device, and which ones are active
13   * at the moment.
14   * <p>
15   * The data type on the wire will look significantly different. The name
16   * and severity of each conditions don't typically change and are considered
17   * metadata. The actual alarm state (which ones are on or off) and the error
18   * message are what changes at each notification.
19   *
20   * @author carcassi
21   */
22  public interface VExtendedAlarm extends VType {
23      
24      /**
25       * A single alarm condition that can be on or off.
26       */
27      public interface AlarmCondition extends Alarm {
28          
29          /**
30           * Whether the alarm is on.
31           * 
32           * @return true if alarm condition is on
33           */
34          public boolean isActive();
35      }
36      
37      /**
38       * All the possible alarm conditions for the device, and
39       * whether they are on or off.
40       * 
41       * @return the alarm conditions
42       */
43      public List<AlarmCondition> getConditions();
44      
45      
46      /**
47       * A human readable error message associated to the error(s).
48       * 
49       * @return a message or null
50       */
51      public String getMessage();
52  }

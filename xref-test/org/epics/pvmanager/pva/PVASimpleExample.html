5   package org.epics.pvmanager.pva;
6   
7   import static org.epics.pvmanager.ExpressionLanguage.channel;
8   
9   import org.epics.pvmanager.PVManager;
10  import org.epics.pvmanager.PVReader;
11  import org.epics.pvmanager.PVReaderEvent;
12  import org.epics.pvmanager.PVReaderListener;
13  import org.epics.util.time.TimeDuration;
14  
15  public class PVASimpleExample {
16  	
17  	public static void main(String[] args) throws InterruptedException
18  	{
19  		PVManager.setDefaultDataSource(new PVADataSource());
20  		PVReader<Object> reader = PVManager.read(channel("testCounter")).
21  				readListener(new PVReaderListener<Object>() {
22  
23  					@Override
24  					public void pvChanged(PVReaderEvent<Object> event) {
25  						if (event.isValueChanged())
26  							System.out.println(event.getPvReader().getValue());
27  						else
28  							System.out.println(event.toString());
29  						
30  					}
31  				}).maxRate(TimeDuration.ofHertz(10));
32  		
33  		// forever
34  		while (System.currentTimeMillis() != 0)
35  			Thread.sleep(Long.MAX_VALUE);
36  		
37  		reader.close();
38  	}
39  
40  }

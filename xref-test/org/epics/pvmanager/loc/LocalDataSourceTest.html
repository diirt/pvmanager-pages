119     //@Test
120     public void fullSyncPipeline() throws Exception {
121         LocalDataSource dataSource = new LocalDataSource();
122         PVReader<Object> pv = PVManager.read(channel(channelName1)).from(dataSource).maxRate(ofHertz(100));
123         PVWriter<Object> writer = PVManager.write(channel(channelName1)).from(dataSource).sync();
124         writer.addPVWriterListener(listener);
125         writer.write(10);
126         
127         verify(listener).pvChanged(null);
128         Thread.sleep(50);
129         pv.close();
130         writer.close();
131         
132         assertThat(((VDouble) pv.getValue()).getValue(), equalTo(10.0));
133     }

137         LocalDataSource dataSource1 = new LocalDataSource();

140         compositeSource.putDataSource("loc1", dataSource1);
141         compositeSource.putDataSource("loc2", dataSource2);
142         PVReader<Object> pv1 = PVManager.read(channel("loc1://test")).from(compositeSource).maxRate(ofHertz(100));
143         PVReader<Object> pv2 = PVManager.read(channel("loc2://test")).from(compositeSource).maxRate(ofHertz(100));
144         PVWriter<Object> writer = PVManager.write(channel("loc1://test")).from(compositeSource).sync();

146         writer.write(10);
147         
148         verify(listener).pvChanged(null);
149         Thread.sleep(50);
150         pv1.close();
151         writer.close();
152         
153         assertThat(((VDouble) pv1.getValue()).getValue(), equalTo(10.0));

156     //@Test
157     public void fullAsyncPipeline() throws Exception {
158         LocalDataSource dataSource = new LocalDataSource();
159         PVReader<Object> pv = PVManager.read(channel(channelName1)).from(dataSource).maxRate(ofHertz(100));
160         PVWriter<Object> writer = PVManager.write(channel(channelName1)).from(dataSource).async();
161         writer.addPVWriterListener(listener);
162         writer.write(10);
163         // On some machines, the local write is fast enough that it may already
164         // be done
165         verify(listener, atMost(1)).pvChanged(null);
166         
167         Thread.sleep(50);
168         pv.close();
169         writer.close();
170         
171         verify(listener).pvChanged(null);
172         assertThat(((VDouble) pv.getValue()).getValue(), equalTo(10.0));
173     }

175     @Test
176     public void writeMultipleNamesForSameChannel() throws Exception {
177         ReadExpressionTester exp = new ReadExpressionTester(mapOf(latestValueOf(channels("foo", "foo(2.0)"))));
178         ReadRecipe recipe = exp.getReadRecipe();
179         
180         // Connect and make sure only one channel is created
181         LocalDataSource dataSource = new LocalDataSource();
182         dataSource.connectRead(recipe);
183         assertThat(dataSource.getChannels().size(), equalTo(1));
184         dataSource.disconnectRead(recipe);
185         dataSource.close();
186     }

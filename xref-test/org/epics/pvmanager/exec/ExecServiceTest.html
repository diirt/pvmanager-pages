7   import java.util.HashMap;
8   import java.util.Map;
9   import java.util.concurrent.ExecutorService;
10  import java.util.concurrent.Executors;
11  import org.epics.pvmanager.service.Service;
12  import org.epics.pvmanager.service.ServiceUtil;
13  import org.epics.vtype.VString;
14  import org.epics.vtype.ValueFactory;
15  import org.junit.Test;
16  import static org.junit.Assert.*;
17  import static org.hamcrest.Matchers.*;
18  
19  /**
20   *
21   * @author carcassi
22   */
23  public class ExecServiceTest {
24  
25      @Test
26      public void runCommand1() {
27          ExecutorService executor = Executors.newSingleThreadExecutor();
28          Service service = new ExecService(new ExecServiceDescription("execSample", "A simple exec service")
29                  .executorService(executor)
30                  .addServiceMethod(new ExecServiceMethodDescription("echo", "A simple command")
31                                   .command("echo This is a test!")));
32          Map<String, Object> params = new HashMap<>();
33          Map<String, Object> result = ServiceUtil.syncExecuteMethod(service.getServiceMethods().get("echo"), params);
34          VString output = (VString) result.get("output");
35          assertThat(output.getValue(), equalTo("This is a test!\n"));
36      }
37  
38      @Test
39      public void runCommand2() {
40          ExecutorService executor = Executors.newSingleThreadExecutor();
41          Service service = new ExecService(new ExecServiceDescription("execSample", "A simple exec service")
42                  .executorService(executor)
43                  .addServiceMethod(new ExecServiceMethodDescription("echo", "A simple command")
44                                   .command("echo You entered #param#")
45                                   .addArgument("param", "The parameter", VString.class)));
46          Map<String, Object> params = new HashMap<>();
47          params.put("param", ValueFactory.newVString("FOO!", ValueFactory.alarmNone(), ValueFactory.timeNow()));
48          Map<String, Object> result = ServiceUtil.syncExecuteMethod(service.getServiceMethods().get("echo"), params);
49          VString output = (VString) result.get("output");
50          assertThat(output.getValue(), equalTo("You entered FOO!\n"));
51      }

53  }

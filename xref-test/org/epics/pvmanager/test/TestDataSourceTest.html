36      public static void waitForChannelToClose(DataSource source, String channelName) {
37          TimeDuration timeout = ofMillis(5000);
38          TimeInterval timeoutInterval = timeout.after(Timestamp.now());
39          while (timeoutInterval.contains(Timestamp.now())) {
40              if (source.getChannels().get(channelName) == null || !source.getChannels().get(channelName).isConnected()) {
41                  return;
42              }
43              try {
44                  Thread.sleep(100);
45              } catch(Exception ex) {
46                  
47              }
48          }
49          fail("Channel " + channelName + " didn't close after 5 seconds");
50      }

65      public void closePVsAndDataSource() {
66          if (pv != null) {
67              pv.close();
68              pv = null;
69          }
70          if (pvReader != null) {
71              pvReader.close();
72              pvReader = null;
73          }
74          if (pvReader2 != null) {
75              pvReader2.close();
76              pvReader2 = null;
77          }
78          if (pvWriter != null) {
79              pvWriter.close();
80              pvWriter = null;
81          }
82  
83          waitForChannelToClose(dataSource, "delayedWrite");
84          waitForChannelToClose(dataSource, "delayedConnection");
85          dataSource.close();
86          dataSource = null;
87      }

192     @Test
193     public void delayedWriteWithTimeout2() throws Exception {
194         // Test a write that happens 2 seconds late
195         // Checks whether we get a timeout beforehand
196         CountDownPVWriterListener<Object> writerListener = new CountDownPVWriterListener<>(1);
197         pvWriter = PVManager.write(channel("delayedWrite")).timeout(ofMillis(500))
198                 .writeListener(writerListener)
199                 .from(dataSource).async();
200         pvWriter.write("test");
201 
202         // Wait for the first notification, should be the timeout
203         writerListener.await(TimeDuration.ofMillis(600));
204         assertThat(writerListener.getCount(), equalTo(0));
205         writerListener.resetCount(1);
206         
207         Exception ex = pvWriter.lastWriteException(); 
208         assertThat(ex, not(nullValue()));
209         assertThat(ex, instanceOf(TimeoutException.class));
210 
211         // Wait for the second notification, should be
212         // the success notification
213         writerListener.await(TimeDuration.ofMillis(2000));
214         assertThat(writerListener.getCount(), equalTo(0));
215         writerListener.resetCount(1);
216         
217         ex = pvWriter.lastWriteException();
218         assertThat(ex, nullValue());
219         
220         // Write again
221         pvWriter.write("test2");
222         
223         // Wait for a notification: should not come
224         writerListener.await(TimeDuration.ofMillis(400));
225         assertThat(writerListener.getCount(), equalTo(1));
226         ex = pvWriter.lastWriteException();
227         assertThat(ex, nullValue());
228         
229         writerListener.await(TimeDuration.ofMillis(250));
230         assertThat(writerListener.getCount(), equalTo(0));
231         writerListener.resetCount(1);
232         ex = pvWriter.lastWriteException();
233         assertThat(ex, not(nullValue()));
234         assertThat(ex, instanceOf(TimeoutException.class));
235         
236         writerListener.await(TimeDuration.ofMillis(2000));
237         assertThat(writerListener.getCount(), equalTo(0));
238         ex = pvWriter.lastWriteException();
239         assertThat(ex, nullValue());
240     }

300     @Test
301     public void delayedReadOnPVWithTimeoutAndCustomMessage() throws Exception {
302         String message = "Ouch! Timeout!";
303         CountDownPVReaderListener readListener = new CountDownPVReaderListener(1);
304         pv = PVManager.readAndWrite(channel("delayedConnection"))
305                 .readListener(readListener)
306                 .timeout(ofMillis(500), message)
307                 .from(dataSource)
308                 .asynchWriteAndMaxReadRate(ofMillis(50));
309         
310         readListener.await(TimeDuration.ofMillis(50));
311         assertThat(readListener.getCount(), equalTo(1));
312         
313         TimeoutException ex = (TimeoutException) pv.lastException();
314         assertThat(ex, nullValue());
315         
316         readListener.await(TimeDuration.ofMillis(600));
317         assertThat(readListener.getCount(), equalTo(0));
318         readListener.resetCount(2);
319         
320         ex = (TimeoutException) pv.lastException();
321         assertThat(ex, not(nullValue()));
322         assertThat(ex.getMessage(), equalTo(message));
323         
324         readListener.await(TimeDuration.ofMillis(1000));
325         // It may get CONNECTION and VALUE event separate
326         assertThat(readListener.getCount(), lessThanOrEqualTo(1));
327         
328         ex = (TimeoutException) pv.lastException();
329         assertThat(ex, nullValue());
330         assertThat((String) pv.getValue(), equalTo("Initial value"));
331     }

333     @Test
334     public void delayedMultipleReadWithConnectionError() throws Exception {
335         CountDownPVReaderListener readListener1 = new CountDownPVReaderListener(1);
336         CountDownPVReaderListener readListener2 = new CountDownPVReaderListener(1);
337         pvReader = PVManager.read(channel("delayedConnectionError"))
338                 .readListener(readListener1)
339                 .from(dataSource).maxRate(ofMillis(50));
340         pvReader2 = PVManager.read(channel("delayedConnectionError"))
341                 .readListener(readListener2)
342                 .from(dataSource).maxRate(ofMillis(50));
343         
344         Thread.sleep(50);
345         
346         RuntimeException ex = (RuntimeException) pvReader.lastException();
347         assertThat(ex, nullValue());
348         assertThat(readListener1.getCount(), equalTo(1));
349         ex = (RuntimeException) pvReader2.lastException();
350         assertThat(ex, nullValue());
351         assertThat(readListener2.getCount(), equalTo(1));
352         
353         readListener1.await(TimeDuration.ofMillis(1500));
354         readListener1.resetCount(1);
355         readListener2.await(TimeDuration.ofMillis(1500));
356         readListener2.resetCount(1);
357         
358         ex = (RuntimeException) pvReader.lastException();
359         assertThat(ex, instanceOf(RuntimeException.class));
360         assertThat(ex.getMessage(), equalTo("Connection error"));
361         ex = (RuntimeException) pvReader2.lastException();
362         assertThat(ex, instanceOf(RuntimeException.class));
363         assertThat(ex.getMessage(), equalTo("Connection error"));
364     }

5   package org.epics.pvmanager.service;
6   
7   import org.epics.pvmanager.service.ServiceMethod;
8   import org.epics.pvmanager.service.ServiceMethodDescription;
9   import java.util.HashMap;
10  import java.util.Map;
11  import org.epics.pvmanager.WriteFunction;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  public class MultiplyServiceMethod extends ServiceMethod {
18  
19      public MultiplyServiceMethod() {
20          super(new ServiceMethodDescription("multiply", "Multiplies two numbers.")
21                  .addArgument("arg1", "First argument", Number.class)
22                  .addArgument("arg2", "Second argument", Number.class)
23                  .addResult("result", "The product of arg1 and arg2", Number.class));
24      }
25  
26      @Override
27      public void executeMethod(Map<String, Object> parameters, WriteFunction<Map<String, Object>> callback, WriteFunction<Exception> errorCallback) {
28          Number arg1 = (Number) parameters.get("arg1");
29          Number arg2 = (Number) parameters.get("arg2");
30          Number result = arg1.doubleValue() * arg2.doubleValue();
31          Map<String, Object> resultMap = new HashMap<>();
32          resultMap.put("result", result);
33          callback.writeValue(resultMap);
34      }
35  }

36      @Test
37      public void channel1() {
38          ChannelExpression<Object, Object> exp = channel("my pv");
39          assertThat(exp.getFunction(), instanceOf(ValueCache.class));
40          assertThat(exp.getName(), equalTo("my pv"));
41          ValueCache<Object> cache = (ValueCache<Object>) exp.getFunction();
42          assertThat(cache.getType(), equalTo(Object.class));
43          assertThat(cache.readValue(), nullValue());
44          WriteCache<Object> writeCache = (WriteCache<Object>) exp.getWriteFunction();
45          assertThat(writeCache.getPrecedingChannels().isEmpty(), equalTo(true));
46          assertThat(writeCache.getValue(), nullValue());
47          assertThat(writeCache.getChannelName(), equalTo("my pv"));
48      }

50      @Test
51      public void queue1() {
52          Queue<String> queue = queueOf(String.class).maxSize(5);
53          ReadExpressionTester exp = new ReadExpressionTester(queue);
54          assertThat(exp.getReadRecipe().getChannelReadRecipes().isEmpty(), equalTo(true));
55          assertThat(exp.getValue(), equalTo((Object) Collections.EMPTY_LIST));
56          queue.add("one");
57          queue.add("two");
58          assertThat(exp.getValue(), equalTo((Object) Arrays.asList("one", "two")));
59          queue.add("one");
60          queue.add("two");
61          queue.add("three");
62          queue.add("four");
63          queue.add("five");
64          queue.add("six");
65          assertThat(exp.getValue(), equalTo((Object) Arrays.asList("two", "three", "four", "five", "six")));
66          assertThat(exp.getValue(), equalTo((Object) Collections.EMPTY_LIST));
67      }

93      @Test
94      public void mapOf1() {
95          // Dynamically adding constant expressions (i.e. that don't require connection)
96          ReadMap<String> map = readMapOf(String.class);
97          ReadExpressionTester exp = new ReadExpressionTester(map);
98          Map<String, String> referenceValue = new HashMap<String, String>();
99          assertThat(exp.getValue(), equalTo((Object) referenceValue));
100         map.add(constant("Gabriele").as("name"));
101         referenceValue.put("name", "Gabriele");
102         assertThat(exp.getValue(), equalTo((Object) referenceValue));
103         map.add(constant("Carcassi").as("surname"));
104         referenceValue.put("surname", "Carcassi");
105         assertThat(exp.getValue(), equalTo((Object) referenceValue));
106         assertThat(exp.getValue(), sameInstance(exp.getValue()));
107         map.remove("name");
108         referenceValue.remove("name");
109         assertThat(exp.getValue(), equalTo((Object) referenceValue));
110     }

145     @Test
146     public void mapOf5() throws Exception {
147         ReadWriteMap<Double, Double> map = mapOf(latestValueOf(channels(Arrays.asList("SETPOINT", "READBACK"), Double.class, Double.class)));
148         ReadExpressionTester readExp = new ReadExpressionTester(map);
149         WriteExpressionTester writeExp = new WriteExpressionTester(map);
150         
151         Map<String, Double> referenceValue = new HashMap<String, Double>();
152         referenceValue.put("READBACK", 2.0);
153         referenceValue.put("SETPOINT", 5.0);
154         readExp.writeValue("READBACK", 2.0);
155         readExp.writeValue("SETPOINT", 5.0);
156         assertThat(readExp.getValue(), equalTo((Object) referenceValue));
157         
158         Map<String, Double> value = new HashMap<String, Double>();
159         value.put("SETPOINT", 1.0);
160         value.put("READBACK", 3.0);
161         writeExp.setValue(value);
162         assertThat(writeExp.readValue("SETPOINT"), equalTo((Object) 1.0));
163         assertThat(writeExp.readValue("READBACK"), equalTo((Object) 3.0));
164     }

27      @Test
28      public void point2DDatsetFromVTable1() {
29          VTable data = ValueFactory.newVTable(Arrays.<Class<?>>asList(double.class, double.class),
30                  Arrays.asList("x", "y"), Arrays.<Object>asList(new ArrayDouble(1,2,3), new ArrayDouble(5,4,6)));
31          Point2DDataset dataset = DatasetConversions.point2DDatasetFromVTable(data, "x", "y");
32          assertThat(dataset.getCount(), equalTo(3));
33          assertThat(dataset.getXValues(), equalTo((ListNumber) new ArrayDouble(1,2,3)));
34          assertThat(dataset.getYValues(), equalTo((ListNumber) new ArrayDouble(5,4,6)));
35      }

124         Point2DDataset dataset = datasets.get(0);
125         assertThat(dataset.getCount(), equalTo(3));
126         assertThat(dataset.getXValues(), equalTo((ListNumber) new ArrayDouble(1,2,3)));
127         assertThat(dataset.getYValues(), equalTo((ListNumber) new ArrayDouble(5,4,6)));
128         dataset = datasets.get(1);
129         assertThat(dataset.getCount(), equalTo(3));
130         assertThat(dataset.getXValues(), equalTo((ListNumber) new ArrayDouble(10,20,30)));
131         assertThat(dataset.getYValues(), equalTo((ListNumber) new ArrayDouble(50,40,60)));
132     }

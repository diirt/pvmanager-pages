25      @Test
26      public void multipleRead1() throws Exception {
27          String channelName = "ramp(0,100,1,0.01)";
28          DataSource dataSource = new SimulationDataSource();
29          
30          PVReader<Object> pv1 = PVManager.read(channel(channelName)).from(dataSource).maxRate(ofHertz(50));
31          PVReader<Object> pv2 = PVManager.read(channel(channelName)).from(dataSource).maxRate(ofHertz(50));
32          waitForValue(pv1, ofMillis(100));
33          waitForValue(pv2, ofMillis(100));
34  
35          assertThat(pv1.getValue(), not(nullValue()));
36          assertThat(pv2.getValue(), not(nullValue()));
37          assertThat(pv1.isConnected(), equalTo(true));
38          assertThat(pv2.isConnected(), equalTo(true));
39          pv1.close();
40          pv2.close();
41      }

5   package org.epics.pvmanager.sim;
6   
7   import org.epics.vtype.AlarmSeverity;
8   import org.epics.vtype.VDouble;
9   import org.junit.Test;
10  import static org.junit.Assert.*;
11  import static org.hamcrest.CoreMatchers.*;
12  
13  /**
14   *
15   * @author carcassi
16   */
17  public class SineTest {
18  
19      @Test
20      public void values() {
21          // Creates the function
22          Sine sine = new Sine(0.0, 10.0, 4.0, 1.0);
23          VDouble firstValue = sine.nextValue();
24  
25          // Check limits
26          assertEquals(5.0, firstValue.getValue(), 0.0001);
27          assertThat(firstValue.getAlarmSeverity(), equalTo(AlarmSeverity.NONE));
28          assertThat(firstValue.getAlarmName(), equalTo("NONE"));
29          assertThat(firstValue.getLowerCtrlLimit(), equalTo(0.0));
30          assertThat(firstValue.getLowerDisplayLimit(), equalTo(0.0));
31          assertThat(firstValue.getLowerAlarmLimit(), equalTo(1.0));
32          assertThat(firstValue.getLowerWarningLimit(), equalTo(2.0));
33          assertThat(firstValue.getUpperWarningLimit(), equalTo(8.0));
34          assertThat(firstValue.getUpperAlarmLimit(), equalTo(9.0));
35          assertThat(firstValue.getUpperDisplayLimit(), equalTo(10.0));
36          assertThat(firstValue.getUpperCtrlLimit(), equalTo(10.0));
37  
38          // Check values
39          assertEquals(10.0, sine.nextValue().getValue(), 0.0001);
40          assertEquals(5.0, sine.nextValue().getValue(), 0.0001);
41          assertEquals(0.0, sine.nextValue().getValue(), 0.0001);
42          assertEquals(5.0, sine.nextValue().getValue(), 0.0001);
43          assertEquals(10.0, sine.nextValue().getValue(), 0.0001);
44          assertEquals(5.0, sine.nextValue().getValue(), 0.0001);
45          assertEquals(0.0, sine.nextValue().getValue(), 0.0001);
46          assertEquals(5.0, sine.nextValue().getValue(), 0.0001);
47          assertThat(sine.nextValue().getAlarmSeverity(), equalTo(AlarmSeverity.MAJOR));
48          assertThat(sine.nextValue().getAlarmSeverity(), equalTo(AlarmSeverity.NONE));
49          assertThat(sine.nextValue().getAlarmSeverity(), equalTo(AlarmSeverity.MAJOR));
50          assertThat(sine.nextValue().getAlarmSeverity(), equalTo(AlarmSeverity.NONE));
51      }
52  }

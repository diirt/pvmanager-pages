5   package org.epics.pvmanager.sim;
6   
7   import org.epics.vtype.VDouble;
8   import org.junit.Test;
9   import static org.junit.Assert.*;
10  import static org.hamcrest.CoreMatchers.*;
11  
12  /**
13   * Tests gaussian sim function
14   *
15   * @author carcassi
16   */
17  public class GaussianNoiseTest {
18  
19      /**
20       * Tests values follow the right distribution
21       */
22      @Test
23      public void values() {
24          // Creates the function
25          GaussianNoise gaussian = new GaussianNoise(10.0, 10.0, 1.0);
26          VDouble firstValue = gaussian.nextValue();
27  
28          // Check limits
29          assertThat(firstValue.getLowerCtrlLimit(), equalTo(-30.0));
30          assertThat(firstValue.getLowerDisplayLimit(), equalTo(-30.0));
31          assertThat(firstValue.getLowerAlarmLimit(), equalTo(-10.0));
32          assertThat(firstValue.getLowerWarningLimit(), equalTo(0.0));
33          assertThat(firstValue.getUpperWarningLimit(), equalTo(20.0));
34          assertThat(firstValue.getUpperAlarmLimit(), equalTo(30.0));
35          assertThat(firstValue.getUpperDisplayLimit(), equalTo(50.0));
36          assertThat(firstValue.getUpperCtrlLimit(), equalTo(50.0));
37  
38          // Calculate histogram
39          int quart1 = 0;
40          int quart2 = 0;
41          int quart3 = 0;
42          int quart4 = 0;
43  
44          for (int i = 0; i < 100000; i++) {
45              double value = gaussian.nextValue().getValue();
46              if (value < 10.0) {
47                  if (value < 0.0) {
48                      quart1++;
49                  } else {
50                      quart2++;
51                  }
52              } else {
53                  if (value < 20.0) {
54                      quart3++;
55                  } else {
56                      quart4++;
57                  }
58              }
59          }
60  
61          // Check distribution
62          // Theoretical distribution 15.87% 34.13% 34.13% 15.87%
63          assertTrue(quart1 < 16870);
64          assertTrue(quart2 < 35130);
65          assertTrue(quart3 < 35130);
66          assertTrue(quart4 < 16870);
67          assertEquals(100000, quart1+quart2+quart3+quart4);
68      }
69  }

31      @Test
32      public void columnOf() {
33          VTable data = newVTable(Arrays.<Class<?>>asList(String.class, double.class, double.class),
34                  Arrays.asList("x", "y", "z"), Arrays.<Object>asList(Arrays.asList("a", "b", "c"), new ArrayDouble(1,2,3), new ArrayDouble(5,4,6)));
35          VStringArray expected1 = newVStringArray(Arrays.asList("a", "b", "c"), alarmNone(), timeNow());
36          VNumberArray expected2 = newVDoubleArray(new ArrayDouble(1,2,3), alarmNone(), timeNow(), displayNone());
37          FunctionTester.findByName(set, "columnOf")
38                  .compareReturnValue(expected1, data, "x")
39                  .compareReturnValue(expected2, data, "y")
40                  .compareReturnValue(null, data, null)
41                  .compareReturnValue(null, null, "y");
42      }

82      @Test
83      public void tableRangeFilter1() {
84          VTable table = newVTable(column("Rack", newVStringArray(Arrays.asList("A", "A", "B"), alarmNone(), timeNow())),
85                                   column("Slot", newVDoubleArray(new ArrayDouble(1,2,3), alarmNone(), timeNow(), displayNone())),
86                                   column("CPU", newVStringArray(Arrays.asList("286", "286", "386"), alarmNone(), timeNow())));
87          VTable expected = newVTable(column("Rack", newVStringArray(Arrays.asList("A", "A"), alarmNone(), timeNow())),
88                                   column("Slot", newVDoubleArray(new ArrayDouble(1,2), alarmNone(), timeNow(), displayNone())),
89                                   column("CPU", newVStringArray(Arrays.asList("286", "286"), alarmNone(), timeNow())));
90  
91          FunctionTester.findBySignature(set, "tableRangeFilter", VTable.class, VString.class, VType.class, VType.class)
92                  .compareReturnValue(expected, table, "Slot", 1.0, 2.5)
93                  .compareReturnValue(null, null, "Slot", 1.0, 2.5)
94                  .compareReturnValue(null, table, null, 1.0, 2.5)
95                  .compareReturnValue(null, table, "Slot", null, 2.5)
96                  .compareReturnValue(null, table, "Slot", 1.0, null);
97      }

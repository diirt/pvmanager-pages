27      @Test
28      public void testCast1() {
29          LastOfChannelExpression<Object> exp = new LastOfChannelExpression<Object>("test", Object.class);
30          LastOfChannelExpression<Object> exp2 = exp.cast(Object.class);
31          assertThat(exp2, sameInstance(exp));
32      }
33  
34      @Test
35      public void testCast2() {
36          LastOfChannelExpression<Object> exp = new LastOfChannelExpression<Object>("test", Object.class);
37          LastOfChannelExpression<VDouble> vDoubleExp = exp.cast(VDouble.class);
38          assertThat(vDoubleExp.getType(), equalTo(VDouble.class));
39          assertThat(vDoubleExp.getName(), equalTo("test"));
40      }
41  
42      @Test
43      public void testCast3() {
44          LastOfChannelExpression<VDouble> exp = new LastOfChannelExpression<VDouble>("test", VDouble.class);
45          LastOfChannelExpression<?> exp2 = exp.cast(Object.class);
46          assertThat(exp2 == exp, equalTo(true));
47          assertThat(exp2.getName(), equalTo("test"));
48      }

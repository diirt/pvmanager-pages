36  public class JDBCXMLServiceFactoryTest {

38      @Test
39      public void new1() throws Exception {
40          File file = new File(getClass().getResource(".").toURI());
41          JDBCXMLServiceFactory factory = new JDBCXMLServiceFactory(file);
42          Collection<Service> services = factory.createServices();
43          assertThat(services.size(), equalTo(1));
44          Service service = services.iterator().next();
45          assertThat(service.getName(), equalTo("jdbcSample"));
46          assertThat(service.getDescription(), equalTo("A test service"));
47          assertThat(service.getServiceMethods().size(), equalTo(2));
48          assertThat(service.getServiceMethods().get("query").getName(), equalTo("query"));
49          assertThat(service.getServiceMethods().get("query").getDescription(), equalTo("A test query"));
50          assertThat(service.getServiceMethods().get("query").getArgumentDescriptions().size(), equalTo(0));
51          assertThat(service.getServiceMethods().get("query").getResultTypes().get("result"), equalTo((Class) VTable.class));
52          assertThat(service.getServiceMethods().get("query").getResultDescriptions().get("result"), equalTo("The query result"));
53          assertThat(service.getServiceMethods().get("insert").getName(), equalTo("insert"));
54          assertThat(service.getServiceMethods().get("insert").getDescription(), equalTo("A test insert query"));
55          assertThat(service.getServiceMethods().get("insert").getResultTypes().size(), equalTo(0));
56          assertThat(service.getServiceMethods().get("insert").getResultDescriptions().size(), equalTo(0));
57          assertThat(service.getServiceMethods().get("insert").getArgumentDescriptions().get("name"), equalTo("The name"));
58          assertThat(service.getServiceMethods().get("insert").getArgumentTypes().get("name"), equalTo((Class) VString.class));
59          assertThat(service.getServiceMethods().get("insert").getArgumentDescriptions().get("index"), equalTo("The index"));
60          assertThat(service.getServiceMethods().get("insert").getArgumentTypes().get("index"), equalTo((Class) VNumber.class));
61          assertThat(service.getServiceMethods().get("insert").getArgumentDescriptions().get("value"), equalTo("The value"));
62          assertThat(service.getServiceMethods().get("insert").getArgumentTypes().get("value"), equalTo((Class) VNumber.class));
63      }

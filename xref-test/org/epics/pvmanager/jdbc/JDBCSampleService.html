5   package org.epics.pvmanager.jdbc;
6   
7   import java.util.concurrent.Executors;
8   import org.epics.vtype.VNumber;
9   import org.epics.vtype.VString;
10  
11  /**
12   *
13   * @author carcassi
14   */
15  public class JDBCSampleService extends JDBCService {
16  
17      public JDBCSampleService() {
18          super(new JDBCServiceDescription("jdbcSample", "A test service")
19                  .dataSource(new SimpleDataSource("jdbc:mysql://localhost/test?user=root&password=root"))
20                  .executorService(Executors.newSingleThreadExecutor(org.epics.pvmanager.util.Executors.namedPool("jdbcSample")))
21                  .addServiceMethod(new JDBCServiceMethodDescription("query", "A test query")
22                      .query("SELECT * FROM Data")
23                      .queryResult("result", "The query result")
24                  )
25                  .addServiceMethod(new JDBCServiceMethodDescription("insert", "A test insertquery")
26                      .query("INSERT INTO `test`.`Data` (`Name`, `Index`, `Value`, `Time`) VALUES (?, ?, ?, now())")
27                      .addArgument("name", "The name", VString.class)
28                      .addArgument("index", "The index", VNumber.class)
29                      .addArgument("value", "The value", VNumber.class)
30                  ));
31      }
32      
33  }

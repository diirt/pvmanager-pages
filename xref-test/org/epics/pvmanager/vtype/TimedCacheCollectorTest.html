27  public class TimedCacheCollectorTest {
28  
29      public TimedCacheCollectorTest() {
30      }
31  
32      @BeforeClass
33      public static void setUpClass() throws Exception {
34          // Force type support loading
35          DataTypeSupport.install();
36      }
37  
38      @Test
39      public void correctNumberOfValuesInCache() throws InterruptedException {
40          ReadExpressionTester exp = new ReadExpressionTester(timedCacheOf(vDouble("x"), TimeDuration.ofMillis(100)));
41          
42          Timestamp reference = Timestamp.now();
43          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(0)))));
44          assertThat(((List) exp.getValue()).size(), equalTo(1));
45  
46          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(10)))));
47          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(20)))));
48          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(30)))));
49          assertThat(((List) exp.getValue()).size(), equalTo(4));
50  
51          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(40)))));
52          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(50)))));
53          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(60)))));
54          assertThat(((List) exp.getValue()).size(), equalTo(7));
55  
56          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(115)))));
57          assertThat(((List) exp.getValue()).size(), equalTo(6));
58  
59          exp.writeValue("x", newVDouble(0.0, newTime(reference.plus(TimeDuration.ofMillis(155)))));
60          assertThat(((List) exp.getValue()).size(), equalTo(3));
61      }

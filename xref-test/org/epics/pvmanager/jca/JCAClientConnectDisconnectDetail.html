21  /**
22   * This class tests opening and closing the same channel multiple times
23   * to check whether any of the initial events are lost. This test is designed
24   * to give some debug information if problems appear.
25   * 
26   * @author carcassi
27   */
28  public class JCAClientConnectDisconnectDetail {
29      public static void main(String[] args) throws Exception {
30          JCADataSource jca = new JCADataSource();
31          PVManager.setDefaultDataSource(jca);
32          
33          String channelName = "TEST_1";
34  
35  
36          final AtomicInteger connectCount = new AtomicInteger(0);
37          final AtomicInteger valueNotificationCount = new AtomicInteger(0);
38          final AtomicInteger connectionNotificationCount = new AtomicInteger(0);
39          final AtomicInteger totalNotificationCount = new AtomicInteger(0);
40          
41          int countToPrint = 10;
42          while (true) {
43              final CountDownLatch latch = new CountDownLatch(1);
44              
45              final PVReader<?> pv = PVManager.read(channel(channelName))
46                          .readListener(new PVReaderListener<Object>() {
47                              @Override
48                              public void pvChanged(PVReaderEvent<Object> event) {
49                                  if (event.isConnectionChanged()) {
50                                      connectionNotificationCount.incrementAndGet();
51                                  }
52                                  if (event.isValueChanged()) {
53                                      valueNotificationCount.incrementAndGet();
54                                  }
55                                  totalNotificationCount.incrementAndGet();
56                                  if (event.getPvReader().isConnected() && event.getPvReader().getValue() != null) {
57                                      latch.countDown();
58                                  }
59                              }
60                          }).maxRate(ofHertz(100));
61              connectCount.incrementAndGet();
62              boolean notified = latch.await(500, TimeUnit.MILLISECONDS);
63              if (!notified) {
64                  System.out.println("Connection: " + pv.isConnected() + " Value: " + pv.getValue());
65                  System.out.println(jca.getChannels().get(channelName).getProperties());
66              }
67              pv.close();
68              
69              if (connectCount.get() == countToPrint) {
70                  System.out.println("Connections: " + connectCount + " Conn notifications: " + connectionNotificationCount 
71                          + " Value notifications: " + valueNotificationCount + " Total notifications: " +totalNotificationCount);
72                  countToPrint += 10;
73              }
74          }
75      }
76  }

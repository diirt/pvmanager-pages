17  public class DataUtilsTest {
18  
19      @Test
20      public void timestampOf1() {
21          Timestamp timestamp = DataUtils.timestampOf(new TimeStamp(0, 1234567890));
22          assertThat(timestamp, equalTo(Timestamp.of(DataUtils.TS_EPOCH_SEC_PAST_1970 + 1, 234567890)));
23      }
24  
25      @Test
26      public void isTimeValid1() {
27          TimeStamp timeStamp = new TimeStamp(123456789, 1234567890);
28          assertThat(DataUtils.isTimeValid(timeStamp), equalTo(false));
29      }
30  
31      @Test
32      public void isTimeValid2() {
33          TimeStamp timeStamp = new TimeStamp(123456789, 123456789);
34          assertThat(DataUtils.isTimeValid(timeStamp), equalTo(true));
35      }

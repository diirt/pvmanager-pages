60      public void compareJson(JsonObject json, String text) {
61          StringWriter writer = new StringWriter();
62          JsonWriter jsonWriter = Json.createWriter(writer);
63          jsonWriter.writeObject(json);
64          assertThat(writer.toString(), equalTo(text));
65      }
66      
67      public void compareVType(VType expected, VType actual) {
68          assertThat("Type mismatch", VTypeValueEquals.typeEquals(actual, expected), equalTo(true));
69          assertThat("Value mismatch", VTypeValueEquals.valueEquals(actual, expected), equalTo(true));
70          if (expected instanceof Alarm) {
71              assertThat("Alarm mismatch", VTypeValueEquals.alarmEquals((Alarm) actual, (Alarm) expected), equalTo(true));
72          }
73          if (expected instanceof Time) {
74              assertThat("Time mismatch", VTypeValueEquals.timeEquals((Time) actual, (Time) expected), equalTo(true));
75          }
76      }

78      public JsonObject parseJson(String json) {
79          try (JsonReader reader = Json.createReader(new StringReader(json))) {
80              return reader.readObject();
81          }
82      }

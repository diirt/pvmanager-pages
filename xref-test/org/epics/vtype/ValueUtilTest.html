150     @Test
151     public void highestSeverityOf1() {
152         Alarm none = ValueFactory.alarmNone();
153         Alarm minor = ValueFactory.newAlarm(AlarmSeverity.MINOR, "Minor alarm");
154         Alarm otherMinor = ValueFactory.newAlarm(AlarmSeverity.MINOR, "Other minor alarm");
155         Alarm major = ValueFactory.newAlarm(AlarmSeverity.MAJOR, "Major alarm");
156         Alarm invalid = ValueFactory.newAlarm(AlarmSeverity.INVALID, "Invalid alarm");
157         Alarm undefined = ValueFactory.newAlarm(AlarmSeverity.UNDEFINED, "Undefined alarm");
158         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(none, minor), false), sameInstance(minor));
159         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(none, minor, otherMinor), false), sameInstance(minor));
160         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(null, minor, otherMinor), false), sameInstance(minor));
161         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(null, minor, otherMinor), true).getAlarmSeverity(), sameInstance(AlarmSeverity.UNDEFINED));
162         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(none, major, minor, otherMinor), false), sameInstance(major));
163         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(none, major, minor, otherMinor, invalid), false), sameInstance(invalid));
164         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(none, major, minor, undefined, invalid), false), sameInstance(undefined));
165         assertThat(ValueUtil.highestSeverityOf(Arrays.<Object>asList(none, major, minor, undefined, invalid, null), true), sameInstance(undefined));
166     }

168     @Test
169     public void latestTimeOf1() {
170         Time time1 = newTime(Timestamp.of(12340000, 0));
171         Time time2 = newTime(Timestamp.of(12340000, 0));
172         Time time3 = newTime(Timestamp.of(12350000, 0));
173         Time time4 = newTime(Timestamp.of(12360000, 0));
174         assertThat(ValueUtil.latestTimeOf(Arrays.<Object>asList(time1, time3)), sameInstance(time3));
175         assertThat(ValueUtil.latestTimeOf(Arrays.<Object>asList(time3, time1)), sameInstance(time3));
176         assertThat(ValueUtil.latestTimeOf(Arrays.<Object>asList(time1, time2)), sameInstance(time1));
177         assertThat(ValueUtil.latestTimeOf(Arrays.<Object>asList(time2, time1)), sameInstance(time2));
178         assertThat(ValueUtil.latestTimeOf(Arrays.<Object>asList(time1, time4, time2, time3)), sameInstance(time4));
179     }

181     @Test
182     public void latestValidTimeOrNewOf1() {
183         Time time1 = newTime(Timestamp.of(12340000, 0));
184         Time time2 = newTime(Timestamp.of(12340000, 0));
185         Time time3 = newTime(Timestamp.of(12350000, 0));
186         Time time4 = newTime(Timestamp.of(12360000, 0));
187         Time time5 = newTime(Timestamp.of(12370000, 0), 1, false);
188         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(time1, time3)), sameInstance(time3));
189         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(time3, time1)), sameInstance(time3));
190         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(time1, time2)), sameInstance(time1));
191         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(time2, time1)), sameInstance(time2));
192         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(time1, time4, time5, time3)), sameInstance(time4));
193         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(time5)), not(sameInstance(time5)));
194         assertThat(ValueUtil.latestValidTimeOrNowOf(Arrays.<Object>asList(null, null)), not(nullValue()));
195     }

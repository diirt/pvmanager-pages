26      @Test
27      public void toStringVNumber() {
28          assertThat(VTypeToString.toString(newVDouble(3.0, alarmNone(), newTime(Timestamp.of(1234567, 123000000)), displayNone())),
29                  equalTo("VDouble[3.0, 1970/01/15 01:56:07.123]"));
30          assertThat(VTypeToString.toString(newVDouble(133.0, newAlarm(AlarmSeverity.MINOR, "LOW"), newTime(Timestamp.of(1234567, 123000000)), displayNone())),
31                  equalTo("VDouble[133.0, MINOR(LOW), 1970/01/15 01:56:07.123]"));
32          assertThat(VTypeToString.toString(newVDouble(3.14, newAlarm(AlarmSeverity.MINOR, "HIGH"), newTime(Timestamp.of(1234567, 123000000)), displayNone())),
33                  equalTo("VDouble[3.14, MINOR(HIGH), 1970/01/15 01:56:07.123]"));
34          assertThat(VTypeToString.toString(newVInt(3, alarmNone(), newTime(Timestamp.of(1234567, 123000000)), displayNone())),
35                  equalTo("VInt[3, 1970/01/15 01:56:07.123]"));
36          assertThat(VTypeToString.toString(newVInt(4, newAlarm(AlarmSeverity.MINOR, "HIGH"), newTime(Timestamp.of(1234567, 123000000)), displayNone())),
37                  equalTo("VInt[4, MINOR(HIGH), 1970/01/15 01:56:07.123]"));
38      }

56      @Test

60          assertThat(VTypeToString.toString(newVDoubleArray(new ArrayDouble(1.0, 2.0, 3.0, 4.0), newAlarm(AlarmSeverity.MINOR, "HIGH"), newTime(Timestamp.of(1234567, 123000000)), displayNone())),
61                  equalTo("VDoubleArray[[1.0, 2.0, 3.0, ...], size 4, MINOR(HIGH), 1970/01/15 01:56:07.123]"));

89      public void toStringVTable() {
90          assertThat(VTypeToString.toString(newVTable(Arrays.<Class<?>>asList(String.class, double.class, int.class),
91                  Arrays.asList("Name", "Value", "Index"),
92                  Arrays.<Object>asList(Arrays.asList("A", "B", "C", "D"),
93                  new ArrayDouble(1.3,4.23,13,321.4),
94                  new ArrayInt(1,2,3,4)))),

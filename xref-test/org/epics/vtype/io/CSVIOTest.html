97                                     Arrays.asList("Name", "Value", "Index", "Time"),

106                 "\"C\" 2.34567891234E11 3 \"1969/12/31 19:02:15.0 -0500\"\n" +
107                 "\"D\" 1.23E-7 4 \"1969/12/31 19:02:16.0 -0500\"\n" +
108                 "\"E\" 123.0 5 \"1969/12/31 19:02:17.0 -0500\"\n");

111     public static void exportTest(CSVIO io, Object value, String csv) {
112         assertThat(io.canExport(value), equalTo(true));
113         StringWriter writer = new StringWriter();
114         io.export(value, writer);
115         assertThat(writer.toString(), equalTo(csv));
116     }

118     @Test
119     public void importVTable1() {
120         String inputText = "\"Name\" \"Value\" \"Index\"\n" +
121                 "\"A\" 0.234 1\n" +
122                 "\"B\" 1.456 2\n" +
123                 "\"C\" 2.34567891234E11 3\n" +
124                 "\"D\" 1.23E-7 4\n" +
125                 "\"E\" 123.0 5\n";
126         
127         CSVIO io = new CSVIO();
128         VTable value = io.importVTable(new StringReader(inputText));
129         assertThat(value.getColumnCount(), equalTo(3));
130         assertThat(value.getColumnName(0), equalTo("Name"));
131         assertThat(value.getColumnName(1), equalTo("Value"));
132         assertThat(value.getColumnName(2), equalTo("Index"));
133         assertThat((Object) value.getColumnType(0), equalTo((Object) String.class));
134         assertThat((Object) value.getColumnType(1), equalTo((Object) double.class));
135         assertThat((Object) value.getColumnType(2), equalTo((Object) double.class));
136         assertThat(value.getColumnData(0), equalTo((Object) Arrays.asList("A", "B", "C", "D", "E")));
137         assertThat(value.getColumnData(1), equalTo((Object) new ArrayDouble(0.234, 1.456, 234567891234.0, 0.000000123, 123)));
138         assertThat(value.getColumnData(2), equalTo((Object) new ArrayDouble(1,2,3,4,5)));
139     }
